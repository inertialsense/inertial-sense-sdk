49 bool cDeviceLogJSON::GetNextItemForFile()
50 {
51  m_jsonString.clear();
52  if (m_pFile == NULLPTR)
53  {
54  return false;
55  }
56  int stack = 0;
57  int pc = 0;
58  int i;
59  while ((i = m_pFile->getch()) != EOF)
60  {
61  char c = (char)i;
62  if (c == '{' && pc != '\\')
63  {
64  stack++;
65  }
66  else if (c == '}' && pc != '\\')
67  {
68  stack--;
69  if (stack == 0)
70  {
71  m_jsonString.append(1, c);
72  break;
73  }
74  }
75 
76  if (stack != 0)
77  {
78  m_jsonString.append(1, c);
79  }
80 
81  pc = c;
82  }
83  return (m_jsonString.size() != 0);
84 }
