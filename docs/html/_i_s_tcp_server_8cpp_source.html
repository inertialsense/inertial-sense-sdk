45 int cISTcpServer::Open(const string& ipAddress, int port)
46 {
47  m_ipAddress = ipAddress;
48  m_port = port;
49  int status;
50  char portString[64];
51  snprintf(portString, sizeof(portString), "%ld", (long)m_port);
52  addrinfo* result = NULL;
53  addrinfo hints = addrinfo();
54  hints.ai_family = AF_INET;
55  hints.ai_socktype = SOCK_STREAM;
56  hints.ai_protocol = IPPROTO_TCP;
57  hints.ai_flags = AI_PASSIVE;
58  status = getaddrinfo(m_ipAddress.length() == 0 ? NULL : m_ipAddress.c_str(), portString, &hints, &result);
59  if (status != 0)
60  {
61  Close();
62  return status;
63  }
64 
65  // setup socket
66  m_socket = socket(result->ai_family, result->ai_socktype, result->ai_protocol);
67  if (m_socket == 0)
68  {
69  freeaddrinfo(result);
70  Close();
71  return -1;
72  }
73 
74  int enable = 1;
75  if (setsockopt(m_socket, SOL_SOCKET, SO_REUSEADDR, (char*)&enable, sizeof(enable)) < 0)
76  {
77  freeaddrinfo(result);
78  Close();
79  return -1;
80  }
81 
82 #if 0 // Disable Nagles algorithm which identifies if multiple packets can be grouped and sent together
83  if (setsockopt(m_socket, IPPROTO_TCP, TCP_NODELAY, (const char *)&enable,sizeof(enable)) < 0)
84  {
85  freeaddrinfo(result);
86  Close();
87  return -1;
88  }
89 #endif
90 
91  // setup listener socket
92  status = ::bind(m_socket, result->ai_addr, (int)result->ai_addrlen);
93  if (status != 0)
94  {
95  freeaddrinfo(result);
96  Close();
97  return -1;
98  }
99 
100  freeaddrinfo(result);
101 
102  status = listen(m_socket, SOMAXCONN);
103  if (status != 0)
104  {
105  Close();
106  return -1;
107  }
108 
109  return status;
110 }

123 void cISTcpServer::Update()
124 {
125  uint8_t readBuff[8192];
126 
127  // accept new sockets
128  while (ISSocketCanRead(m_socket, 1))
129  {
130  if (m_delegate != NULLPTR)
131  {
132  m_delegate->OnClientConnecting(this);
133  }
134  socket_t socket = accept(m_socket, NULLPTR, NULLPTR);
135  if (socket != 0)
136  {
137  ISSocketSetBlocking(socket, false);
138  m_clients.push_back(socket);
139  if (m_delegate != NULLPTR)
140  {
141  m_delegate->OnClientConnected(this, socket);
142  }
143  }
144  else if (m_delegate != NULLPTR)
145  {
146  m_delegate->OnClientConnectFailed(this);
147  }
148  }
149 
150  for (size_t i = 0; i < m_clients.size(); i++)
151  {
152  if (ISSocketCanRead(m_clients[i], 1))
153  {
154  int count;
155  if ((count = ISSocketRead(m_clients[i], readBuff, sizeof(readBuff))) < 0)
156  {
157  // remove the client
158  if (m_delegate != NULLPTR)
159  {
160  m_delegate->OnClientDisconnected(this, m_clients[i]);
161  }
162  ISSocketClose(m_clients[i]);
163  m_clients.erase(m_clients.begin() + i--);
164  }
165  else if (count > 0 && m_delegate != NULLPTR)
166  {
167  m_delegate->OnClientDataReceived(this, m_clients[i], readBuff, count);
168  }
169  }
170  }
171 }

173 int cISTcpServer::Write(const uint8_t* data, int dataLength)
174 {
175  for (size_t i = 0; i < m_clients.size(); i++)
176  {
177  int written = 0;
178  int count;
179  while (written < dataLength)
180  {
181  count = ISSocketWrite(m_clients[i], data + written, dataLength - written);
182  if (count < 1)
183  {
184  // remove the client
185  if (m_delegate != NULLPTR)
186  {
187  m_delegate->OnClientDisconnected(this, m_clients[i]);
188  }
189  ISSocketClose(m_clients[i]);
190  m_clients.erase(m_clients.begin() + i--);
191  break;
192  }
193  else
194  {
195  written += count;
196  if (written == dataLength)
197  {
198  break;
199  }
200  }
201  }
202  }
203  return dataLength; // TODO: Maybe be smarter about detecting difference in bytes written for each client
