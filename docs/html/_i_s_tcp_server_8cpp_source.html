48int cISTcpServer::Open(const string& ipAddress, int port)
49{
50 m_ipAddress = ipAddress;
51 m_port = port;
52 int status;
53 char portString[64];
54 snprintf(portString, sizeof(portString), "%ld", (long)m_port);
55 addrinfo* result = NULL;
56 addrinfo hints = addrinfo();
57 hints.ai_family = AF_INET;
58 hints.ai_socktype = SOCK_STREAM;
59 hints.ai_protocol = IPPROTO_TCP;
60 hints.ai_flags = AI_PASSIVE;
61 status = getaddrinfo(m_ipAddress.length() == 0 ? NULL : m_ipAddress.c_str(), portString, &hints, &result);
62 if (status != 0)
63 {
64 Close();
65 return status;
66 }
67
68 // setup socket
69 m_socket = socket(result->ai_family, result->ai_socktype, result->ai_protocol);
70 if (m_socket == 0)
71 {
72 freeaddrinfo(result);
73 Close();
74 return -1;
75 }
76
77 int enable = 1;
78 if (setsockopt(m_socket, SOL_SOCKET, SO_REUSEADDR, (char*)&enable, sizeof(enable)) < 0)
79 {
80 freeaddrinfo(result);
81 Close();
82 return -1;
83 }
84
85#if 0 // Disable Nagles algorithm which identifies if multiple packets can be grouped and sent together
86 if (setsockopt(m_socket, IPPROTO_TCP, TCP_NODELAY, (const char *)&enable,sizeof(enable)) < 0)
87 {
88 freeaddrinfo(result);
89 Close();
90 return -1;
91 }
92#endif
93
94 // setup listener socket
95 status = ::bind(m_socket, result->ai_addr, (int)result->ai_addrlen);
96 if (status != 0)
97 {
98 freeaddrinfo(result);
99 Close();
100 return -1;
101 }
102
103 freeaddrinfo(result);
104
105 status = listen(m_socket, SOMAXCONN);
106 if (status != 0)
107 {
108 Close();
109 return -1;
110 }
111
112 return status;
113}

115int cISTcpServer::Close()
116{
117 int status = ISSocketClose(m_socket);
118 for (size_t i = 0; i < m_clients.size(); i++)
119 {
120 status |= ISSocketClose(m_clients[i]);
121 }
122 m_clients.clear();
123 return status;

126void cISTcpServer::Update()
127{
128 uint8_t readBuff[8192];
129
130 // accept new sockets
131 while (ISSocketCanRead(m_socket, 1))
132 {
133 if (m_delegate != NULLPTR)
134 {
135 m_delegate->OnClientConnecting(this);
136 }
137 socket_t socket = accept(m_socket, NULLPTR, NULLPTR);
138 if (socket != 0)
139 {
140 ISSocketSetBlocking(socket, false);
141 m_clients.push_back(socket);
142 if (m_delegate != NULLPTR)
143 {
144 m_delegate->OnClientConnected(this, socket);
145 }
146 }
147 else if (m_delegate != NULLPTR)
148 {
149 m_delegate->OnClientConnectFailed(this);
150 }
151 }
152
153 for (size_t i = 0; i < m_clients.size(); i++)
154 {
155 if (ISSocketCanRead(m_clients[i], 1))
156 {
157 int count;
158 if ((count = ISSocketRead(m_clients[i], readBuff, sizeof(readBuff))) < 0)
159 {
160 // remove the client
161 if (m_delegate != NULLPTR)
162 {
163 m_delegate->OnClientDisconnected(this, m_clients[i]);
164 }
165 ISSocketClose(m_clients[i]);
166 m_clients.erase(m_clients.begin() + i--);
167 }
168 else if (count > 0 && m_delegate != NULLPTR)
169 {
170 m_delegate->OnClientDataReceived(this, m_clients[i], readBuff, count);
171 }
172 }
173 }
174}

176int cISTcpServer::Write(const uint8_t* data, int dataLength)
177{
178 for (size_t i = 0; i < m_clients.size(); i++)
179 {
180 int written = 0;
181 int count;
182 while (written < dataLength)
183 {
184 count = ISSocketWrite(m_clients[i], data + written, dataLength - written);
185 if (count < 1)
186 {
187 // remove the client
188 if (m_delegate != NULLPTR)
189 {
190 m_delegate->OnClientDisconnected(this, m_clients[i]);
191 }
192 ISSocketClose(m_clients[i]);
193 m_clients.erase(m_clients.begin() + i--);
194 break;
195 }
196 else
197 {
198 written += count;
199 if (written == dataLength)
200 {
201 break;
202 }
203 }
204 }
205 }
206 return dataLength; // TODO: Maybe be smarter about detecting difference in bytes written for each client
