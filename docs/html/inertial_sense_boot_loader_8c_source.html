87static uint16_t crc_update(uint16_t crc_in, int incr)
88{
89 uint16_t crc = crc_in >> 15;
90 uint16_t out = crc_in << 1;
91
92 if (incr)
93 {
94 out++;
95 }
96
97 if (crc)
98 {
99 out ^= CRC_POLY;
100 }
101
102 return out;
103}

106static uint16_t crc16(uint8_t* data, uint16_t size)
107{
108 uint16_t crc, i;
109
110 for (crc = 0; size > 0; size--, data++)
111 {
112 for (i = 0x80; i; i >>= 1)
113 {
114 crc = crc_update(crc, *data & i);
115 }
116 }
117
118 for (i = 0; i < 16; i++)
119 {
120 crc = crc_update(crc, 0);
121 }
122
123 return crc;
124}

126static int xModemSend(serial_port_t* s, uint8_t* buf, size_t len)
127{
128 int ret;
129 uint8_t eot = X_EOT;
130 uint8_t answer;
131 xmodem_chunk_t chunk;
132
133 chunk.block = 1;
134 chunk.start = X_SOH;
135
136 // wait for receiver ping
137 do
138 {
139 if (serialPortRead(s, &answer, sizeof(uint8_t)) != sizeof(uint8_t))
140 {
141 return -1;
142 }
143 } while (answer != 'C');
144
145 // write up to one sector
146 while (len)
147 {
148 size_t z = 0;
149 int next = 0;
150// char status;
151
152 z = _MIN(len, sizeof(chunk.payload));
153 memcpy(chunk.payload, buf, z);
154 memset(chunk.payload + z, 0xff, sizeof(chunk.payload) - z);
155
156 chunk.crc = SWAP16(crc16(chunk.payload, sizeof(chunk.payload)));
157 chunk.block_neg = 0xff - chunk.block;
158
159 ret = serialPortWrite(s, (const uint8_t*)&chunk, sizeof(xmodem_chunk_t));
160 if (ret != sizeof(xmodem_chunk_t))
161 {
162 return -1;
163 }
164
165 ret = serialPortReadTimeout(s, &answer, sizeof(uint8_t), BOOTLOADER_TIMEOUT_DEFAULT);
166 if (ret != sizeof(uint8_t))
167 {
168 return -1;
169 }
170
171 switch (answer)
172 {
173 case X_NAK:
174// status = 'N';
175 break;
176 case X_ACK:
177// status = '.';
178 next = 1;
179 break;
180 case X_CAN:
181 return -1;
182 default:
183// status = '?';
184 break;
185 }
186
187 if (next)
188 {
189 chunk.block++;
190 len -= z;
191 buf += z;
192 }
193 }
194
195 ret = serialPortWrite(s, &eot, sizeof(uint8_t));
196 if (ret != sizeof(uint8_t))
197 {
198 return -1;
199 }
200 serialPortReadChar(s, &eot);
201 return 0;
202}

235// negotiate the bootloader version, once a 'U' character has been read, we read another character, timing out after 500 milliseconds
236// if nothing comes back, we are using version 1, otherwise the version is the number sent back
237static int bootloaderNegotiateVersion(bootloader_state_t* state)
238{
239 unsigned char v = 0;
240
241 do {
242 if (serialPortReadCharTimeout(state->param->port, &v, 500) == 0)
243 {
244 v = '1';
245 }
246 } while (v == 'U');
247
248 if (v == '1')
249 {
250 // version 1
251 state->version = 1;
252 state->firstPageSkipBytes = 8192;
253 }
254 else if (v >= '2' && v <= '5')
255 {
256 // version 2, 3 (which sent v2), 4
257 state->version = v-'0';
258 state->firstPageSkipBytes = 16384;
259 }
260 else
261 {
262 bootloader_perror(state->param->port, "Invalid version sent from bootloader: 0x%02X\n", (int)v);
263 return 0;
264 }
265
266#if PLATFORM_IS_WINDOWS
267
268 // EvalTool and multiple bootloads under Windows 10 have issues with dropped data if verify runs too fast
269 state->verifyChunkSize = 125;
270
271#else
272
273 state->verifyChunkSize = MAX_VERIFY_CHUNK_SIZE;
274
275#endif
276
277 return 1;
278}

298// calculate checksum - start is INCLUSIVE and end is EXCLUSIVE, if checkSumPosition is not 0, the checkSum is written to ptr + checkSumPosition
299static int bootloaderChecksum(int checkSum, unsigned char* ptr, int start, int end, int checkSumPosition, int finalCheckSum)
300{
301 unsigned char c1, c2;
302 unsigned char* currentPtr = (unsigned char*)(ptr + start);
303 unsigned char* endPtr = (unsigned char*)(ptr + end - 1);
304 unsigned char b;
305
306 while (currentPtr < endPtr)
307 {
308 c1 = *(currentPtr++) | 0x20;
309 c1 = (c1 <= '9' ? c1 + 0xD0 : c1 + 0xA9);
310 c2 = *(currentPtr++) | 0x20;
311 c2 = (c2 <= '9' ? c2 + 0xD0 : c2 + 0xA9);
312 b = (c1 << 4) | c2;
313 checkSum += b;
314 }
315
316 if (finalCheckSum)
317 {
318 checkSum = (unsigned char)(~checkSum + 1);
319 }
320 if (checkSumPosition != 0)
321 {
322 bootloader_snprintf((char*)(ptr + checkSumPosition), 3, "%.2X", checkSum);
323 }
324
325 return checkSum;
326}

370static int bootloaderBeginProgramForCurrentPage(serial_port_t* s, int startOffset, int endOffset)
371{
372 // Atmel begin program command is 0x01, different from standard intel hex where command 0x01 is end of file
373 // After the 0x01 is a 00 which means begin writing program
374 // The begin program command uses the current page and specifies two 16 bit addresses that specify where in the current page
375 // the program code will be written
376 unsigned char programPage[24];
377 bootloader_snprintf((char*)programPage, 24, ":0500000100%.4X%.4XCC", startOffset, endOffset);
378 bootloaderChecksum(0, programPage, 1, 19, 19, 1);
379
380 if (serialPortWriteAndWaitForTimeout(s, programPage, 21, (unsigned char*)".\r\n", 3, BOOTLOADER_TIMEOUT_DEFAULT) == 0)
381 {
382 bootloader_perror(s, "Failed to select offset %X to %X\n", startOffset, endOffset);
383 return 0;
384 }
385
386 return 1;
387}

422static int bootloaderUploadHexDataPage(serial_port_t* s, unsigned char* hexData, int byteCount, int* currentOffset, int* totalBytes, int* verifyCheckSum)
423{
424 int i;
425
426 if (byteCount == 0)
427 {
428 return 1;
429 }
430
431 // create a program request with just the hex characters that will fit on this page
432 unsigned char programLine[12];
433 bootloader_snprintf((char*)programLine, 12, ":%.2X%.4X00", byteCount, *currentOffset);
434 if (serialPortWrite(s, programLine, 9) != 9)
435 {
436 bootloader_perror(s, "Failed to write start page at offset %d\n", *currentOffset);
437 return 0;
438 }
439
440 // add the previously written chars to the checksum
441 int checkSum = bootloaderChecksum(0, programLine, 1, 9, 0, 0);
442
443 // write all of the hex chars
444 int charsForThisPage = byteCount * 2;
445 if (serialPortWrite(s, hexData, charsForThisPage) != charsForThisPage)
446 {
447 bootloader_perror(s, "Failed to write page data at offset %d\n", *currentOffset);
448 return 0;
449 }
450
451 // calculate verification checksum for this data
452 for (i = 0; i < charsForThisPage; i++)
453 {
454 *verifyCheckSum = ((*verifyCheckSum << 5) + *verifyCheckSum) + hexData[i];
455 }
456
457 checkSum = bootloaderChecksum(checkSum, hexData, 0, charsForThisPage, 0, 1);
458 unsigned char checkSumHex[3];
459 bootloader_snprintf((char*)checkSumHex, 3, "%.2X", checkSum);
460 if (serialPortWriteAndWaitForTimeout(s, checkSumHex, 2, (unsigned char*)".\r\n", 3, BOOTLOADER_TIMEOUT_DEFAULT) == 0)
461 {
462 bootloader_perror(s, "Failed to write checksum %s at offset %d\n", checkSumHex, *currentOffset);
463 return 0;
464 }
465
466 *totalBytes += byteCount;
467 *currentOffset += byteCount;
468
469 return 1;
470}

472static int bootloaderUploadHexData(serial_port_t* s, unsigned char* hexData, int charCount, int* currentOffset, int* currentPage, int* totalBytes, int* verifyCheckSum)
473{
474 if (charCount > MAX_SEND_COUNT)
475 {
476 bootloader_perror(s, "Unexpected char count of %d for page %d at offset %X\n", charCount, *currentPage, *currentOffset);
477 return 0;
478 }
479 else if (charCount == 0)
480 {
481 return 1;
482 }
483
484 int byteCount = charCount / 2;
485
486 // check if we will overrun the current page
487 if (*currentOffset + byteCount > FLASH_PAGE_SIZE)
488 {
489 int pageByteCount = FLASH_PAGE_SIZE - *currentOffset;
490 if (bootloaderUploadHexDataPage(s, hexData, pageByteCount, currentOffset, totalBytes, verifyCheckSum) == 0)
491 {
492 bootloader_perror(s, "Failed to upload %d bytes to page %d at offset %d\n", pageByteCount, *currentPage, *currentOffset);
493 return 0;
494 }
495 hexData += (pageByteCount * 2);
496 charCount -= (pageByteCount * 2);
497
498 // change to the next page
499 *currentOffset = 0;
500 (*currentPage)++;
501 if (!bootloaderSelectPage(s, *currentPage) || !bootloaderBeginProgramForCurrentPage(s, 0, FLASH_PAGE_SIZE - 1))
502 {
503 bootloader_perror(s, "Failure issuing select page command for upload\n");
504 return 0;
505 }
506 }
507
508 if (charCount != 0 && bootloaderUploadHexDataPage(s, hexData, charCount / 2, currentOffset, totalBytes, verifyCheckSum) == 0)
509 {
510 bootloader_perror(s, "Failed to upload %d bytes to page %d at offset %d\n", charCount / 2, *currentPage, *currentOffset);
511 return 0;
512 }
513
514 return 1;
515}

517static int bootloaderFillCurrentPage(serial_port_t* s, int* currentPage, int* currentOffset, int* totalBytes, int* verifyCheckSum)
518{
519 if (*currentOffset < FLASH_PAGE_SIZE)
520 {
521 unsigned char hexData[256];
522 memset(hexData, 'F', 256);
523
524 while (*currentOffset < FLASH_PAGE_SIZE)
525 {
526 int byteCount = (FLASH_PAGE_SIZE - *currentOffset) * 2;
527 if (byteCount > 256)
528 {
529 byteCount = 256;
530 }
531 memset(hexData, 'F', byteCount);
532
533 if (bootloaderUploadHexDataPage(s, hexData, byteCount / 2, currentOffset, totalBytes, verifyCheckSum) == 0)
534 {
535 bootloader_perror(s, "Failed to fill page %d with %d bytes at offset %d\n", *currentPage, byteCount, *currentOffset);
536 return 0;
537 }
538 }
539 }
540
541 return 1;
542}

562static int bootloaderVerify(int lastPage, int checkSum, bootloader_state_t* state)
563{
564 int verifyChunkSize = state->verifyChunkSize;
565 int chunkSize = bootloader_min(FLASH_PAGE_SIZE, verifyChunkSize);
566 int realCheckSum = 5381;
567 int totalCharCount = state->firstPageSkipBytes * 2;
568 int grandTotalCharCount = (lastPage + 1) * FLASH_PAGE_SIZE * 2; // char count
569 int i, pageOffset, readCount, actualPageOffset, pageChars, chunkIndex, lines;
570 int verifyByte = -1;
571 unsigned char chunkBuffer[(MAX_VERIFY_CHUNK_SIZE * 2) + 64]; // extra space for overhead
572 float percent = 0.0f;
573 unsigned char c=0;
574 FILE* verifyFile = 0;
575
576 if (state->param->verifyFileName != 0)
577 {
578
579#ifdef _MSC_VER
580
581 fopen_s(&verifyFile, state->param->verifyFileName, "wb");
582
583#else
584
585 verifyFile = fopen(state->param->verifyFileName, "wb");
586
587#endif
588
589 }
590
591 for (i = 0; i <= lastPage; i++)
592 {
593 if (bootloaderSelectPage(state->param->port, i) == 0)
594 {
595 bootloader_perror(state->param->port, "Failure issuing select page command for verify\n");
596 return 0;
597 }
598 pageOffset = (i == 0 ? state->firstPageSkipBytes : 0);
599 while (pageOffset < FLASH_PAGE_SIZE)
600 {
601 readCount = bootloader_min(chunkSize, FLASH_PAGE_SIZE - pageOffset);
602
603 // range is inclusive on the uINS, so subtract one
604 if (bootloaderDownloadData(state->param->port, pageOffset, pageOffset + readCount - 1) == 0)
605 {
606 bootloader_perror(state->param->port, "Failure issuing download data command\n");
607 return 0;
608 }
609
610 // each line has 7 overhead bytes, plus two bytes (hex) for each byte on the page and max 255 bytes per line
611 lines = (int)ceilf((float)readCount / 255.0f);
612 readCount = serialPortReadTimeout(state->param->port, chunkBuffer, (7 * lines) + (readCount * 2), BOOTLOADER_TIMEOUT_DEFAULT);
613 chunkIndex = 0;
614
615 while (chunkIndex < readCount)
616 {
617 if (chunkIndex > readCount - 5)
618 {
619 bootloader_perror(state->param->port, "Unexpected start line during validation\n");
620 return 0;
621 }
622
623 // skip the first 5 chars, they are simply ####=
624 if (chunkBuffer[chunkIndex] == 'X')
625 {
626 bootloader_perror(state->param->port, "Invalid checksum during validation\n");
627 return 0;
628 }
629 else if (chunkBuffer[chunkIndex += 4] != '=')
630 {
631 bootloader_perror(state->param->port, "Unexpected start line during validation\n");
632 return 0;
633 }
634 chunkBuffer[chunkIndex] = '\0';
635 actualPageOffset = strtol((char*)(chunkBuffer + chunkIndex - 4), 0, 16);
636 if (actualPageOffset != pageOffset)
637 {
638 bootloader_perror(state->param->port, "Unexpected offset during validation\n");
639 return 0;
640 }
641 pageChars = 0;
642 chunkIndex++;
643 while (chunkIndex < readCount)
644 {
645 c = chunkBuffer[chunkIndex++];
646 if ((c >= '0' && c <= '9') || (c >= 'A' && c <= 'F'))
647 {
648 pageChars++;
649 totalCharCount++;
650 realCheckSum = ((realCheckSum << 5) + realCheckSum) + c;
651
652 if (verifyFile != 0)
653 {
654 if (verifyByte == -1)
655 {
656 verifyByte = ((c >= '0' && c <= '9') ? c - '0' : c - 'A' + 10) << 4;
657 }
658 else
659 {
660 verifyByte |= (c >= '0' && c <= '9') ? c - '0' : c - 'A' + 10;
661 fputc(verifyByte, verifyFile);
662 verifyByte = -1;
663 }
664 }
665 }
666 else if (c == '\r')
667 {
668 continue;
669 }
670 else if (c == '\n')
671 {
672 break;
673 }
674 else
675 {
676 bootloader_perror(state->param->port, "Unexpected hex data during validation: 0x%02x [%c]\n", c, c);
677 return 0;
678 }
679 }
680
681 if (c != '\n')
682 {
683 bootloader_perror(state->param->port, "Unexpected end line character found during validation: 0x%02x [%c]\n", c, c);
684 return 0;
685 }
686
687 // increment page offset
688 pageOffset += (pageChars / 2);
689
690 if (state->param->verifyProgress != 0)
691 {
692 percent = (float)totalCharCount / (float)grandTotalCharCount;
693 if (state->param->verifyProgress(state->param->obj, percent) == 0)
694 {
695 bootloader_perror(state->param->port, "Validate firmware cancelled\n");
696 return 0;
697 }
698 }
699 }
700 }
701 }
702
703 if (verifyFile != 0)
704 {
705 fclose(verifyFile);
706 }
707
708 if (realCheckSum != checkSum)
709 {
710 bootloader_perror(state->param->port, "Download checksum 0x%08x != calculated checksum 0x%08x\n", realCheckSum, checkSum);
711 return 0;
712 }
713
714 return 1;
715}

718static int bootloaderProcessHexFile(FILE* file, bootloader_state_t* state)
719{
720
721#define BUFFER_SIZE 1024
722
723 int currentPage = 0;
724 int currentOffset = state->firstPageSkipBytes;
725 int lastSubOffset = currentOffset;
726 int subOffset;
727 int totalBytes = state->firstPageSkipBytes;

729 int verifyCheckSum = 5381;

731 float percent = 0.0f;

733 unsigned char output[BUFFER_SIZE * 2]; // big enough to store an entire extra line of buffer if needed
734 unsigned char* outputPtr = output;
735 const unsigned char* outputPtrEnd = output + (BUFFER_SIZE * 2);
736 int outputSize;
737 int page;
738 int pad;
739 int fileSize;
740 unsigned char tmp[5];
741 int i;
742
743 fseek(file, 0, SEEK_END);
744 fileSize = ftell(file);
745 fseek(file, 0, SEEK_SET);
746
747 while ((lineLength = bootloaderReadLine(file, line)) != 0)
748 {
749 if (lineLength > 12 && line[7] == '0' && line[8] == '0')
750 {
751 if (lineLength > BUFFER_SIZE * 4)
752 {
753 bootloader_perror(state->param->port, "Line length of %d was too large\n", lineLength);
754 return -1;
755 }
756
757 // we need to know the offset that this line was supposed to be stored at so we can check if offsets are skipped
758 memcpy(tmp, line + 3, 4);
759 tmp[4] = '\0';
760 subOffset = strtol((char*)tmp, 0, 16);
761
762 // check if we skipped an offset, the intel hex file format can do this, in which case we need to make sure
763 // that the bytes that were skipped get set to something
764 if (subOffset > lastSubOffset)
765 {
766 // pad with FF bytes, this is an internal implementation detail to how the device stores unused memory
767 pad = (subOffset - lastSubOffset);
768 if (outputPtr + pad >= outputPtrEnd)
769 {
770 bootloader_perror(state->param->port, "FF padding overflowed output buffer\n");
771 return -1;
772 }
773
774 while (pad-- != 0)
775 {
776 *outputPtr++ = 'F';
777 *outputPtr++ = 'F';
778 }
779 }
780
781 // skip the first 9 chars which are not data, then take everything else minus the last two chars which are a checksum
782 // check for overflow
783 pad = lineLength - 11;
784 if (outputPtr + pad >= outputPtrEnd)
785 {
786 bootloader_perror(state->param->port, "Line data overflowed output buffer\n");
787 return -1;
788 }
789
790 for (i = 9; i < lineLength - 2; i++)
791 {
792 *outputPtr++ = line[i];
793 }
794
795 // set the end offset so we can check later for skipped offsets
796 lastSubOffset = subOffset + ((lineLength - 11) / 2);
797 outputSize = (int)(outputPtr - output);
798
799 // we try to send the most allowed by this hex file format
800 if (outputSize < MAX_SEND_COUNT)
801 {
802 // keep buffering
803 continue;
804 }
805
806 // upload this chunk
807 else if (!bootloaderUploadHexData(state->param->port, output, (outputSize > MAX_SEND_COUNT ? MAX_SEND_COUNT : outputSize), &currentOffset, &currentPage, &totalBytes, &verifyCheckSum))
808 {
809 return -1;
810 }
811
812 outputSize -= MAX_SEND_COUNT;
813
814 if (outputSize < 0 || outputSize > BUFFER_SIZE)
815 {
816 bootloader_perror(state->param->port, "Output size of %d was too large\n", outputSize);
817 return -1;
818 }
819 else if (outputSize > 0)
820 {
821 // move the left-over data to the beginning
822 memmove(output, output + MAX_SEND_COUNT, outputSize);
823 }
824
825 // reset output ptr back to the next chunk of data
826 outputPtr = output + outputSize;
827 }
828 else if (strncmp(line, ":020000048", 10) == 0 && strlen(line) >= 13)
829 {
830 memcpy(tmp, line + 10, 3);
831 tmp[3] = '\0';
832 page = strtol((char*)tmp, 0, 16);
833
834 if (page != 0)
835 {
836 // we found a change page command beyond the first, let's finish up this page and fill it to the max
837 lastSubOffset = 0;
838 outputSize = (int)(outputPtr - output);
839
840 if (outputSize < 0 || outputSize > BUFFER_SIZE)
841 {
842 bootloader_perror(state->param->port, "Output size of %d was too large\n", outputSize);
843 return -1;
844 }
845
846 // flush the remainder of data to the page
847 else if (bootloaderUploadHexData(state->param->port, output, outputSize, &currentOffset, &currentPage, &totalBytes, &verifyCheckSum) == 0)
848 {
849 return -1;
850 }
851
852 // fill the remainder of the current page, the next time that bytes try to be written the page will be automatically incremented
853 else if (bootloaderFillCurrentPage(state->param->port, &currentPage, &currentOffset, &totalBytes, &verifyCheckSum) == 0)
854 {
855 return -1;
856 }
857
858 // set the output ptr back to the beginning, no more data is in the queue
859 outputPtr = output;
860 }
861 }
862
863 if (state->param->uploadProgress != 0)
864 {
865 percent = (float)ftell(file) / (float)fileSize; // Dummy line to call ftell() once
866 percent = (float)ftell(file) / (float)fileSize;
867 if (state->param->uploadProgress(state->param->obj, percent) == 0)
868 {
869 bootloader_perror(state->param->port, "Upload firmware cancelled\n");
870 return -1;
871 }
872 }
873 }
874
875 // upload any left over data
876 outputSize = (int)(outputPtr - output);
877 if (bootloaderUploadHexData(state->param->port, output, outputSize, &currentOffset, &currentPage, &totalBytes, &verifyCheckSum) == 0)
878 {
879 return -1;
880 }
881
882 // pad the remainder of the page with fill bytes
883 if (currentOffset != 0 && bootloaderFillCurrentPage(state->param->port, &currentPage, &currentOffset, &totalBytes, &verifyCheckSum) == 0)
884 {
885 return -1;
886 }
887
888 if (state->param->uploadProgress != 0 && percent != 1.0f)
889 {
890 state->param->uploadProgress(state->param->obj, 1.0f);
891 }
892
893 if (state->param->flags.bitFields.enableVerify && state->param->verifyProgress != 0)
894 {
895 if (state->param->statusText)
896 state->param->statusText(state->param->obj, "Verifying flash...");
897
898 if (state->param->verifyProgress != 0 && bootloaderVerify(currentPage, verifyCheckSum, state) == 0)
899 {
900 return -1;
901 }
902 }
903
904 // send the "reboot to program mode" command and the device should start in program mode
905 if(state->param->statusText)
906 state->param->statusText(state->param->obj, "Rebooting unit...");
907 serialPortWrite(state->param->port, (unsigned char*)":020000040300F7", 15);
908 serialPortSleep(state->param->port, 250);
909
910 return 0; // Success
911}

913static int bootloaderProcessBinFile(FILE* file, bootload_params_t* p)
914{
915 bootloader_state_t state;
916 memset(&state, 0, sizeof(state));
917 unsigned char c;
918 int dataLength;
919 int verifyCheckSum;
920 int lastPage;
921 int commandCount;
922 int commandLength;
923 unsigned char buf[16];
924 unsigned char commandType;
925
926 float percent = 0.0f;
927 fseek(file, 0, SEEK_END);
928 int fileSize = ftell(file);
929 fseek(file, 0, SEEK_SET);
930
931 // verify checksum is first four bytes
932 verifyCheckSum = fgetc(file) | (fgetc(file) << 8) | (fgetc(file) << 16) | (fgetc(file) << 24);
933
934 // last page is next two bytes
935 lastPage = fgetc(file) | (fgetc(file) << 8);
936
937 // the file is formatted like this:
938 // 4 bytes : verify checksum
939 // 2 bytes : last page
940 // LOOP:
941 // 1 byte - command type (0 = data, 1 = raw)
942 // 2 bytes - number of commands
943 // n commands...
944 // Data command (0)
945 // 2 bytes : Offset
946 // 1 byte : Data Length
947 // n bytes : Data
948 // 1 byte : Checksum
949 // Raw command (1)
950 // 1 byte : Command length
951 // n bytes : Command characters
952 // 1 byte : wait for character, '\0' for none
953 // 1 byte : sleep time after command is sent, in 250 millisecond intervals, 0 for no sleep
954 // 1 byte : wait for character timeout time in 250 millisecond intervals, 0 for no wait
955 // if not EOF, go to LOOP
956
957 while (ftell(file) != fileSize)
958 {
959 commandType = (unsigned char)fgetc(file);
960 commandCount = fgetc(file) | (fgetc(file) << 8);
961
962 if (commandType == 0)
963 {
964 while (commandCount-- > 0)
965 {
966 // write the :, data length, offset and 00 (data command)
967 commandLength = (fgetc(file) | (fgetc(file) << 8));
968 dataLength = fgetc(file); // data length
969 bootloader_snprintf((char*)buf, 16, ":%.2X%.4X00", dataLength, commandLength);
970 serialPortWrite(p->port, buf, 9);
971
972 // write the data - we use the fact that the checksum byte is after the data bytes to loop it in with this loop
973 while (dataLength-- > -1)
974 {
975 c = (unsigned char)fgetc(file);
976 buf[0] = hexLookupTable[(c >> 4) & 0x0F];
977 buf[1] = hexLookupTable[(c & 0x0F)];
978 serialPortWrite(p->port, buf, 2);
979 }
980
981 serialPortWaitForTimeout(p->port, (unsigned char*)".\r\n", 3, BOOTLOADER_TIMEOUT_DEFAULT);
982
983 if (p->uploadProgress != 0)
984 {
985 percent = (float)ftell(file) / (float)fileSize;
986 p->uploadProgress(p->obj, percent);
987 }
988 }
989 }
990 else if (commandType == 1)
991 {
992 // raw command, send it straight to the serial port as is
993 while (commandCount-- > 0)
994 {
995 commandLength = fgetc(file);
996 c = (unsigned char)fgetc(file);
997
998 // handshake char, ignore
999 if (commandLength == 1 && c == 'U')
1000 {
1001 // read sleep interval and timeout interval, ignored
1002 c = (unsigned char)fgetc(file);
1003 c = (unsigned char)fgetc(file);
1004 continue;
1005 }
1006
1007 // write command to serial port
1008 serialPortWrite(p->port, &c, 1);
1009 while (--commandLength > 0)
1010 {
1011 c = (unsigned char)fgetc(file);
1012 serialPortWrite(p->port, &c, 1);
1013 }
1014
1015 // read sleep interval and sleep
1016 c = (unsigned char)fgetc(file);
1017 commandLength = fgetc(file) * 250;
1018 if (commandLength != 0)
1019 {
1020 serialPortSleep(p->port, commandLength);
1021 }
1022
1023 // read timeout interval
1024 commandLength = fgetc(file) * 250;
1025 if (commandLength != 0)
1026 {
1027 unsigned char waitFor[4];
1028 bootloader_snprintf((char*)waitFor, 4, "%c\r\n", c);
1029 serialPortWaitForTimeout(p->port, waitFor, 3, commandLength);
1030 }
1031 }
1032
1033 if (p->uploadProgress != 0)
1034 {
1035 percent = (float)ftell(file) / (float)fileSize;
1036 p->uploadProgress(p->obj, percent);
1037 }
1038 }
1039 else
1040 {
1041 bootloader_perror(p->port, "Invalid data in .bin file\n");
1042 return 0;
1043 }
1044 }
1045
1046 if (p->uploadProgress != 0 && percent != 1.0f)
1047 {
1048 p->uploadProgress(p->obj, 1.0f);
1049 }
1050
1051 // re-purpose this variable for the return code
1052 dataLength = 1;
1053
1054 if (state.param->flags.bitFields.enableVerify && state.param->verifyProgress != 0)
1055 {
1056 dataLength = bootloaderVerify(lastPage, verifyCheckSum, &state);
1057 }
1058
1059 if (dataLength == 1)
1060 {
1061 // send the "reboot to program mode" command and the device should start in program mode
1062 serialPortWrite(state.param->port, (unsigned char*)":020000040300F7", 15);
1063 }
1064
1065 serialPortSleep(state.param->port, 250);
1066
1067 return dataLength;
1068}

1205{

1207 p->statusText(p->obj, "Starting bootloader...");

1210 //If we have an error, exit
1211 if (p->error[0] != 0)
1212 return -1;
1213
1214 if (p->statusText)
1215 p->statusText(p->obj, "Unable to find bootloader.");

1217 if (p->bootName != 0)
1218 {

1220 p->statusText(p->obj, "Attempting to reload bootloader...");
1221

1231 params.statusText = p->statusText;

1234
1235 if (!bootloadUpdateBootloaderSendFile(&params))
1236 return -1;
1237
1238 SLEEP_MS(1000);

1240 return 1;

1242

1277 p->statusText(p->obj, str);

1319 params.statusText = p->statusText;
1320 params.numberOfDevices = 1;
1321 params.flags.bitFields.enableVerify = 1;
1322

1324 return -1;
1325
1326 SLEEP_MS(1000);
1327 //Bootloader updated successfully. Return 1 to indicate firmware update is now needed.
1328 return 1;
1329 }
1330 else
1331 { //Bootloader is out of date but SAM-BA is not supported on port
1332 if (p->statusText)
1333 p->statusText(p->obj, "WARNING! Bootloader is out of date. Port used to communicate with bootloader does not support updating the bootloader. To force firmware update, run update without including the bootloader file.");
1334 bootloader_snprintf(p->error, BOOTLOADER_ERROR_LENGTH, "Bootloader is out of date but current port does not support updating bootloader. To force update, run firmware update without including the bootloader file.\n");
1335 return -1;
1336 }
1337 }
1338 }
1339 }
1340
1341 if (p->statusText)
1342 p->statusText(p->obj, "Erasing flash...");
1343 if ( !bootloaderEraseFlash(p->port) /*erase all flash */ || !bootloaderSelectPage(p->port, 0) /*select the first page*/)

1347 p->statusText(p->obj, "Programming flash...");
1348 // begin programming the first page
1349 if(!bootloaderBeginProgramForCurrentPage(p->port, state.firstPageSkipBytes, FLASH_PAGE_SIZE - 1))
1350 return -1;
1351
1352 return bootloaderProcessHexFile(file, &state);
1353 }
