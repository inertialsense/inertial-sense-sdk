536static __inline void lpf_alpha_beta( float dt, float cornerFreqHz, float *alpha, float *beta )
537{
538 float dc = dt * cornerFreqHz;
539 *alpha = dc / (1.0f + dc);
540 *beta = 1.0f - *alpha;
541}

544// LPF zero order: val = beta*lastVal + alpha*input
545#define O0_LP_FILTER(val,input,alph,beta) (val = (((beta)*(val)) + ((alph)*(input))))
546
547// LPF first order model coefficient: c = beta*c + alph*((input - val) / dt) (dt long)
548// LPF input into state estimate: val = beta*((val) + c*dt) + alph*input (dt short)
549#define O1_LP_FILTER(val,input,alph,beta,c,dt) { c = beta*c + alph*((input-val)/dt); val = beta*(val + c*dt) + alph*input; }

552// Propagate filter estimate when no input is available
553#define O1X_LP_FILTER_NO_INPUT(val,input,c,dt) { val = val + c*dt; } // (shorter dt)

556// LPF input into state estimate: val = val2 = beta*((val2) + c*dt2) + alph*input
557#define O1X_LP_FILTER(val,val2,input,alph,beta,c,dt2) { c = beta*c + alph*((input-val2)/dt2); val = val2 = beta*(val2 + c*dt2) + alph*input; }
