70 bool read_did_argument(stream_did_t *dataset, string s)
71 {
72  // Try to use DID number
73  eDataIDs did = strtol(s.c_str(), NULL, 10);
74 
75  std::string::size_type pos = s.find('=');
76 
77  if (did <= DID_NULL || did >= DID_COUNT)
78  { // Number is invalid. Use DID name.
79  string name = s;
80  if (pos != std::string::npos)
81  { // Remove equal sign
82  name = s.substr(0, pos);
83  }
84 
85  did = cISDataMappings::GetDataSetId(name);
86  }
87 
88  if (did > DID_NULL && did < DID_COUNT)
89  { // DID is valid
90  dataset->did = did;
91  dataset->periodMultiple = 1;
92 
93  if (pos != std::string::npos)
94  { // Contains '='
95  string m = s.substr(pos + 1);
96 
97  if (m.find(" ") == std::string::npos)
98  { // Found period multiple following equal sign (i.e. DID=periodMultiple: 4=4 or DID_INS_1=4)
99  dataset->periodMultiple = strtol(m.c_str(), NULL, 10);
100  }
101  }
102 
103  return true;
104  }
105 
106  return false;
107 }

109 bool cltool_parseCommandLine(int argc, char* argv[])

111  // set defaults
112  g_commandLineOptions.baudRate = CL_DEFAULT_BAUD_RATE;
113  g_commandLineOptions.comPort = CL_DEFAULT_COM_PORT;
114  g_commandLineOptions.displayMode = CL_DEFAULT_DISPLAY_MODE;

116  g_commandLineOptions.enableLogging = CL_DEFAULT_ENABLE_LOGGING;
117  g_commandLineOptions.logType = CL_DEFAULT_LOG_TYPE;
118  g_commandLineOptions.logPath = CL_DEFAULT_LOGS_DIRECTORY;
119  g_commandLineOptions.logSubFolder = cISLogger::CreateCurrentTimestamp();
120  g_commandLineOptions.maxLogFileSize = CL_DEFAULT_MAX_LOG_FILE_SIZE;
121  g_commandLineOptions.maxLogSpacePercent = CL_DEFAULT_MAX_LOG_SPACE_PERCENT;
122  g_commandLineOptions.replaySpeed = CL_DEFAULT_REPLAY_SPEED;
123  g_commandLineOptions.bootloaderVerify = CL_DEFAULT_BOOTLOAD_VERIFY;

128  g_commandLineOptions.surveyIn.state = 0;

130  g_commandLineOptions.surveyIn.minAccuracy = 0;
131  g_commandLineOptions.outputOnceDid = 0;
132 
133 
134  if(argc <= 1)
135  { // Display usage menu if no options are provided
136  cltool_outputUsage();
137  return false;
138  }
139 
140  // printf("Arguments: %d\n\n%s\n\n", argc, argv[0]);
141 
142  // parse command line. Keep these options in alphabetic order!
143  for (int i = 1; i < argc; i++)
144  {
145  const char* a = argv[i];
146  if (startsWith(a, "--"))
147  { // Treat "--" same as "-"
148  a++;
149  }
150 
151  if (startsWith(a, "-asciiMessages="))
152  {
153  g_commandLineOptions.asciiMessages = &a[15];
154  }
155  else if (startsWith(a, "-base="))
156  {
157  g_commandLineOptions.baseConnection = &a[6];
158  }
159  else if (startsWith(a, "-baud="))
160  {
161  g_commandLineOptions.baudRate = strtol(&a[6], NULL, 10);
162  }
163  else if (startsWith(a, "-chipEraseEvb"))
164  {
165  g_commandLineOptions.chipEraseEvb2 = true;
166  }
167  else if (startsWith(a, "-chipEraseUins"))
168  {
169  g_commandLineOptions.chipEraseUins = true;
170  }
171  else if (matches(a, "-c") && (i + 1) < argc)
172  {
173  g_commandLineOptions.comPort = argv[++i]; // use next argument
174  }
175  else if (startsWith(a, "-dboc"))
176  {
177  g_commandLineOptions.disableBroadcastsOnClose = true;
178  }
179  else if (startsWith(a, "-dids"))
180  {
181 #if defined(INCLUDE_LUNA_DATA_SETS)
182  for (eDataIDs id = 0; id < DID_COUNT; id++)
183 #else
184  for (eDataIDs id = 0; id < DID_COUNT_UINS; id++)
185 #endif
186  {
187  printf("(%d) %s\n", id, cISDataMappings::GetDataSetName(id));
188  }
189  cltool_outputHelp();
190  return false;
191  }
192  else if (startsWith(a, "-did") && (i + 1) < argc)
193  {
194  while ((i + 1) < argc && !startsWith(argv[i + 1], "-")) // next argument doesn't start with "-"
195  {
196  if (g_commandLineOptions.outputOnceDid)
197  {
198  i++;
199  }
200  else
201  {
202  stream_did_t dataset = {};
203  if (read_did_argument(&dataset, argv[++i])) // use next argument
204  {
205  if (dataset.periodMultiple == 0)
206  {
207  g_commandLineOptions.outputOnceDid = dataset.did;
208  g_commandLineOptions.datasets.clear();
209  }
210  g_commandLineOptions.datasets.push_back(dataset);
211  }
212  }
213  }
214  }
215  else if (startsWith(a, "-edit") && (i + 1) < argc)
216  {
217  stream_did_t dataset = {};
218  if (read_did_argument(&dataset, argv[++i])) // use next argument
219  {
220  g_commandLineOptions.datasetEdit = dataset;
221  }
222  }
223  else if (startsWith(a, "-evbFlashCfg="))
224  {
225  g_commandLineOptions.evbFlashCfg = &a[13];
226  }
227  else if (startsWith(a, "-evbFlashCfg"))
228  {
229  g_commandLineOptions.evbFlashCfg = ".";
230  }
231  else if (startsWith(a, "-factoryReset"))
232  {
233  g_commandLineOptions.factoryResetUins = true;
234  }
235  else if (startsWith(a, "-fb"))
236  {
237  g_commandLineOptions.forceBootloaderUpdate = true;
238  }
239  else if (startsWith(a, "-flashCfg="))
240  {
241  g_commandLineOptions.flashCfg = &a[10];
242  }
243  else if (startsWith(a, "-flashCfg"))
244  {
245  g_commandLineOptions.flashCfg = ".";
246  }
247  else if (startsWith(a, "-h") || startsWith(a, "--h") || startsWith(a, "-help") || startsWith(a, "--help"))
248  {
249  cltool_outputUsage();
250  return false;
251  }
252  else if (startsWith(a, "-lms="))
253  {
254  g_commandLineOptions.maxLogSpacePercent = (float)atof(&a[5]);
255  }
256  else if (startsWith(a, "-lmf="))
257  {
258  g_commandLineOptions.maxLogFileSize = (uint32_t)strtoul(&a[5], NULL, 10);
259  }
260  else if (startsWith(a, "-log-flush-timeout="))
261  {
262  g_commandLineOptions.timeoutFlushLoggerSeconds = strtoul(&a[19], NULLPTR, 10);
263  }
264  else if (startsWith(a, "-lts="))
265  {
266  const char* subFolder = &a[5];
267  if (*subFolder == '1' || startsWith(subFolder, "true"))
268  {
269  g_commandLineOptions.logSubFolder = cISLogger::CreateCurrentTimestamp();
270  }
271  else if (*subFolder == '\0' || *subFolder == '0' || startsWith(subFolder, "false"))
272  {
273  g_commandLineOptions.logSubFolder = cISLogger::g_emptyString;
274  }
275  else
276  {
277  g_commandLineOptions.logSubFolder = subFolder;
278  }
279  }
280  else if (startsWith(a, "-lp") && (i + 1) < argc)
281  {
282  g_commandLineOptions.logPath = argv[++i]; // use next argument;
283  }
284  else if (startsWith(a, "-lt="))
285  {
286  g_commandLineOptions.logType = &a[4];
287  }
288  else if (startsWith(a, "-lon"))
289  {
290  g_commandLineOptions.enableLogging = true;
291  }
292  else if (startsWith(a, "-magRecal"))
293  {
294  g_commandLineOptions.rmcPreset = 0;
295  g_commandLineOptions.magRecal = true;
296  g_commandLineOptions.magRecalMode = strtol(a + 9, NULL, 10);
297  }
298  else if (startsWith(a, "-presetPPD"))
299  {
300  g_commandLineOptions.rmcPreset = RMC_PRESET_PPD_GROUND_VEHICLE;
301  }
302  else if (startsWith(a, "-presetINS2"))
303  {
304  g_commandLineOptions.rmcPreset = RMC_PRESET_INS_BITS;
305  }
306  else if (startsWith(a, "-persistent"))
307  {
308  g_commandLineOptions.persistentMessages = true;
309  }
310  else if (startsWith(a, "-q"))
311  {
312  g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_QUIET;
313  }
314  else if (startsWith(a, "-rp") && (i + 1) < argc)
315  {
316  g_commandLineOptions.replayDataLog = true;
317  g_commandLineOptions.logPath = argv[++i]; // use next argument
318  }
319  else if (startsWith(a, "-rs="))
320  {
321  g_commandLineOptions.replayDataLog = true;
322  g_commandLineOptions.replaySpeed = (float)atof(&a[4]);
323  }
324  else if (startsWith(a, "-resetEvb"))
325  {
326  g_commandLineOptions.softwareResetEvb = true;
327  }
328  else if (startsWith(a, "-reset"))
329  {
330  g_commandLineOptions.softwareResetUins = true;
331  }
332  else if (startsWith(a, "-rover="))
333  {
334  g_commandLineOptions.roverConnection = &a[7];
335 
336  // DID_GPS1_POS must be enabled for NTRIP VRS to supply rover position.
337  stream_did_t dataset = {};
338  read_did_argument(&dataset, "DID_GPS1_POS");
339  g_commandLineOptions.datasets.push_back(dataset);
340  }
341  else if (startsWith(a, "-r"))
342  {
343  g_commandLineOptions.replayDataLog = true;
344  }
345  else if (startsWith(a, "-stats"))
346  {
347  g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_STATS;
348  }
349  else if (startsWith(a, "-survey="))
350  {
351  g_commandLineOptions.rmcPreset = 0;
352  g_commandLineOptions.surveyIn.state = strtol(a + 8, NULL, 10);
353  int maxDurationSec = strtol(a + 10, NULL, 10);
354  if (maxDurationSec > 5)
355  {
356  g_commandLineOptions.surveyIn.maxDurationSec = maxDurationSec;
357  }
358  }
359  else if (startsWith(a, "-s"))
360  {
361  g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_SCROLL;
362  }
363  else if (startsWith(a, "-ub") && (i + 1) < argc)
364  {
365  g_commandLineOptions.updateBootloaderFilename = argv[++i]; // use next argument
366  }
367  else if (startsWith(a, "-uf") && (i + 1) < argc)
368  {
369  g_commandLineOptions.updateAppFirmwareFilename = argv[++i]; // use next argument
370  }
371  else if (startsWith(a, "-uv"))
372  {
373  g_commandLineOptions.bootloaderVerify = true;
374  }
375  else
376  {
377  cout << "Unrecognized command line option: " << a << endl;
378  cltool_outputHelp();
379  return false;
380  }
381  }
382 
383  // We are either using a serial port or replaying data
384  if ((g_commandLineOptions.comPort.length() == 0) && !g_commandLineOptions.replayDataLog)
385  {
386  cltool_outputUsage();
387  return false;
388  }
389  else if (g_commandLineOptions.updateAppFirmwareFilename.length() != 0 && g_commandLineOptions.comPort.length() == 0)
390  {
391  cout << "Use COM_PORT option \"-c \" with bootloader" << endl;
392  return false;
393  }
394  else if (g_commandLineOptions.updateBootloaderFilename.length() != 0 && g_commandLineOptions.comPort.length() == 0)
395  {
396  cout << "Use COM_PORT option \"-c \" with bootloader" << endl;
397  return false;
398  }
399 
400  return true;
401 }

403 bool cltool_replayDataLog()
404 {
405  if (g_commandLineOptions.logPath.length() == 0)
406  {
407  cout << "Please specify the replay log path!" << endl;
408  return false;
409  }
410 
411  cISLogger logger;
412  if (!logger.LoadFromDirectory(g_commandLineOptions.logPath, cISLogger::ParseLogType(g_commandLineOptions.logType), { "ALL" }))
413  {
414  cout << "Failed to load log files: " << g_commandLineOptions.logPath << endl;
415  return false;
416  }
417 
418  cout << "Replaying log files: " << g_commandLineOptions.logPath << endl;
419  p_data_t *data;
420  while ((data = logger.ReadData()) != NULL)
421  {
422  g_inertialSenseDisplay.ProcessData(data, g_commandLineOptions.replayDataLog, g_commandLineOptions.replaySpeed);
423 
424 // if (data->hdr.id == DID_GPS1_RAW)
425 // {
426 // // Insert your code for processing data here
427 // }
428  }
429 
430  cout << "Done replaying log files: " << g_commandLineOptions.logPath << endl;
431  g_inertialSenseDisplay.Goodbye();
432  return true;
433 }

531 bool cltool_updateFlashCfg(InertialSense& inertialSenseInterface, string flashCfgString)
532 {
533  nvm_flash_cfg_t flashCfg = inertialSenseInterface.GetFlashConfig();
534  const map_name_to_info_t& flashMap = *cISDataMappings::GetMapInfo(DID_FLASH_CONFIG);
535 
536  if (flashCfgString.length() < 2)
537  {
538  // read flash config and display
539  data_mapping_string_t stringBuffer;
540  cout << "Current flash config" << endl;
541  for (map_name_to_info_t::const_iterator i = flashMap.begin(); i != flashMap.end(); i++)
542  {
543  if (cISDataMappings::DataToString(i->second, NULL, (const uint8_t*)&flashCfg, stringBuffer))
544  {
545  cout << i->second.name << " = " << stringBuffer << endl;
546  }
547  }
548  return false;
549  }
550  else
551  {
552  vector<string> keyValues;
553  splitString(flashCfgString, '|', keyValues);
554  for (size_t i = 0; i < keyValues.size(); i++)
555  {
556  vector<string> keyAndValue;
557  splitString(keyValues[i], '=', keyAndValue);
558  if (keyAndValue.size() == 2)
559  {
560  if (flashMap.find(keyAndValue[0]) == flashMap.end())
561  {
562  cout << "Unrecognized flash config key '" << keyAndValue[0] << "' specified, ignoring." << endl;
563  }
564  else
565  {
566  const data_info_t& info = flashMap.at(keyAndValue[0]);
567  int radix = (keyAndValue[1].compare(0, 2, "0x") == 0 ? 16 : 10);
568  int substrIndex = 2 * (radix == 16); // skip 0x for hex
569  const string& str = keyAndValue[1].substr(substrIndex);
570  cISDataMappings::StringToData(str.c_str(), (int)str.length(), NULL, (uint8_t*)&flashCfg, info, radix);
571  cout << "Updated flash config key '" << keyAndValue[0] << "' to '" << keyAndValue[1].c_str() << "'" << endl;
572  }
573  }
574  }
575  inertialSenseInterface.SetFlashConfig(flashCfg);
576  g_inertialSenseDisplay.Clear();
577  return true;
578  }
