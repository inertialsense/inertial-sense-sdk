51static bool startsWith(const char* str, const char* pre)
52{
53 size_t lenpre = strlen(pre), lenstr = strlen(str);
54 return lenstr < lenpre ? false : strncasecmp(pre, str, lenpre) == 0;
55}
56
57static bool matches(const char* str, const char* pre)
58{
59 size_t lenpre = strlen(pre), lenstr = strlen(str);
60 return lenstr != lenpre ? false : strncasecmp(pre, str, lenpre) == 0;
61}

74bool read_did_argument(stream_did_t *dataset, string s)
75{
76 // Try to use DID number
77 eDataIDs did = strtol(s.c_str(), NULL, 10);
78
79 std::string::size_type pos = s.find('=');
80
81 if (did <= DID_NULL || did >= DID_COUNT)
82 { // Number is invalid. Use DID name.
83 string name = s;
84 if (pos != std::string::npos)
85 { // Remove equal sign
86 name = s.substr(0, pos);
87 }
88
89 did = cISDataMappings::GetDataSetId(name);
90 }
91
92 if (did > DID_NULL && did < DID_COUNT)
93 { // DID is valid
94 dataset->did = did;
95 dataset->periodMultiple = 1;
96
97 if (pos != std::string::npos)
98 { // Contains '='
99 string m = s.substr(pos + 1);
100
101 if (m.find(" ") == std::string::npos)
102 { // Found period multiple following equal sign (i.e. DID=periodMultiple: 4=4 or DID_INS_1=4)
103 dataset->periodMultiple = strtol(m.c_str(), NULL, 10);
104 }
105 }
106
107 return true;
108 }
109
110 return false;
111}

113bool cltool_parseCommandLine(int argc, char* argv[])
114{
115 // set defaults
116 g_commandLineOptions.baudRate = CL_DEFAULT_BAUD_RATE;
117 g_commandLineOptions.comPort = CL_DEFAULT_COM_PORT;
118 g_commandLineOptions.displayMode = CL_DEFAULT_DISPLAY_MODE;
119 g_commandLineOptions.rmcPreset = 0;
120 g_commandLineOptions.enableLogging = CL_DEFAULT_ENABLE_LOGGING;
121 g_commandLineOptions.logType = CL_DEFAULT_LOG_TYPE;
122 g_commandLineOptions.logPath = CL_DEFAULT_LOGS_DIRECTORY;
123 g_commandLineOptions.logSubFolder = cISLogger::CreateCurrentTimestamp();
124 g_commandLineOptions.maxLogFileSize = CL_DEFAULT_MAX_LOG_FILE_SIZE;
125 g_commandLineOptions.maxLogSpacePercent = CL_DEFAULT_MAX_LOG_SPACE_PERCENT;
126 g_commandLineOptions.replaySpeed = CL_DEFAULT_REPLAY_SPEED;
127 g_commandLineOptions.bootloaderVerify = CL_DEFAULT_BOOTLOAD_VERIFY;
128 g_commandLineOptions.timeoutFlushLoggerSeconds = 3;
129 g_commandLineOptions.asciiMessages = "";
130 g_commandLineOptions.updateBootloaderFilename = "";
131
132 g_commandLineOptions.surveyIn.state = 0;
133 g_commandLineOptions.surveyIn.maxDurationSec = 15 * 60; // default survey of 15 minutes
134 g_commandLineOptions.surveyIn.minAccuracy = 0;
135 g_commandLineOptions.outputOnceDid = 0;
136
137
138 if(argc <= 1)
139 { // Display usage menu if no options are provided
140 cltool_outputUsage();
141 return false;
142 }
143
144 // printf("Arguments: %d\n\n%s\n\n", argc, argv[0]);
145
146 // parse command line. Keep these options in alphabetic order!
147 for (int i = 1; i < argc; i++)
148 {
149 const char* a = argv[i];
150 if (startsWith(a, "--"))
151 { // Treat "--" same as "-"
152 a++;
153 }
154
155 if (startsWith(a, "-asciiMessages="))
156 {
157 g_commandLineOptions.asciiMessages = &a[15];
158 }
159 else if (startsWith(a, "-base="))
160 {
161 g_commandLineOptions.baseConnection = &a[6];
162 }
163 else if (startsWith(a, "-baud="))
164 {
165 g_commandLineOptions.baudRate = strtol(&a[6], NULL, 10);
166 }
167 else if (startsWith(a, "-chipEraseEvb"))
168 {
169 g_commandLineOptions.chipEraseEvb2 = true;
170 }
171 else if (startsWith(a, "-chipEraseUins"))
172 {
173 g_commandLineOptions.chipEraseUins = true;
174 }
175 else if (matches(a, "-c") && (i + 1) < argc)
176 {
177 g_commandLineOptions.comPort = argv[++i]; // use next argument
178 }
179 else if (startsWith(a, "-dboc"))
180 {
181 g_commandLineOptions.disableBroadcastsOnClose = true;
182 }
183 else if (startsWith(a, "-dids"))
184 {
185#if defined(INCLUDE_LUNA_DATA_SETS)
186 for (eDataIDs id = 0; id < DID_COUNT; id++)
187#else
188 for (eDataIDs id = 0; id < DID_COUNT_UINS; id++)
189#endif
190 {
191 printf("(%d) %s\n", id, cISDataMappings::GetDataSetName(id));
192 }
193 cltool_outputHelp();
194 return false;
195 }
196 else if (startsWith(a, "-did") && (i + 1) < argc)
197 {
198 while ((i + 1) < argc && !startsWith(argv[i + 1], "-")) // next argument doesn't start with "-"
199 {
200 if (g_commandLineOptions.outputOnceDid)
201 {
202 i++;
203 }
204 else
205 {
206 stream_did_t dataset = {};
207 if (read_did_argument(&dataset, argv[++i])) // use next argument
208 {
209 if (dataset.periodMultiple == 0)
210 {
211 g_commandLineOptions.outputOnceDid = dataset.did;
212 g_commandLineOptions.datasets.clear();
213 }
214 g_commandLineOptions.datasets.push_back(dataset);
215 }
216 }
217 }
218 }
219 else if (startsWith(a, "-edit") && (i + 1) < argc)
220 {
221 stream_did_t dataset = {};
222 if (read_did_argument(&dataset, argv[++i])) // use next argument
223 {
224 g_commandLineOptions.datasetEdit = dataset;
225 }
226 }
227 else if (startsWith(a, "-evbFlashCfg="))
228 {
229 g_commandLineOptions.evbFlashCfg = &a[13];
230 }
231 else if (startsWith(a, "-evbFlashCfg"))
232 {
233 g_commandLineOptions.evbFlashCfg = ".";
234 }
235 else if (startsWith(a, "-evbReset"))
236 {
237 g_commandLineOptions.softwareResetEvb = true;
238 }
239 else if (startsWith(a, "-factoryReset"))
240 {
241 g_commandLineOptions.factoryResetUins = true;
242 }
243 else if (startsWith(a, "-fb"))
244 {
245 g_commandLineOptions.forceBootloaderUpdate = true;
246 }
247 else if (startsWith(a, "-flashCfg="))
248 {
249 g_commandLineOptions.flashCfg = &a[10];
250 }
251 else if (startsWith(a, "-flashCfg"))
252 {
253 g_commandLineOptions.flashCfg = ".";
254 }
255 else if (startsWith(a, "-h") || startsWith(a, "--h") || startsWith(a, "-help") || startsWith(a, "--help"))
256 {
257 cltool_outputUsage();
258 return false;
259 }
260 else if (startsWith(a, "-lms="))
261 {
262 g_commandLineOptions.maxLogSpacePercent = (float)atof(&a[5]);
263 }
264 else if (startsWith(a, "-lmf="))
265 {
266 g_commandLineOptions.maxLogFileSize = (uint32_t)strtoul(&a[5], NULL, 10);
267 }
268 else if (startsWith(a, "-log-flush-timeout="))
269 {
270 g_commandLineOptions.timeoutFlushLoggerSeconds = strtoul(&a[19], NULLPTR, 10);
271 }
272 else if (startsWith(a, "-lts="))
273 {
274 const char* subFolder = &a[5];
275 if (*subFolder == '1' || startsWith(subFolder, "true"))
276 {
277 g_commandLineOptions.logSubFolder = cISLogger::CreateCurrentTimestamp();
278 }
279 else if (*subFolder == '\0' || *subFolder == '0' || startsWith(subFolder, "false"))
280 {
281 g_commandLineOptions.logSubFolder = cISLogger::g_emptyString;
282 }
283 else
284 {
285 g_commandLineOptions.logSubFolder = subFolder;
286 }
287 }
288 else if (startsWith(a, "-lp") && (i + 1) < argc)
289 {
290 g_commandLineOptions.logPath = argv[++i]; // use next argument;
291 }
292 else if (startsWith(a, "-lt="))
293 {
294 g_commandLineOptions.logType = &a[4];
295 }
296 else if (startsWith(a, "-lon"))
297 {
298 g_commandLineOptions.enableLogging = true;
299 }
300 else if (startsWith(a, "-magRecal"))
301 {
302 g_commandLineOptions.rmcPreset = 0;
303 g_commandLineOptions.magRecal = true;
304 g_commandLineOptions.magRecalMode = strtol(a + 9, NULL, 10);
305 }
306 else if (startsWith(a, "-presetPPD"))
307 {
308 g_commandLineOptions.rmcPreset = RMC_PRESET_PPD_GROUND_VEHICLE;
309 }
310 else if (startsWith(a, "-presetINS2"))
311 {
312 g_commandLineOptions.rmcPreset = RMC_PRESET_INS_BITS;
313 }
314 else if (startsWith(a, "-persistent"))
315 {
316 g_commandLineOptions.persistentMessages = true;
317 }
318 else if (startsWith(a, "-q"))
319 {
320 g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_QUIET;
321 }
322 else if (startsWith(a, "-rp") && (i + 1) < argc)
323 {
324 g_commandLineOptions.replayDataLog = true;
325 g_commandLineOptions.logPath = argv[++i]; // use next argument
326 }
327 else if (startsWith(a, "-rs="))
328 {
329 g_commandLineOptions.replayDataLog = true;
330 g_commandLineOptions.replaySpeed = (float)atof(&a[4]);
331 }
332 else if (startsWith(a, "-resetEvb"))
333 {
334 g_commandLineOptions.softwareResetEvb = true;
335 }
336 else if (startsWith(a, "-reset"))
337 {
338 g_commandLineOptions.softwareResetUins = true;
339 }
340 else if (startsWith(a, "-rover="))
341 {
342 g_commandLineOptions.roverConnection = &a[7];
343
344 // DID_GPS1_POS must be enabled for NTRIP VRS to supply rover position.
345 stream_did_t dataset = {};
346 read_did_argument(&dataset, "DID_GPS1_POS");
347 g_commandLineOptions.datasets.push_back(dataset);
348 }
349 else if (startsWith(a, "-r"))
350 {
351 g_commandLineOptions.replayDataLog = true;
352 }
353 else if (startsWith(a, "-stats"))
354 {
355 g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_STATS;
356 }
357 else if (startsWith(a, "-survey="))
358 {
359 g_commandLineOptions.rmcPreset = 0;
360 g_commandLineOptions.surveyIn.state = strtol(a + 8, NULL, 10);
361 int maxDurationSec = strtol(a + 10, NULL, 10);
362 if (maxDurationSec > 5)
363 {
364 g_commandLineOptions.surveyIn.maxDurationSec = maxDurationSec;
365 }
366 }
367 else if (startsWith(a, "-s"))
368 {
369 g_commandLineOptions.displayMode = cInertialSenseDisplay::DMODE_SCROLL;
370 }
371 else if (startsWith(a, "-ub") && (i + 1) < argc)
372 {
373 g_commandLineOptions.updateBootloaderFilename = argv[++i]; // use next argument
374 }
375 else if (startsWith(a, "-uf") && (i + 1) < argc)
376 {
377 g_commandLineOptions.updateAppFirmwareFilename = argv[++i]; // use next argument
378 }
379 else if (startsWith(a, "-uv"))
380 {
381 g_commandLineOptions.bootloaderVerify = true;
382 }
383 else
384 {
385 cout << "Unrecognized command line option: " << a << endl;
386 cltool_outputHelp();
387 return false;
388 }
389 }
390
391 // We are either using a serial port or replaying data
392 if ((g_commandLineOptions.comPort.length() == 0) && !g_commandLineOptions.replayDataLog)
393 {
394 cltool_outputUsage();
395 return false;
396 }
397 else if (g_commandLineOptions.updateAppFirmwareFilename.length() != 0 && g_commandLineOptions.comPort.length() == 0)
398 {
399 cout << "Use COM_PORT option \"-c \" with bootloader" << endl;
400 return false;
401 }
402 else if (g_commandLineOptions.updateBootloaderFilename.length() != 0 && g_commandLineOptions.comPort.length() == 0)
403 {
404 cout << "Use COM_PORT option \"-c \" with bootloader" << endl;
405 return false;
406 }
407
408 return true;
409}

411bool cltool_replayDataLog()
412{
413 if (g_commandLineOptions.logPath.length() == 0)
414 {
415 cout << "Please specify the replay log path!" << endl;
416 return false;
417 }
418
419 cISLogger logger;
420 if (!logger.LoadFromDirectory(g_commandLineOptions.logPath, cISLogger::ParseLogType(g_commandLineOptions.logType), { "ALL" }))
421 {
422 cout << "Failed to load log files: " << g_commandLineOptions.logPath << endl;
423 return false;
424 }
425
426 cout << "Replaying log files: " << g_commandLineOptions.logPath << endl;
427 p_data_t *data;
428 while ((data = logger.ReadData()) != NULL)
429 {
430 g_inertialSenseDisplay.ProcessData(data, g_commandLineOptions.replayDataLog, g_commandLineOptions.replaySpeed);
431
432// if (data->hdr.id == DID_GPS1_RAW)
433// {
434// // Insert your code for processing data here
435// }
436 }
437
438 cout << "Done replaying log files: " << g_commandLineOptions.logPath << endl;
439 g_inertialSenseDisplay.Goodbye();
440 return true;
441}

540{
541 nvm_flash_cfg_t flashCfg = inertialSenseInterface.GetFlashConfig();
542 const map_name_to_info_t& flashMap = *cISDataMappings::GetMapInfo(DID_FLASH_CONFIG);
543
544 if (flashCfgString.length() < 2)
545 {
546 // read flash config and display
547 data_mapping_string_t stringBuffer;
548 cout << "Current flash config" << endl;
549 for (map_name_to_info_t::const_iterator i = flashMap.begin(); i != flashMap.end(); i++)
550 {
551 if (cISDataMappings::DataToString(i->second, NULL, (const uint8_t*)&flashCfg, stringBuffer))
552 {
553 cout << i->second.name << " = " << stringBuffer << endl;
554 }
555 }
556 return false;
557 }
558 else
559 {
560 vector<string> keyValues;
561 splitString(flashCfgString, '|', keyValues);
562 for (size_t i = 0; i < keyValues.size(); i++)
563 {
564 vector<string> keyAndValue;
565 splitString(keyValues[i], '=', keyAndValue);
566 if (keyAndValue.size() == 2)
567 {
568 if (flashMap.find(keyAndValue[0]) == flashMap.end())
569 {
570 cout << "Unrecognized flash config key '" << keyAndValue[0] << "' specified, ignoring." << endl;
571 }
572 else
573 {
574 const data_info_t& info = flashMap.at(keyAndValue[0]);
575 int radix = (keyAndValue[1].compare(0, 2, "0x") == 0 ? 16 : 10);
576 int substrIndex = 2 * (radix == 16); // skip 0x for hex
577 const string& str = keyAndValue[1].substr(substrIndex);
578 cISDataMappings::StringToData(str.c_str(), (int)str.length(), NULL, (uint8_t*)&flashCfg, info, radix);
579 cout << "Updated flash config key '" << keyAndValue[0] << "' to '" << keyAndValue[1].c_str() << "'" << endl;
580 }
581 }
582 }
583 inertialSenseInterface.SetFlashConfig(flashCfg);
584 g_inertialSenseDisplay.Clear();
585 return true;
