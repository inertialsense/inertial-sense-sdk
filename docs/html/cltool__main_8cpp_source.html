47  static float serverKBps = 0;
48  static uint64_t serverByteCount = 0;
49  static uint64_t serverByteRateTimeMsLast = 0;
50  static uint64_t serverByteCountLast = 0;
51  static stringstream outstream;
52 
53  uint64_t newServerByteCount = i->GetClientServerByteCount();
54  if (serverByteCount != newServerByteCount)

57 
58  // Data rate of server bytes
59  uint64_t timeMs = getTickCount();
60  uint64_t dtMs = timeMs - serverByteRateTimeMsLast;
61  if (dtMs >= 1000)
62  {
63  uint64_t serverBytesDelta = serverByteCount - serverByteCountLast;
64  serverKBps = ((float)serverBytesDelta / (float)dtMs);
65 
66  // Update history
67  serverByteCountLast = serverByteCount;
68  serverByteRateTimeMsLast = timeMs;

71  outstream.str(""); // clear

80  }

82 
83  if (server)
84  { // Server
85  outstream << "Connections: " << i->GetClientConnectionCurrent() << " current, " << i->GetClientConnectionTotal() << " total \n";
86  if (showMessageSummary)

101  outstream << i->getClientMessageStatsSummary();

103  }
104  }

171 static bool cltool_setupCommunications(InertialSense& inertialSenseInterface)
172 {
173  inertialSenseInterface.StopBroadcasts(); // Stop streaming any prior messages
174 
175  if (g_commandLineOptions.asciiMessages.size() != 0)
176  {
177  serialPortWriteAscii(inertialSenseInterface.GetSerialPort(), g_commandLineOptions.asciiMessages.c_str(), (int)g_commandLineOptions.asciiMessages.size());
178  return true;
179  }
180 
181  // ask for device info every 2 seconds
182  inertialSenseInterface.BroadcastBinaryData(DID_DEV_INFO, 2000);
183 
184  // depending on command line options. stream various data sets
185  if (g_commandLineOptions.datasetEdit.did)
186  { // Dataset to edit
187  g_inertialSenseDisplay.SelectEditDataset(g_commandLineOptions.datasetEdit.did);
188  inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasetEdit.did, g_commandLineOptions.datasetEdit.periodMultiple);
189  }
190  else while (g_commandLineOptions.datasets.size())
191  { // Datasets to stream
192  inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasets.back().did, g_commandLineOptions.datasets.back().periodMultiple);
193  switch (g_commandLineOptions.datasets.back().did)
194  {
195  case DID_RTOS_INFO:
196  system_command_t cfg;
197  cfg.command = SYS_CMD_ENABLE_RTOS_STATS;
198  cfg.invCommand = ~cfg.command;
199  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
200  break;
201  }
202  g_commandLineOptions.datasets.pop_back();
203  }
204  if (g_commandLineOptions.timeoutFlushLoggerSeconds > 0)
205  {
206  inertialSenseInterface.SetTimeoutFlushLoggerSeconds(g_commandLineOptions.timeoutFlushLoggerSeconds);
207  }
208  if (g_commandLineOptions.magRecal)
209  {
210  // Enable broadcase of DID_MAG_CAL so we can observe progress and tell when the calibration is done (i.e. DID_MAG_CAL.recalCmd == 100).
211  inertialSenseInterface.BroadcastBinaryData(DID_MAG_CAL, 100);
212  // Enable mag recal
213  inertialSenseInterface.SendRawData(DID_MAG_CAL, (uint8_t*)&g_commandLineOptions.magRecalMode, sizeof(g_commandLineOptions.magRecalMode), offsetof(mag_cal_t, recalCmd));
214  }
215  if (g_commandLineOptions.surveyIn.state)
216  { // Enable mult-axis
217  inertialSenseInterface.SendRawData(DID_SURVEY_IN, (uint8_t*)&g_commandLineOptions.surveyIn, sizeof(survey_in_t), 0);
218  }
219  if (g_commandLineOptions.rmcPreset)
220  {
221  inertialSenseInterface.BroadcastBinaryDataRmcPreset(g_commandLineOptions.rmcPreset, RMC_OPTIONS_PRESERVE_CTRL);
222  }
223  if (g_commandLineOptions.persistentMessages)
224  { // Save persistent messages to flash
225  system_command_t cfg;
226  cfg.command = SYS_CMD_SAVE_PERSISTENT_MESSAGES;
227  cfg.invCommand = ~cfg.command;
228  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
229  }
230  if (g_commandLineOptions.softwareResetUins)
231  { // Issue software reset
232  system_command_t cfg;
233  cfg.command = SYS_CMD_SOFTWARE_RESET;
234  cfg.invCommand = ~cfg.command;
235  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
236  }
237  if (g_commandLineOptions.softwareResetEvb)
238  { // Issue software reset to EVB
239  uint32_t sysCommand = SYS_CMD_SOFTWARE_RESET;
240  inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
241  }
242  if (g_commandLineOptions.chipEraseEvb2)
243  { // Chip erase EVB
244  uint32_t sysCommand;
245 
246  sysCommand = SYS_CMD_MANF_UNLOCK;
247  inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
248  sysCommand = SYS_CMD_MANF_CHIP_ERASE;
249  inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
250  }
251  if (g_commandLineOptions.chipEraseUins)
252  { // Chip erase uINS
253  system_command_t cfg;
254 
255  cfg.command = SYS_CMD_MANF_UNLOCK;
256  cfg.invCommand = ~cfg.command;
257  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
258 
259  cfg.command = SYS_CMD_MANF_CHIP_ERASE;
260  cfg.invCommand = ~cfg.command;
261  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
262  }
263  if (g_commandLineOptions.factoryResetUins)
264  { // Reset flash config defaults on uINS
265  system_command_t cfg;
266 
267  cfg.command = SYS_CMD_MANF_UNLOCK;
268  cfg.invCommand = ~cfg.command;
269  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
270 
271  cfg.command = SYS_CMD_MANF_FACTORY_RESET;
272  cfg.invCommand = ~cfg.command;
273  inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
274  }
275 
276  if (g_commandLineOptions.roverConnection.length() != 0)
277  {
278  vector<string> pieces;
279  splitString(g_commandLineOptions.roverConnection, ':', pieces);
280  if (pieces[0] != "TCP" &&
281  pieces[0] != "SERIAL")
282  {

284  return false;
285  }
286  if (pieces[1] != "RTCM3" &&
287  pieces[1] != "IS" &&
288  pieces[1] != "UBLOX")
289  {
290  cout << "Invalid base connection, 2nd field must be: RTCM3, UBLOX, or IS\n -rover=" << g_commandLineOptions.roverConnection << endl;
291  return false;
292  }
293 
294  if (!inertialSenseInterface.OpenConnectionToServer(g_commandLineOptions.roverConnection))
295  {
296  cout << "Failed to connect to server (base)." << endl;
297  }
298  }
299  if (g_commandLineOptions.flashCfg.length() != 0)
300  {
301  return cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg);
302  }
303  if (g_commandLineOptions.evbFlashCfg.length() != 0)
304  {
305  return cltool_updateEvbFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg);
306  }
307  return true;

341 static int cltool_createHost()
342 {
343  InertialSense inertialSenseInterface;
344  if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
345  {
346  cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
347  return -1;
348  }
349  else if (g_commandLineOptions.flashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg))
350  {
351  cout << "Failed to update flash config" << endl;
352  return -1;
353  }
354  else if (g_commandLineOptions.evbFlashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg))
355  {
356  cout << "Failed to update EVB flash config" << endl;
357  return -1;
358  }
359  else if (!inertialSenseInterface.CreateHost(g_commandLineOptions.baseConnection))
360  {
361  cout << "Failed to create host at " << g_commandLineOptions.baseConnection << endl;
362  return -1;
363  }
364 
365  inertialSenseInterface.StopBroadcasts();
366 
367  unsigned int timeSinceClearMs = 0, curTimeMs;
368  while (!g_inertialSenseDisplay.ExitProgram())
369  {
370  inertialSenseInterface.Update();
371  curTimeMs = current_timeMs();
372  bool refresh = false;
373  if (curTimeMs - timeSinceClearMs > 2000 || curTimeMs < timeSinceClearMs)
374  { // Clear terminal
375  g_inertialSenseDisplay.Clear();
376  timeSinceClearMs = curTimeMs;
377  refresh = true;
378  }
379  g_inertialSenseDisplay.Home();
380  cout << g_inertialSenseDisplay.Hello();
381  display_server_client_status(&inertialSenseInterface, true, true, refresh);
382  }
383  cout << "Shutting down..." << endl;
384 
385  // close the interface cleanly, this ensures serial port and any logging are shutdown properly
386  inertialSenseInterface.Close();
387 
388  return 0;
389 }
