41static void display_server_client_status(InertialSense* i, bool server=false, bool showMessageSummary=false, bool refreshDisplay=false)
42{
43 if (g_inertialSenseDisplay.GetDisplayMode() == cInertialSenseDisplay::DMODE_QUIET ||
44 g_inertialSenseDisplay.GetDisplayMode() == cInertialSenseDisplay::DMODE_SCROLL)
45 {
46 return;
47 }
48
49 static float serverKBps = 0;
50 static uint64_t serverByteCount = 0;
51 static uint64_t serverByteRateTimeMsLast = 0;
52 static uint64_t serverByteCountLast = 0;
53 static stringstream outstream;
54
55 uint64_t newServerByteCount = i->GetClientServerByteCount();
56 if (serverByteCount != newServerByteCount)
57 {
58 serverByteCount = newServerByteCount;
59
60 // Data rate of server bytes
61 uint64_t timeMs = getTickCount();
62 uint64_t dtMs = timeMs - serverByteRateTimeMsLast;
63 if (dtMs >= 1000)
64 {
65 uint64_t serverBytesDelta = serverByteCount - serverByteCountLast;
66 serverKBps = ((float)serverBytesDelta / (float)dtMs);
67
68 // Update history
69 serverByteCountLast = serverByteCount;
70 serverByteRateTimeMsLast = timeMs;
71 }
72
73 outstream.str(""); // clear
74 outstream << "\n";
75 if (server)
76 {
77 outstream << "Server: " << i->GetTcpServerIpAddressPort() << " Tx: ";
78 }
79 else
80 {
81 outstream << "Client: " << i->GetClientConnectionInfo() << " Rx: ";
82 }
83 outstream << fixed << setw(3) << setprecision(1) << serverKBps << " KB/s, " << (long long)i->GetClientServerByteCount() << " bytes \n";
84
85 if (server)
86 { // Server
87 outstream << "Connections: " << i->GetClientConnectionCurrent() << " current, " << i->GetClientConnectionTotal() << " total \n";
88 if (showMessageSummary)
89 {
90 outstream << i->getServerMessageStatsSummary();
91 }
92 refreshDisplay = true;
93 }
94 else
95 { // Client
96 com_manager_status_t* status = comManagerGetStatus(0);
97 if (status != NULLPTR && status->communicationErrorCount>2)
98 {
99 outstream << "Com errors: " << status->communicationErrorCount << " \n";
100 }
101 if (showMessageSummary)
102 {
103 outstream << i->getClientMessageStatsSummary();
104 }
105 }
106 }
107
108 if (refreshDisplay)
109 {
110 cout << outstream.str();
111 }
112}

170// Where we tell the uINS what data to send and at what rate.
171// "cltool_dataCallback()" is registered as the callback functions for all received data.
172// All DID messages are found in data_sets.h
173static bool cltool_setupCommunications(InertialSense& inertialSenseInterface)
174{
175 inertialSenseInterface.StopBroadcasts(); // Stop streaming any prior messages
176
177 if (g_commandLineOptions.asciiMessages.size() != 0)
178 {
179 serialPortWriteAscii(inertialSenseInterface.GetSerialPort(), g_commandLineOptions.asciiMessages.c_str(), (int)g_commandLineOptions.asciiMessages.size());
180 return true;
181 }
182
183 // ask for device info every 2 seconds
184 inertialSenseInterface.BroadcastBinaryData(DID_DEV_INFO, 2000);
185
186 // depending on command line options. stream various data sets
187 if (g_commandLineOptions.datasetEdit.did)
188 { // Dataset to edit
189 g_inertialSenseDisplay.SelectEditDataset(g_commandLineOptions.datasetEdit.did);
190 inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasetEdit.did, g_commandLineOptions.datasetEdit.periodMultiple);
191 }
192 else while (g_commandLineOptions.datasets.size())
193 { // Datasets to stream
194 inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasets.back().did, g_commandLineOptions.datasets.back().periodMultiple);
195 switch (g_commandLineOptions.datasets.back().did)
196 {
197 case DID_RTOS_INFO:
198 system_command_t cfg;
199 cfg.command = SYS_CMD_ENABLE_RTOS_STATS;
200 cfg.invCommand = ~cfg.command;
201 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
202 break;
203 }
204 g_commandLineOptions.datasets.pop_back();
205 }
206 if (g_commandLineOptions.timeoutFlushLoggerSeconds > 0)
207 {
208 inertialSenseInterface.SetTimeoutFlushLoggerSeconds(g_commandLineOptions.timeoutFlushLoggerSeconds);
209 }
210 if (g_commandLineOptions.magRecal)
211 {
212 // Enable broadcase of DID_MAG_CAL so we can observe progress and tell when the calibration is done (i.e. DID_MAG_CAL.state: 200=in progress, 201=done).
213 inertialSenseInterface.BroadcastBinaryData(DID_MAG_CAL, 100);
214 // Enable mag recal
215 inertialSenseInterface.SendRawData(DID_MAG_CAL, (uint8_t*)&g_commandLineOptions.magRecalMode, sizeof(g_commandLineOptions.magRecalMode), offsetof(mag_cal_t, state));
216 }
217 if (g_commandLineOptions.surveyIn.state)
218 { // Enable mult-axis
219 inertialSenseInterface.SendRawData(DID_SURVEY_IN, (uint8_t*)&g_commandLineOptions.surveyIn, sizeof(survey_in_t), 0);
220 }
221 if (g_commandLineOptions.rmcPreset)
222 {
223 inertialSenseInterface.BroadcastBinaryDataRmcPreset(g_commandLineOptions.rmcPreset, RMC_OPTIONS_PRESERVE_CTRL);
224 }
225 if (g_commandLineOptions.persistentMessages)
226 { // Save persistent messages to flash
227 system_command_t cfg;
228 cfg.command = SYS_CMD_SAVE_PERSISTENT_MESSAGES;
229 cfg.invCommand = ~cfg.command;
230 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
231 }
232 if (g_commandLineOptions.softwareResetUins)
233 { // Issue software reset
234 system_command_t cfg;
235 cfg.command = SYS_CMD_SOFTWARE_RESET;
236 cfg.invCommand = ~cfg.command;
237 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
238 }
239 if (g_commandLineOptions.softwareResetEvb)
240 { // Issue software reset to EVB
241 uint32_t sysCommand = SYS_CMD_SOFTWARE_RESET;
242 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
243 }
244 if (g_commandLineOptions.chipEraseEvb2)
245 { // Chip erase EVB
246 uint32_t sysCommand;
247
248 sysCommand = SYS_CMD_MANF_UNLOCK;
249 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
250 sysCommand = SYS_CMD_MANF_CHIP_ERASE;
251 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
252 }
253 if (g_commandLineOptions.chipEraseUins)
254 { // Chip erase uINS
255 system_command_t cfg;
256
257 cfg.command = SYS_CMD_MANF_UNLOCK;
258 cfg.invCommand = ~cfg.command;
259 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
260
261 cfg.command = SYS_CMD_MANF_CHIP_ERASE;
262 cfg.invCommand = ~cfg.command;
263 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
264 }
265 if (g_commandLineOptions.factoryResetUins)
266 { // Reset flash config defaults on uINS
267 system_command_t cfg;
268
269 cfg.command = SYS_CMD_MANF_UNLOCK;
270 cfg.invCommand = ~cfg.command;
271 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
272
273 cfg.command = SYS_CMD_MANF_FACTORY_RESET;
274 cfg.invCommand = ~cfg.command;
275 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
276 }
277
278 if (g_commandLineOptions.roverConnection.length() != 0)
279 {
280 vector<string> pieces;
281 splitString(g_commandLineOptions.roverConnection, ':', pieces);
282 if (pieces[0] != "TCP" &&
283 pieces[0] != "SERIAL")
284 {
285 cout << "Invalid base connection, 1st field must be: TCP or SERIAL\n -rover=" << g_commandLineOptions.roverConnection << endl;
286 return false;
287 }
288 if (pieces[1] != "RTCM3" &&
289 pieces[1] != "IS" &&
290 pieces[1] != "UBLOX")
291 {
292 cout << "Invalid base connection, 2nd field must be: RTCM3, UBLOX, or IS\n -rover=" << g_commandLineOptions.roverConnection << endl;
293 return false;
294 }
295
296 if (!inertialSenseInterface.OpenConnectionToServer(g_commandLineOptions.roverConnection))
297 {
298 cout << "Failed to connect to server (base)." << endl;
299 }
300 }
301 if (g_commandLineOptions.flashCfg.length() != 0)
302 {
303 return cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg);
304 }
305 if (g_commandLineOptions.evbFlashCfg.length() != 0)
306 {
307 return cltool_updateEvbFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg);
308 }
309 return true;
310}

312static int cltool_updateFirmware()
313{
314 // [BOOTLOADER INSTRUCTION] Update firmware
315 if (g_commandLineOptions.updateBootloaderFilename.size() > 0)
316 {
317 cout << "Checking bootloader firmware: " << g_commandLineOptions.updateBootloaderFilename << endl;
318 }
319 cout << "Updating application firmware: " << g_commandLineOptions.updateAppFirmwareFilename << endl;
320 vector<InertialSense::bootload_result_t> results = InertialSense::BootloadFile(g_commandLineOptions.comPort,
321 g_commandLineOptions.updateAppFirmwareFilename,
322 g_commandLineOptions.baudRate,
323 bootloadUploadProgress,
324 (g_commandLineOptions.bootloaderVerify ? bootloadVerifyProgress : 0),
325 bootloadStatusInfo,
326 g_commandLineOptions.updateBootloaderFilename,
327 g_commandLineOptions.forceBootloaderUpdate
328 );
329 cout << endl << "Results:" << endl;
330 int errorCount = 0;
331 for (size_t i = 0; i < results.size(); i++)
332 {
333 cout << results[i].port << ": " << (results[i].error.size() == 0 ? "Success\n" : results[i].error);
334 errorCount += (int)(results[i].error.size() != 0);
335 }
336 if (errorCount != 0)
337 {
338 cout << endl << errorCount << " ports failed." << endl;
339 }
340 return (errorCount == 0 ? 0 : -1);
341}

343static int cltool_createHost()
344{
345 InertialSense inertialSenseInterface;
346 if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
347 {
348 cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
349 return -1;
350 }
351 else if (g_commandLineOptions.flashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg))
352 {
353 cout << "Failed to update flash config" << endl;
354 return -1;
355 }
356 else if (g_commandLineOptions.evbFlashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg))
357 {
358 cout << "Failed to update EVB flash config" << endl;
359 return -1;
360 }
361 else if (!inertialSenseInterface.CreateHost(g_commandLineOptions.baseConnection))
362 {
363 cout << "Failed to create host at " << g_commandLineOptions.baseConnection << endl;
364 return -1;
365 }
366
367 inertialSenseInterface.StopBroadcasts();
368
369 unsigned int timeSinceClearMs = 0, curTimeMs;
370 while (!g_inertialSenseDisplay.ExitProgram())
371 {
372 inertialSenseInterface.Update();
373 curTimeMs = current_timeMs();
374 bool refresh = false;
375 if (curTimeMs - timeSinceClearMs > 2000 || curTimeMs < timeSinceClearMs)
376 { // Clear terminal
377 g_inertialSenseDisplay.Clear();
378 timeSinceClearMs = curTimeMs;
379 refresh = true;
380 }
381 g_inertialSenseDisplay.Home();
382 cout << g_inertialSenseDisplay.Hello();
383 display_server_client_status(&inertialSenseInterface, true, true, refresh);
384 }
385 cout << "Shutting down..." << endl;
386
387 // close the interface cleanly, this ensures serial port and any logging are shutdown properly
388 inertialSenseInterface.Close();
389
390 return 0;
391}

393static int inertialSenseMain()
394{
395 // clear display
396 g_inertialSenseDisplay.SetDisplayMode((cInertialSenseDisplay::eDisplayMode)g_commandLineOptions.displayMode);
397 g_inertialSenseDisplay.SetKeyboardNonBlock();
398 g_inertialSenseDisplay.Clear();
399
400 // if replay data log specified on command line, do that now and return
401 if (g_commandLineOptions.replayDataLog)
402 {
403 // [REPLAY INSTRUCTION] 1.) Replay data log
404 return !cltool_replayDataLog();
405 }
406 // if app firmware was specified on the command line, do that now and return
407 else if (g_commandLineOptions.updateAppFirmwareFilename.length() != 0)
408 {
409 if (g_commandLineOptions.updateAppFirmwareFilename.substr(g_commandLineOptions.updateAppFirmwareFilename.length() - 4) != ".hex")
410 {
411 cout << "Incorrect firmware file extension: " << g_commandLineOptions.updateAppFirmwareFilename << endl;
412 return -1;
413 }
414
415 if (g_commandLineOptions.updateBootloaderFilename.length() != 0 &&
416 g_commandLineOptions.updateBootloaderFilename.substr(g_commandLineOptions.updateBootloaderFilename.length() - 4) != ".bin")
417 {
418 cout << "Incorrect bootloader file extension: " << g_commandLineOptions.updateBootloaderFilename << endl;
419 return -1;
420 }
421
422 // [BOOTLOADER INSTRUCTION] 1.) Run bootloader
423 return cltool_updateFirmware();
424 }
425 else if (g_commandLineOptions.updateBootloaderFilename.length() != 0)
426 {
427 cout << "option -uf [FILENAME] must be used with option -ub [FILENAME] " << endl;
428 return -1;
429 }
430 // if host was specified on the command line, create a tcp server
431 else if (g_commandLineOptions.baseConnection.length() != 0)
432 {
433 return cltool_createHost();
434 }
435 else if (g_commandLineOptions.asciiMessages.size() != 0)
436 {
437 serial_port_t serialForAscii;
438 serialPortPlatformInit(&serialForAscii);
439 serialPortOpen(&serialForAscii, g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate, 0);
440 serialPortWriteAscii(&serialForAscii, "STPB", 4);
441 serialPortWriteAscii(&serialForAscii, ("ASCB," + g_commandLineOptions.asciiMessages).c_str(), (int)(5 + g_commandLineOptions.asciiMessages.size()));
442 unsigned char line[512];
443 unsigned char* asciiData;
444 while (!g_inertialSenseDisplay.ExitProgram())
445 {
446 int count = serialPortReadAsciiTimeout(&serialForAscii, line, sizeof(line), 100, &asciiData);
447 if (count > 0)
448 {
449 printf("%s", (char*)asciiData);
450 printf("\r\n");
451 }
452 }
453 }
454 // open the device, start streaming data and logging if needed
455 else
456 {
457 // [C++ COMM INSTRUCTION] STEP 1: Instantiate InertialSense Class
458 // Create InertialSense object, passing in data callback function pointer.
459 InertialSense inertialSenseInterface(cltool_dataCallback);
460
461 // [C++ COMM INSTRUCTION] STEP 2: Open serial port
462 if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate, g_commandLineOptions.disableBroadcastsOnClose))
463 {
464 cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
465 return -1; // Failed to open serial port
466 }
467
468 // [C++ COMM INSTRUCTION] STEP 3: Enable data broadcasting
469 if (cltool_setupCommunications(inertialSenseInterface))
470 {
471 // [LOGGER INSTRUCTION] Setup and start data logger
472 if (g_commandLineOptions.asciiMessages.size() == 0 && !cltool_setupLogger(inertialSenseInterface))
473 {
474 cout << "Failed to setup logger!" << endl;
475 inertialSenseInterface.Close();
476 return -1;
477 }
478 try
479 {
480 // Main loop. Could be in separate thread if desired.
481 while (!g_inertialSenseDisplay.ExitProgram())
482 {
483 g_inertialSenseDisplay.GetKeyboardInput();
484
485 if (g_inertialSenseDisplay.UploadNeeded())
486 {
487 cInertialSenseDisplay::edit_data_t *edata = g_inertialSenseDisplay.EditData();
488 inertialSenseInterface.SendData(edata->did, edata->data, edata->info.dataSize, edata->info.dataOffset);
489 }
490
491 // [C++ COMM INSTRUCTION] STEP 4: Read data
492 if (!inertialSenseInterface.Update())
493 { // device disconnected, exit
494 break;
495 }
496
497 // Print to standard output
498 bool refreshDisplay = g_inertialSenseDisplay.PrintData();
499
500 // Collect and print summary list of client messages received
501 display_server_client_status(&inertialSenseInterface, false, false, refreshDisplay);
502 }
503 }
504 catch (...)
505 {
506 cout << "Unknown exception...";
507 }
508 }
509
510 cout << "Shutting down..." << endl;
511
512 // [C++ COMM INSTRUCTION] STEP 6: Close interface
513 // Close cleanly to ensure serial port and logging are shutdown properly. (optional)
514 inertialSenseInterface.Close();
515 }
516
517 return 0;
518}
