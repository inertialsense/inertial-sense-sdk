41static void display_server_client_status(InertialSense* i, bool server=false, bool showMessageSummary=false, bool refreshDisplay=false)
42{
43 if (g_inertialSenseDisplay.GetDisplayMode() == cInertialSenseDisplay::DMODE_QUIET ||
44 g_inertialSenseDisplay.GetDisplayMode() == cInertialSenseDisplay::DMODE_SCROLL)
45 {
46 return;
47 }
48
49 static float serverKBps = 0;
50 static uint64_t serverByteCount = 0;
51 static uint64_t serverByteRateTimeMsLast = 0;
52 static uint64_t serverByteCountLast = 0;
53 static stringstream outstream;
54
55 uint64_t newServerByteCount = i->GetClientServerByteCount();
56 if (serverByteCount != newServerByteCount)
57 {
58 serverByteCount = newServerByteCount;
59
60 // Data rate of server bytes
61 uint64_t timeMs = getTickCount();
62 uint64_t dtMs = timeMs - serverByteRateTimeMsLast;
63 if (dtMs >= 1000)
64 {
65 uint64_t serverBytesDelta = serverByteCount - serverByteCountLast;
66 serverKBps = ((float)serverBytesDelta / (float)dtMs);
67
68 // Update history
69 serverByteCountLast = serverByteCount;
70 serverByteRateTimeMsLast = timeMs;
71 }
72
73 outstream.str(""); // clear
74 outstream << "\n";
75 if (server)
76 {
77 outstream << "Server: " << i->GetTcpServerIpAddressPort() << " Tx: ";
78 }
79 else
80 {
81 outstream << "Client: " << i->GetClientConnectionInfo() << " Rx: ";
82 }
83 outstream << fixed << setw(3) << setprecision(1) << serverKBps << " KB/s, " << (long long)i->GetClientServerByteCount() << " bytes \n";
84
85 if (server)
86 { // Server
87 outstream << "Connections: " << i->GetClientConnectionCurrent() << " current, " << i->GetClientConnectionTotal() << " total \n";
88 if (showMessageSummary)
89 {
90 outstream << i->getServerMessageStatsSummary();
91 }
92 refreshDisplay = true;
93 }
94 else
95 { // Client
96 com_manager_status_t* status = comManagerGetStatus(0);
97 if (status != NULLPTR && status->communicationErrorCount>2)
98 {
99 outstream << "Com errors: " << status->communicationErrorCount << " \n";
100 }
101 if (showMessageSummary)
102 {
103 outstream << i->getClientMessageStatsSummary();
104 }
105 }
106 }
107
108 if (refreshDisplay)
109 {
110 cout << outstream.str();
111 }
112}

170// Where we tell the uINS what data to send and at what rate.
171// "cltool_dataCallback()" is registered as the callback functions for all received data.

173static bool cltool_setupCommunications(InertialSense& inertialSenseInterface)
174{
175 inertialSenseInterface.StopBroadcasts(); // Stop streaming any prior messages
176
177 if (g_commandLineOptions.asciiMessages.size() != 0)
178 {
179 serialPortWriteAscii(inertialSenseInterface.GetSerialPort(), g_commandLineOptions.asciiMessages.c_str(), (int)g_commandLineOptions.asciiMessages.size());
180 return true;
181 }
182
183 // ask for device info every 2 seconds
184 inertialSenseInterface.BroadcastBinaryData(DID_DEV_INFO, 2000);
185
186 // depending on command line options. stream various data sets
187 if (g_commandLineOptions.datasetEdit.did)
188 { // Dataset to edit
189 g_inertialSenseDisplay.SelectEditDataset(g_commandLineOptions.datasetEdit.did);
190 inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasetEdit.did, g_commandLineOptions.datasetEdit.periodMultiple);
191 }
192 else while (g_commandLineOptions.datasets.size())
193 { // Datasets to stream
194 inertialSenseInterface.BroadcastBinaryData(g_commandLineOptions.datasets.back().did, g_commandLineOptions.datasets.back().periodMultiple);
195 switch (g_commandLineOptions.datasets.back().did)
196 {
197 case DID_RTOS_INFO:
198 system_command_t cfg;
199 cfg.command = SYS_CMD_ENABLE_RTOS_STATS;
200 cfg.invCommand = ~cfg.command;
201 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
202 break;
203 }
204 g_commandLineOptions.datasets.pop_back();
205 }
206 if (g_commandLineOptions.timeoutFlushLoggerSeconds > 0)
207 {
208 inertialSenseInterface.SetTimeoutFlushLoggerSeconds(g_commandLineOptions.timeoutFlushLoggerSeconds);
209 }
210 if (g_commandLineOptions.magRecal)
211 {
212 // Enable broadcase of DID_MAG_CAL so we can observe progress and tell when the calibration is done (i.e. DID_MAG_CAL.state: 200=in progress, 201=done).
213 inertialSenseInterface.BroadcastBinaryData(DID_MAG_CAL, 100);
214 // Enable mag recal
215 inertialSenseInterface.SendRawData(DID_MAG_CAL, (uint8_t*)&g_commandLineOptions.magRecalMode, sizeof(g_commandLineOptions.magRecalMode), offsetof(mag_cal_t, state));
216 }
217 if (g_commandLineOptions.surveyIn.state)
218 { // Enable mult-axis
219 inertialSenseInterface.SendRawData(DID_SURVEY_IN, (uint8_t*)&g_commandLineOptions.surveyIn, sizeof(survey_in_t), 0);
220 }
221 if (g_commandLineOptions.rmcPreset)
222 {
223 inertialSenseInterface.BroadcastBinaryDataRmcPreset(g_commandLineOptions.rmcPreset, RMC_OPTIONS_PRESERVE_CTRL);
224 }
225 if (g_commandLineOptions.persistentMessages)
226 { // Save persistent messages to flash
227 system_command_t cfg;
228 cfg.command = SYS_CMD_SAVE_PERSISTENT_MESSAGES;
229 cfg.invCommand = ~cfg.command;
230 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
231 }
232 if (g_commandLineOptions.softwareResetUins)
233 { // Issue software reset
234 system_command_t cfg;
235 cfg.command = SYS_CMD_SOFTWARE_RESET;
236 cfg.invCommand = ~cfg.command;
237 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
238 }
239 if (g_commandLineOptions.softwareResetEvb)
240 { // Issue software reset to EVB
241 uint32_t sysCommand = SYS_CMD_SOFTWARE_RESET;
242 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
243 }
244 if (g_commandLineOptions.chipEraseEvb2)
245 { // Chip erase EVB
246 uint32_t sysCommand;
247
248 sysCommand = SYS_CMD_MANF_UNLOCK;
249 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
250 sysCommand = SYS_CMD_MANF_CHIP_ERASE;
251 inertialSenseInterface.SendRawData(DID_EVB_STATUS, (uint8_t*)&sysCommand, sizeof(uint32_t), offsetof(evb_status_t, sysCommand));
252 }
253 if (g_commandLineOptions.chipEraseUins)
254 { // Chip erase uINS
255 system_command_t cfg;
256
257 cfg.command = SYS_CMD_MANF_UNLOCK;
258 cfg.invCommand = ~cfg.command;
259 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
260
261 cfg.command = SYS_CMD_MANF_CHIP_ERASE;
262 cfg.invCommand = ~cfg.command;
263 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
264 }
265 if (g_commandLineOptions.factoryResetUins)
266 { // Reset flash config defaults on uINS
267 system_command_t cfg;
268
269 cfg.command = SYS_CMD_MANF_UNLOCK;
270 cfg.invCommand = ~cfg.command;
271 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
272
273 cfg.command = SYS_CMD_MANF_FACTORY_RESET;
274 cfg.invCommand = ~cfg.command;
275 inertialSenseInterface.SendRawData(DID_SYS_CMD, (uint8_t*)&cfg, sizeof(system_command_t), 0);
276 }
277
278 if (g_commandLineOptions.roverConnection.length() != 0)
279 {
280 vector<string> pieces;
281 splitString(g_commandLineOptions.roverConnection, ':', pieces);
282 if (pieces[0] != "TCP" &&
283 pieces[0] != "SERIAL")
284 {
285 cout << "Invalid base connection, 1st field must be: TCP or SERIAL\n -rover=" << g_commandLineOptions.roverConnection << endl;
286 return false;
287 }
288 if (pieces[1] != "RTCM3" &&
289 pieces[1] != "IS" &&
290 pieces[1] != "UBLOX")
291 {
292 cout << "Invalid base connection, 2nd field must be: RTCM3, UBLOX, or IS\n -rover=" << g_commandLineOptions.roverConnection << endl;
293 return false;
294 }
295
296 if (!inertialSenseInterface.OpenConnectionToServer(g_commandLineOptions.roverConnection))
297 {
298 cout << "Failed to connect to server (base)." << endl;
299 }
300 }
301 if (g_commandLineOptions.flashCfg.length() != 0)
302 {
303 return cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg);
304 }
305 if (g_commandLineOptions.evbFlashCfg.length() != 0)
306 {
307 return cltool_updateEvbFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg);
308 }
309 return true;
310}

317static int cltool_updateFirmware()
318{
319 // [BOOTLOADER INSTRUCTION] Update firmware
320 if (g_commandLineOptions.updateBootloaderFilename.size() > 0)
321 {
322 cout << "Checking bootloader firmware: " << g_commandLineOptions.updateBootloaderFilename << endl;
323 }
324 cout << "Updating application firmware: " << g_commandLineOptions.updateAppFirmwareFilename << endl;

329 g_commandLineOptions.comPort,
330 0,

332 g_commandLineOptions.updateBootloaderFilename,
333 g_commandLineOptions.forceBootloaderUpdate,

342}

346void printProgress()
347{
348 print_mutex.lock();
349
350 int divisor = 0;
351 float total = 0.0f;
352
353 cISBootloaderThread::m_ctx_mutex.lock();
354
355 for (size_t i = 0; i < cISBootloaderThread::ctx.size(); i++)
356 {
357 if (cISBootloaderThread::ctx[i] && cISBootloaderThread::ctx[i]->m_use_progress)
358 {
359 divisor++;
360
361 if (!cISBootloaderThread::ctx[i]->m_verify)
362 {
363 total += cISBootloaderThread::ctx[i]->m_update_progress;
364 }
365 else
366 {
367 total += cISBootloaderThread::ctx[i]->m_update_progress * 0.5f;
368 total += cISBootloaderThread::ctx[i]->m_verify_progress * 0.5f;
369 }
370 }
371 }
372
373 cISBootloaderThread::m_ctx_mutex.unlock();
374
375 if (divisor) {
376 total /= divisor;
377 int display = (int)(total * 100);
378 printf("Progress: %d%%\r", display);
379 fflush(stdout);
380 }
381
382 print_mutex.unlock();
383}

440static int cltool_createHost()
441{
442 InertialSense inertialSenseInterface;
443 if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate))
444 {
445 cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
446 return -1;
447 }
448 else if (g_commandLineOptions.flashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.flashCfg))
449 {
450 cout << "Failed to update flash config" << endl;
451 return -1;
452 }
453 else if (g_commandLineOptions.evbFlashCfg.length() != 0 && !cltool_updateFlashCfg(inertialSenseInterface, g_commandLineOptions.evbFlashCfg))
454 {
455 cout << "Failed to update EVB flash config" << endl;
456 return -1;
457 }
458 else if (!inertialSenseInterface.CreateHost(g_commandLineOptions.baseConnection))
459 {
460 cout << "Failed to create host at " << g_commandLineOptions.baseConnection << endl;
461 return -1;
462 }
463
464 inertialSenseInterface.StopBroadcasts();
465
466 unsigned int timeSinceClearMs = 0, curTimeMs;
467 while (!g_inertialSenseDisplay.ExitProgram())
468 {
469 inertialSenseInterface.Update();
470 curTimeMs = current_timeMs();
471 bool refresh = false;
472 if (curTimeMs - timeSinceClearMs > 2000 || curTimeMs < timeSinceClearMs)
473 { // Clear terminal
474 g_inertialSenseDisplay.Clear();
475 timeSinceClearMs = curTimeMs;
476 refresh = true;
477 }
478 g_inertialSenseDisplay.Home();
479 cout << g_inertialSenseDisplay.Hello();
480 display_server_client_status(&inertialSenseInterface, true, true, refresh);
481 }
482 cout << "Shutting down..." << endl;
483
484 // close the interface cleanly, this ensures serial port and any logging are shutdown properly
485 inertialSenseInterface.Close();
486
487 return 0;
488}

490static int inertialSenseMain()
491{
492 // clear display
493 g_inertialSenseDisplay.SetDisplayMode((cInertialSenseDisplay::eDisplayMode)g_commandLineOptions.displayMode);
494 g_inertialSenseDisplay.SetKeyboardNonBlock();
495 g_inertialSenseDisplay.Clear();
496
497 // if replay data log specified on command line, do that now and return
498 if (g_commandLineOptions.replayDataLog)
499 {
500 // [REPLAY INSTRUCTION] 1.) Replay data log
501 return !cltool_replayDataLog();
502 }
503 // if app firmware was specified on the command line, do that now and return
504 else if (g_commandLineOptions.updateAppFirmwareFilename.length() != 0)
505 {
506 return cltool_updateFirmware();
507 }
508 else if (g_commandLineOptions.updateBootloaderFilename.length() != 0)
509 {
510 cout << "option -uf [FILENAME] must be used with option -ub [FILENAME] " << endl;
511 return -1;
512 }
513 // if host was specified on the command line, create a tcp server
514 else if (g_commandLineOptions.baseConnection.length() != 0)
515 {
516 return cltool_createHost();
517 }
518 else if (g_commandLineOptions.asciiMessages.size() != 0)
519 {
520 serial_port_t serialForAscii;
521 serialPortPlatformInit(&serialForAscii);
522 serialPortOpen(&serialForAscii, g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate, 0);
523 serialPortWriteAscii(&serialForAscii, "STPB", 4);
524 serialPortWriteAscii(&serialForAscii, ("ASCB," + g_commandLineOptions.asciiMessages).c_str(), (int)(5 + g_commandLineOptions.asciiMessages.size()));
525 unsigned char line[512];
526 unsigned char* asciiData;
527 while (!g_inertialSenseDisplay.ExitProgram())
528 {
529 int count = serialPortReadAsciiTimeout(&serialForAscii, line, sizeof(line), 100, &asciiData);
530 if (count > 0)
531 {
532 printf("%s", (char*)asciiData);
533 printf("\r\n");
534 }
535 }
536 }
537 // open the device, start streaming data and logging if needed
538 else
539 {
540 // [C++ COMM INSTRUCTION] STEP 1: Instantiate InertialSense Class
541 // Create InertialSense object, passing in data callback function pointer.
542 InertialSense inertialSenseInterface(cltool_dataCallback);
543
544 // [C++ COMM INSTRUCTION] STEP 2: Open serial port
545 if (!inertialSenseInterface.Open(g_commandLineOptions.comPort.c_str(), g_commandLineOptions.baudRate, g_commandLineOptions.disableBroadcastsOnClose))
546 {
547 cout << "Failed to open serial port at " << g_commandLineOptions.comPort.c_str() << endl;
548 return -1; // Failed to open serial port
549 }
550
551 // [C++ COMM INSTRUCTION] STEP 3: Enable data broadcasting
552 if (cltool_setupCommunications(inertialSenseInterface))
553 {
554 // [LOGGER INSTRUCTION] Setup and start data logger
555 if (g_commandLineOptions.asciiMessages.size() == 0 && !cltool_setupLogger(inertialSenseInterface))
556 {
557 cout << "Failed to setup logger!" << endl;
558 inertialSenseInterface.Close();
559 return -1;
560 }
561 try
562 {
563 // Main loop. Could be in separate thread if desired.
564 while (!g_inertialSenseDisplay.ExitProgram())
565 {
566 g_inertialSenseDisplay.GetKeyboardInput();
567
568 if (g_inertialSenseDisplay.UploadNeeded())
569 {
570 cInertialSenseDisplay::edit_data_t *edata = g_inertialSenseDisplay.EditData();
571 inertialSenseInterface.SendData(edata->did, edata->data, edata->info.dataSize, edata->info.dataOffset);
572 }
573
574 // [C++ COMM INSTRUCTION] STEP 4: Read data
575 if (!inertialSenseInterface.Update())
576 { // device disconnected, exit
577 break;
578 }
579
580 // Print to standard output
581 bool refreshDisplay = g_inertialSenseDisplay.PrintData();
582
583 // Collect and print summary list of client messages received
584 display_server_client_status(&inertialSenseInterface, false, false, refreshDisplay);
585 }
586 }
587 catch (...)
588 {
589 cout << "Unknown exception...";
590 }
591 }
592
593 cout << "Shutting down..." << endl;
594
595 // [C++ COMM INSTRUCTION] STEP 6: Close interface
596 // Close cleanly to ensure serial port and logging are shutdown properly. (optional)
597 inertialSenseInterface.Close();
598 }
599
600 return 0;
601}
