158 bool cDeviceLogSorted::SaveData(p_data_hdr_t* dataHdr, const uint8_t* dataBuf)
159 {
160  cDeviceLog::SaveData(dataHdr, dataBuf);
161 
162  uint32_t id = dataHdr->id;
163 
164  if (id >= DID_COUNT)
165  {
166  return false;
167  }
168 
169  if (m_chunks[id] == NULLPTR)
170  {
171  m_chunks[id] = new cSortedDataChunk();
172  // First time saving to this chunk
173  m_chunks[id]->m_subHdr.dHdr = *dataHdr;
174  m_chunks[id]->m_hdr.pHandle = m_pHandle;
175  m_chunks[id]->m_hdr.devSerialNum = m_devInfo.serialNumber;
176  }
177  cSortedDataChunk* chunk = m_chunks[id];
178 
179  // Add serial number if available
180  if (id == DID_DEV_INFO)
181  {
182  if (!copyDataPToStructP2(&m_devInfo, dataHdr, dataBuf, sizeof(dev_info_t)))
183  {
184  int start = dataHdr->offset;
185  int end = dataHdr->offset + dataHdr->size;
186  int snOffset = offsetof(dev_info_t, serialNumber);
187 
188  // Did we get the serial number portion of devInfo?
189  if (start <= snOffset && (int)(snOffset + sizeof(uint32_t)) <= end)
190  {
191  chunk->m_hdr.devSerialNum = m_devInfo.serialNumber;
192  }
193  }
194  }
195 
196  int32_t dataBytes = sizeof(int32_t) + dataHdr->size;
197 
198  // depending on devices and sources, some logs may have different size for the same id
199  // if the size changes, flush the chunk and start a new one with the new size
200  if ((chunk->m_subHdr.dHdr.size != dataHdr->size) || // DID chunk size or offset changed
201  (chunk->m_subHdr.dHdr.offset != dataHdr->offset) ||
202  (dataBytes > chunk->GetBuffFree())) // chunk is too full
203  {
204  // write chunk to file and clear chunk.
205  if (!WriteChunkToFile(id))
206  {
207  return false;
208  }
209  chunk = m_chunks[id];
210 
211  // Reset data header in case it changed
212  chunk->m_subHdr.dHdr = *dataHdr;
213  }
214 
215  // Add data header and data buffer to chunk
216  int nBytes = chunk->PushBack((unsigned char*)&(m_dataSerNum), sizeof(uint32_t), (unsigned char*)dataBuf, dataHdr->size);
217  if (nBytes <= 0)
218  {
219  return false;
220  }
221 
222 #if LOG_DEBUG_PRINT_DID_SAVE
223  double timestamp = cISDataMappings::GetTimestamp(dataHdr, dataBuf);
224  printf("sorted did save: %d DID: %2d size: %3d time: %.4lf\n", m_dataSerNum, dataHdr->id, nBytes-(int)sizeof(uint32_t), timestamp);
225 #endif
226 
227  // Prep for next save
228  chunk->m_subHdr.dCount++;
229  m_dataSerNum++;
230 
231  return true;
232 }

235 bool cDeviceLogSorted::WriteChunkToFile(uint32_t id)
236 {
237  if (m_chunks[id] == NULLPTR)
238  {
239  return false;
240  }
241 
242  // Byte size
243  int nBytes = m_chunks[id]->GetDataSize();
244  if (nBytes <= 0)
245  { // No data
246  return true;
247  }
248 
249  // Create new file if needed
250  if (m_pFile == NULLPTR && !OpenNewSaveFile())
251  {
252  return false;
253  }
254 
255 #if LOG_DEBUG_PRINT_CHUNK_SAVE
256  printf("sorted chunk save: DID:%3d dCount: %5d nBytes: %6d dataSN: %2d\n", m_chunks[id]->m_subHdr.dHdr.id, m_chunks[id]->m_subHdr.dCount, nBytes, m_chunks[id]->GetDataSerNum());
257 #endif
258 
259  if (nBytes != 0)
260  {
261  // Write chunk to file and clear chunk
262  if ((nBytes = m_chunks[id]->WriteToFile(m_pFile, 1)) <= 0)
263  {
264  return false;
265  }
266  }
267 
268  m_fileSize += nBytes;
269  m_logSize += nBytes;
270 
271  // Remove data
272  m_chunks[id]->Clear();
273 
274  // File is large enough to be closed
275  if (m_fileSize >= m_maxFileSize)
276  {
277  CloseISLogFile(m_pFile);
278  }
279 
280  return true;
281 }

312 p_data_t* cDeviceLogSorted::SerializeDataFromChunks()
313 {
314 
315 tryAgain:
316 
317  uint32_t foundId = UINT_MAX;
318  uint32_t minSerialNum = UINT_MAX;
319  cSortedDataChunk *chunk;
320 
321  // while there is data in any chunk, find the chunk with the next id
322  for (uint32_t id = 1; id < DID_COUNT; id++)
323  {
324  if (!m_chunksAvailable[id])
325  {
326  continue;
327  }
328 
329  chunk = m_chunks[id];
330  if (chunk == NULLPTR || chunk->GetDataSize() == 0)
331  {
332  // Chunk is empty. Search all files for new chuck.
333  if (ReadNextChunkFromFiles(id))
334  {
335  chunk = m_chunks[id];
336  }
337  if (chunk == NULLPTR || chunk->GetDataSize() == 0)
338  {
339  m_chunksAvailable[id] = false;
340  continue;
341  }
342  }
343 
344  uint32_t dataSerNum = chunk->GetDataSerNum();
345  if (dataSerNum == UINT_MAX)
346  {
347  continue;
348  }
349  if (dataSerNum == m_dataSerNum)
350  { // Found exact match.
351  foundId = id;
352  break;
353  }
354  if (dataSerNum < minSerialNum)
355  { // Search for lowest serial number
356  foundId = id;
357  minSerialNum = dataSerNum;
358  }
359  }
360 
361  if (foundId == UINT_MAX)
362  { // No more data left
363  return NULL;
364  }
365 
366  chunk = m_chunks[foundId];
367  p_cnk_data_t* cnkData = (p_cnk_data_t*)(chunk->GetDataPtr());
368 
369 #if 0 // Error check for gap data serial number. Verify we didn't drop any data.
370 
371  if (m_lastSerNum == UINT_MAX)
372  {
373  m_lastSerNum = m_dataSerNum;
374  }
375  if (m_dataSerNum - m_lastSerNum > 1)
376  {
377  cout << endl << "Gap in data: " << foundId << "\t" << m_dataSerNum;
378  }
379  m_lastSerNum = m_dataSerNum;
380 
381 #endif
382 
383  // Increment data serial number to one larger than current
384  m_dataSerNum = chunk->GetDataSerNum() + 1;
385 
386  if (chunk->m_subHdr.dHdr.size <= MAX_DATASET_SIZE)
387  {
388  // Data fits in temp buf
389 
390  // Copy data header
391  m_data.hdr = chunk->m_subHdr.dHdr;
392 
393  // Copy data buffer, ensure not to overrun chunk memory in case of corrupt data
394  memcpy(m_data.buf, cnkData->buf, _MIN(chunk->GetDataSize(), (int32_t)(m_data.hdr.size)));
395 
396  // Size = serial number plus data size
397  int pSize = chunk->m_subHdr.dHdr.size + sizeof(uint32_t);
398 
399  chunk->PopFront(pSize);
400  return &m_data;
401  }
402  else
403  {
404  perror("Data is larger than max data set size");
405  }
406 
407  goto tryAgain;
408 }

411 // This function reads the next sorted chunk matching the specified DID from all of the files. The file pointers are advanced
412 // if the current chunk dataSerNum is older than the current m_dataSerNum. Files are closed if the last chuch dataSerNum
413 // is older than the current m_dataSerNum.
414 bool cDeviceLogSorted::ReadNextChunkFromFiles(uint32_t id)
415 {
416  // Error check ID
417  if (id >= DID_COUNT || id == DID_NULL)
418  {
419  return false;
420  }
421 
422  // Reset data serial number and file data size
423  m_fileSize = 0;
424 
425  // Read chunk and append it to existing chunks of same type
426  if (!ReadChunkFromFiles(&m_readChunk, id))
427  {
428  return false;
429  }
430 
431  id = m_readChunk.m_subHdr.dHdr.id;
432 
433  if (m_chunks[id] == NULLPTR)
434  {
435  m_chunks[id] = new cSortedDataChunk();
436  }
437 
438  cSortedDataChunk* chunk = m_chunks[id];
439 
440  if (chunk->GetDataSize() == 0)
441  {
442  // Reset buffer
443  chunk->Clear();
444 
445  // Copy chunk sub header
446  chunk->m_subHdr = m_readChunk.m_subHdr;
447 
448  // Find lowest serial number (used for sorting data)
449  m_dataSerNum = _MIN(m_dataSerNum, m_readChunk.GetDataSerNum());
450  }
451 
452  // add data count
453  chunk->m_subHdr.dCount += m_readChunk.m_subHdr.dCount;
454 
455  // Resize chunk if necessary
456 // if (m_readChunk.GetDataSize() > chunk->GetBuffFree())
457 // {
458 // chunk->Resize(chunk->GetDataSize() + m_readChunk.GetDataSize());
459 // }
460 
461  // Append data
462  chunk->PushBack(m_readChunk.GetDataPtr(), m_readChunk.GetDataSize());
463 
464  return m_fileSize != 0;
465 }
