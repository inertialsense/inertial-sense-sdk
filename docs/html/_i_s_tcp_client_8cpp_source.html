78int ISSocketCanWrite(socket_t socket, int timeoutMilliseconds)
79{
80 struct timeval tv = { timeoutMilliseconds / 1000, (timeoutMilliseconds % 1000) * 1000 };
81 fd_set ws;
82 FD_ZERO(&ws);
83 FD_SET(socket, &ws);
84 int numberOfSocketsThatCanWrite = select((int)(socket + 1), NULL, &ws, NULL, &tv);
85 return (numberOfSocketsThatCanWrite > 0);
86}

98int ISSocketWrite(socket_t socket, const uint8_t* data, int dataLength)
99{
100 int totalWriteCount = 0;
101 int writeCount;
102
103 while (totalWriteCount < dataLength)
104 {
105 if (!ISSocketCanWrite(socket))
106 {
107 break;
108 }
109
110 int flags = 0;
111#if PLATFORM_IS_LINUX || PLATFORM_IS_APPLE
112 flags = MSG_NOSIGNAL;
113#endif
114 writeCount = send(socket, (const char*)data, dataLength, flags);
115 if (writeCount < 0)
116 {
117 break;
118 }
119 totalWriteCount += writeCount;
120 }
121
122 return totalWriteCount;
123}

125int ISSocketRead(socket_t socket, uint8_t* data, int dataLength)
126{
127 int count = recv(socket, (char*)data, dataLength, 0);
128 if (count < 0)
129 {
130
131#if PLATFORM_IS_WINDOWS
132
133 DWORD err = GetLastError();
134 if (err == SO_RCVTIMEO || err == WSAETIMEDOUT || err == WSAEWOULDBLOCK)
135 {
136 return 0;
137 }
138
139#else
140
141 if (errno == EAGAIN || errno == EWOULDBLOCK)
142 {
143 return 0;
144 }
145
146#endif
147
148 }
149 return count;
150}

152int ISSocketSetBlocking(socket_t socket, bool blocking)
153{
154
155#if PLATFORM_IS_WINDOWS
156
157 u_long blockingInt = (blocking ? 0 : 1);
158 return ioctlsocket(socket, FIONBIO, &blockingInt);
159
160#else
161
162 int opts = fcntl(socket, F_GETFL);
163 if (opts < 0)
164 {
165 return -1;
166 }
167 opts = (blocking ? opts & (~O_NONBLOCK) : opts | O_NONBLOCK);
168 return fcntl(socket, F_SETFL, opts);
169
170#endif
171
172}

194{

208

210 if (status == 0)
211 {
212 status = close(socket);
213 }
214
215#endif
216
217 }
218
219 socket = 0;
220
221 return status;

238int cISTcpClient::Open(const string& host, int port, int timeoutMilliseconds)
239{
240 Close();
241 int status;
242 this->m_host = host;
243 m_port = port;
244 char portString[64];
245 snprintf(portString, sizeof(portString), "%ld", (long)m_port);
246 addrinfo* result = NULL;
247 addrinfo hints = addrinfo();
248 hints.ai_family = AF_INET;
249 hints.ai_socktype = SOCK_STREAM;
250 hints.ai_protocol = IPPROTO_TCP;
251
252 // attempt to get info about the host and port
253 status = getaddrinfo(m_host.c_str(), portString, &hints, &result);
254 if (status != 0)
255 {
256 // no info, fail
257 Close();
258 return status;
259 }
260
261 // create the socket
262 m_socket = socket(result->ai_family, result->ai_socktype, result->ai_protocol);
263 if (m_socket == 0)
264 {
265 // no socket, fail
266 freeaddrinfo(result);
267 Close();
268 return -1;
269 }
270
271 // make non-blocking socket
272 SetBlocking(false);
273
274 // because non-blocking, connect returns immediately
275 // status return value is unreliable
276 connect(m_socket, result->ai_addr, (int)result->ai_addrlen);
277
278 #if PLATFORM_IS_WINDOWS
279
280 status = ISSocketCanWrite(m_socket, timeoutMilliseconds);
281 if (status < 0)
282 {
283 freeaddrinfo(result);
284 Close();
285 return -1;
286 }
287
288 #else
289
290 // check sock_opt_err in order to confirm socket is actually connected
291 int sock_opt_err;
292 socklen_t sock_opt_err_len = sizeof(sock_opt_err);
293 status = getsockopt(m_socket, SOL_SOCKET, SO_ERROR, &sock_opt_err, &sock_opt_err_len);
294 if (sock_opt_err != 0)
295 {
296 freeaddrinfo(result);
297 Close();
298 return -1;
299 }
300
301 #endif
302
303 freeaddrinfo(result);
304 return 0;
305}

333{
334 string msg = "GET /" + subUrl + " HTTP/1.1\r\n";
