135 bool cDeviceLog::OpenNewSaveFile()
136 {
137  // Close existing file
138  CloseISLogFile(m_pFile);
139 
140  // Ensure directory exists
141  if (m_directory.empty())
142  {
143  return false;
144  }
145 
146  // create directory
147  _MKDIR(m_directory.c_str());
148 
149 #if !PLATFORM_IS_EMBEDDED
150 
151  // clear out space if we need to
152  if (m_maxDiskSpace != 0)
153  {
154  vector<ISFileManager::file_info_t> files;
155  uint64_t spaceUsed = ISFileManager::GetDirectorySpaceUsed(m_directory.c_str(), files, true, false);
156  unsigned int index = 0;
157 
158  // clear out old files until we have space
159  while (spaceUsed > m_maxDiskSpace && index < files.size())
160  {
161  spaceUsed -= files[index].size;
162  ISFileManager::DeleteFile(files[index++].name);
163  }
164  }
165 
166 #endif
167 
168  // Open new file
169  m_fileCount++;
170  uint32_t serNum = m_devInfo.serialNumber;
171  if (!serNum)
172  {
173  serNum = m_pHandle;
174  }
175  string fileName = GetNewFileName(serNum, m_fileCount, NULL);
176  m_pFile = CreateISLogFile(fileName, "wb");
177  m_fileSize = 0;
178 
179  if (m_pFile && m_pFile->isOpened())
180  {
181 #if LOG_DEBUG_FILE_WRITE
182  printf("cDeviceLog::OpenNewSaveFile %s\n", fileName.c_str());
183 #endif
184  return true;
185  }
186  else
187  {
188 #if LOG_DEBUG_FILE_WRITE
189  printf("cDeviceLog::OpenNewSaveFile FAILED %s\n", fileName.c_str());
190 #endif
191  return false;
192  }
193 }
