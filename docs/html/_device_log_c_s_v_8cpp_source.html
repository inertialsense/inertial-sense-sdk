42void cDeviceLogCSV::InitDeviceForReading()
43{
44 cDeviceLog::InitDeviceForReading();
45
46 // Create a list of file for each possible data set
47 m_logs.clear();
48 for (uint32_t id = DID_NULL + 1; id < DID_COUNT; id++)
49 {
50 const char* dataSet = cISDataMappings::GetDataSetName(id);
51 if (dataSet != NULL)
52 {
53 string dataSetRegex = string(dataSet) + "\\.csv$";
54 if (m_devInfo.serialNumber > 0 && m_devInfo.serialNumber < 4294967295) // < 0xFFFFFFFF
55 { // Include serial number if valid
56 dataSetRegex = "LOG_SN" + to_string(m_devInfo.serialNumber) + ".*?" + dataSetRegex;
57 }
58 vector<ISFileManager::file_info_t> infos;
59 vector<string> files;
60 ISFileManager::GetDirectorySpaceUsed(m_directory, dataSetRegex, infos, false, false);
61 if (infos.size() != 0)
62 {
63 cCsvLog log;
64 log.dataId = id;
65 log.dataSize = cISDataMappings::GetSize(log.dataId);
66 for (size_t i = 0; i < infos.size(); i++)
67 {
68 files.push_back(infos[i].name);
69 }
70 m_currentFiles[id] = files;
71 m_currentFileIndex[id] = 0;
72 while (OpenNewFile(log, true) && !GetNextLineForFile(log)) {}
73 if (log.nextLine.length() != 0)
74 {
75 m_logs[id] = log;
76 }
77 }
78 }
79 }
80}

101bool cDeviceLogCSV::OpenNewFile(cCsvLog& log, bool readonly)
102{
103 const char* dataSetName = cISDataMappings::GetDataSetName(log.dataId);
104 if (dataSetName == NULL)
105 {
106 return false;
107 }
108
109 // Close existing file
110 if (log.pFile)
111 {
112 fclose(log.pFile);
113 log.pFile = NULL;
114 }
115
116 // Ensure directory exists
117 if (m_directory.empty())
118 {
119 return false;
120 }
121
122 _MKDIR(m_directory.c_str());
123
124 // Open new file
125 uint32_t serNum = m_devInfo.serialNumber;
126 if (!serNum)
127 {
128 serNum = m_pHandle;
129 }
130
131 string fileName;
132 if (readonly)
133 {
134tryNextFile:
135 {
136 uint32_t index = m_currentFileIndex[log.dataId];
137 vector<string>& files = m_currentFiles[log.dataId];
138 if (index >= files.size())
139 {
140 m_logs.erase(log.dataId);
141 return false;
142 }
143 fileName = files[index++];
144 m_currentFileIndex[log.dataId] = index;
145 log.pFile = fopen(fileName.c_str(), "r");
146 log.fileCount++;
147 struct stat st;
148 stat(fileName.c_str(), &st);
149 log.fileSize = st.st_size;
150 m_logSize += log.fileSize;
151 if (m_csv.ReadHeaderFromFile(log.pFile, log.dataId, log.columnHeaders) == 0)
152 {
153 goto tryNextFile;
154 }
155 }
156 }
157 else
158 {
159 log.fileCount++;
160 fileName = GetNewFileName(serNum, log.fileCount, dataSetName);
161 log.pFile = fopen(fileName.c_str(), "w");
162
163 // Write Header
164 int fileBytes = m_csv.WriteHeaderToFile(log.pFile, log.dataId);
165
166 // File byte size
167 log.fileSize = fileBytes;
168 m_logSize += fileBytes;
169 }
170
171 if (log.pFile)
172 {
173#if LOG_DEBUG_FILE_WRITE
174 printf("cDeviceLogCSV::OpenNewFile %s %s\n", (readonly?"read":"write"), fileName.c_str());
175#endif
176 return true;
177 }
178 else
179 {
180#if LOG_DEBUG_FILE_WRITE
181 printf("cDeviceLogCSV::OpenNewFile FAILED to open save file: %s\n", fileName.c_str());
182#endif
183 m_logs.erase(log.dataId);
184 return false;
185 }
186}

189bool cDeviceLogCSV::GetNextLineForFile(cCsvLog& log)
190{
191 if (log.pFile == NULL)
192 {
193 return false;
194 }
195 char lineBuffer[8192];
196 if (fgets(lineBuffer, _ARRAY_BYTE_COUNT(lineBuffer), log.pFile) == NULL)
197 {
198 return false;
199 }
200 log.nextLine = lineBuffer;
201 size_t index = log.nextLine.find(',', 0);
202 if (index < log.nextLine.size())
203 {
204 log.orderId = strtoull(log.nextLine.c_str(), NULL, 10);
205 }
206 else
207 {
208 return false;
209 }
210 return true;
211}

214bool cDeviceLogCSV::SaveData(p_data_hdr_t* dataHdr, const uint8_t* dataBuf)
215{
216 cDeviceLog::SaveData(dataHdr, dataBuf);
217
218 // Reference current log
219 cCsvLog& log = m_logs[dataHdr->id];
220 log.dataId = dataHdr->id;
221
222 // Create first file it it doesn't exist, return out if failure
223 if (log.pFile == NULL && (!OpenNewFile(log, false) || log.pFile == NULL))
224 {
225 return false;
226 }
227 else if (dataHdr->id == DID_DEV_INFO)
228 {
229 memcpy(&m_devInfo, dataBuf, sizeof(dev_info_t));
230 }
231
232 // Write date to file
233 int nBytes = m_csv.WriteDataToFile(m_nextId++, log.pFile, *dataHdr, dataBuf);
234 if (ferror(log.pFile) != 0)
235 {
236 return false;
237 }
238
239 // File byte size
240 log.fileSize += nBytes;
241 m_logSize += nBytes;
242
243 if (log.fileSize >= m_maxFileSize)
244 {
245 // Close existing file
246 fclose(log.pFile);
247 log.pFile = NULL;
248 log.fileSize = 0;
249 }
250
251 return true;
252}

255p_data_t* cDeviceLogCSV::ReadData()
256{
257
258tryAgain:
259 if (m_logs.size() == 0)
260 {
261 return NULL;
262 }
263
264 p_data_t* data = NULL;
265 cCsvLog* nextLog = NULL;
266 uint64_t nextId = ULLONG_MAX;
267 for (map<uint32_t, cCsvLog>::iterator i = m_logs.begin(); i != m_logs.end(); )
268 {
269 if (i->second.orderId < nextId)
270 {
271 nextLog = &i->second;
272 nextId = i->second.orderId;
273 }
274 i++;
275 }
276 if (nextLog == NULL)
277 {
278 assert(false);
279 return NULL;
280 }
281 if ((data = ReadDataFromFile(*nextLog)) == NULL)
282 {
283 goto tryAgain;
284 }
285
286 cDeviceLog::OnReadData(data);
287 return data;
288}

291p_data_t* cDeviceLogCSV::ReadDataFromFile(cCsvLog& log)
292{
293 if (log.pFile == NULL)
294 {
295 assert(false);
296 return NULL;
297 }
298 m_dataBuffer.hdr.id = log.dataId;
299 m_dataBuffer.hdr.size = log.dataSize;
300 if (m_csv.StringCSVToData(log.nextLine, m_dataBuffer.hdr, m_dataBuffer.buf, _ARRAY_BYTE_COUNT(m_dataBuffer.buf), log.columnHeaders))
301 {
302 if (m_dataBuffer.hdr.id == DID_DEV_INFO)
303 {
304 memcpy(&m_devInfo, m_dataBuffer.buf, sizeof(dev_info_t));
305 }
306 log.nextLine.clear();
307 while (!GetNextLineForFile(log) && OpenNewFile(log, true)) {}
308 return &m_dataBuffer;
309 }
310 log.nextLine.clear();
311 while (!GetNextLineForFile(log) && OpenNewFile(log, true)) {}
312 return NULL;
313}
