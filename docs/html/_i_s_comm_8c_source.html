92 unsigned int getBitsAsUInt32(const unsigned char* buffer, unsigned int pos, unsigned int len)
93 {
94  unsigned int bits = 0;
95  for (unsigned int i = pos; i < pos + len; i++)
96  {
97  bits = (bits << 1) + ((buffer[i / 8] >> (7 - i % 8)) & 1u);
98  }
99  return bits;
100 }

156 static void swapPacket(packet_t* pkt)
157 {
158  if (pkt->hdr.flags & CM_PKT_FLAGS_RAW_DATA_NO_SWAP)
159  {
160  if ((pkt->hdr.pid == PID_DATA || pkt->hdr.pid == PID_SET_DATA) && pkt->body.size >= sizeof(p_data_hdr_t))
161  {
162  // swap the data header only
163  flipEndianess32(pkt->body.ptr, sizeof(p_data_hdr_t));
164  }
165  }
166  else if (pkt->body.size < sizeof(p_data_hdr_t) || (pkt->hdr.pid != PID_DATA && pkt->hdr.pid != PID_SET_DATA))
167  {
168  // swap entire packet, not a data packet
169  flipEndianess32(pkt->body.ptr, pkt->body.size);
170  }
171  else
172  {
173  // swap header
174  flipEndianess32(pkt->body.ptr, sizeof(p_data_hdr_t));
175 
176  // get header
177  p_data_hdr_t* dataHdr = (p_data_hdr_t*)pkt->body.ptr;
178 
179  // if dev_info_t, swap only the uint32 fields, this data structure is handled special as it contains char[] arrays and uint32_t in the same struct
180  if (dataHdr->id == DID_DEV_INFO && pkt->body.size == sizeof(p_data_hdr_t) + sizeof(dev_info_t))
181  {
182  // swap only the pieces that need swapping
183  dev_info_t* devInfo = (dev_info_t*)(pkt->body.ptr + sizeof(p_data_hdr_t));
184  devInfo->buildNumber = SWAP32(devInfo->buildNumber);
185  devInfo->repoRevision = SWAP32(devInfo->repoRevision);
186  devInfo->serialNumber = SWAP32(devInfo->serialNumber);
187  }
188  else if (dataIdShouldSwap(dataHdr->id))
189  {
190  // swap entire packet body
191  flipEndianess32(pkt->body.ptr + sizeof(p_data_hdr_t), pkt->body.size - sizeof(p_data_hdr_t));
192 
193  // flip doubles
194  uint16_t* offsets;
195  uint16_t offsetsLength;
196  uint8_t* dataBuf = pkt->body.ptr + sizeof(p_data_hdr_t);
197 
198  // flip doubles back if needed
199  if ((offsets = getDoubleOffsets(dataHdr->id, &offsetsLength)))
200  {
201  flipDoubles(dataBuf, dataHdr->size, dataHdr->offset, offsets, offsetsLength);
202  }
203 
204  // flip strings back if needed
205  if ((offsets = getStringOffsetsLengths(dataHdr->id, &offsetsLength)))
206  {
207  flipStrings(dataBuf, dataHdr->size, dataHdr->offset, offsets, offsetsLength);
208  }
209  }
210  }

246 static protocol_type_t processInertialSensePkt(is_comm_instance_t* instance)
247 {
248  // Packet to decode into
249  packet_t *pkt = &(instance->pkt);
250  int pktSize = (int)(instance->buf.scan - instance->buf.head);
251  uint8_t* head;
252 
253  // Set location where to decode packet.
254  if(instance->altDecodeBuf)
255  { // Decode to alternate buffer
256  memcpy(instance->altDecodeBuf, instance->buf.head, pktSize);
257  head = pkt->body.ptr = instance->altDecodeBuf;
258  }
259  else
260  { // Decode packet in place on top of the receive buffer to save memory
261  head = instance->buf.head;
262  instance->pkt.body.ptr = instance->buf.start;
263  }
264 
265  instance->pktPtr = instance->buf.head;
266  reset_parser(instance);
267 
268  if (is_decode_binary_packet(pkt, head, pktSize) == 0)
269  {
270  instance->ackNeeded = 0;
271  instance->dataPtr = NULL;
272 
273  switch(pkt->hdr.pid)
274  {
275  case PID_SET_DATA:
276  case PID_DATA:
277  instance->dataHdr = *((p_data_hdr_t*)pkt->body.ptr);
278 
279  // ensure offset and size are in bounds - check the size independent of offset because the size could be a
280  // negative number in case of corrupt data
281  if (instance->dataHdr.id > DID_NULL &&
282 // instance->dataHdr.id < DID_COUNT && // Commented out to allow support for Luna EVB data sets
283  instance->dataHdr.size <= MAX_DATASET_SIZE //&&
284 // instance->dataHdr.offset <= MAX_DATASET_SIZE &&
285 // instance->dataHdr.offset + instance->dataHdr.size <= MAX_DATASET_SIZE
286  )
287  {
288  if(pkt->hdr.pid==PID_SET_DATA)
289  { // acknowledge valid data received
290  instance->ackNeeded = PID_ACK;
291  }
292 
293  // Update data pointer
294  instance->dataPtr = pkt->body.ptr + sizeof(p_data_hdr_t);
295  return _PTYPE_INERTIAL_SENSE_DATA;
296  }
297  else
298  { // negative acknowledge data received
299  instance->ackNeeded = PID_NACK;
300  }
301  break;
302 
303  case PID_GET_DATA:
304  // copy the requested data set info
305  instance->dataHdr = *((p_data_hdr_t*)pkt->body.ptr);
306 
307  if (instance->dataHdr.id > DID_NULL &&
308  instance->dataHdr.size <= MAX_DATASET_SIZE )
309  {
310  // Update data pointer
311  instance->dataPtr = pkt->body.ptr + sizeof(p_data_hdr_t);
312  return _PTYPE_INERTIAL_SENSE_CMD;
313  }
314  break;
315 
316  case PID_STOP_BROADCASTS_ALL_PORTS:
317  case PID_STOP_DID_BROADCAST:
318  case PID_STOP_BROADCASTS_CURRENT_PORT:
319  return _PTYPE_INERTIAL_SENSE_CMD;
320  case PID_ACK:
321  case PID_NACK:
322  return _PTYPE_INERTIAL_SENSE_ACK;
323  }
324  }
325 
326  // Invalid data or checksum failure.
327  instance->rxErrorCount++;
328  return _PTYPE_PARSE_ERROR;
329 }

331 static protocol_type_t processAsciiPkt(is_comm_instance_t* instance)
332 {
333  uint8_t* head = instance->buf.head;
334  reset_parser(instance);
335 
336  // calculate checksum, if pass return special data id
337  if (instance->buf.scan - head > 7)
338  {
339  // parse out checksum, put in temp null terminator
340  uint8_t tmp = *(instance->buf.scan - 2);
341  *(instance->buf.scan - 2) = 0;
342  int actualCheckSum = (int)strtol((const char*)instance->buf.scan - 4, 0, 16);
343  *(instance->buf.scan - 2) = tmp;
344  int dataCheckSum = 0;
345  for (uint8_t* ptr = head + 1, *ptrEnd = instance->buf.scan - 5; ptr < ptrEnd; ptr++)
346  {
347  dataCheckSum ^= (int)*ptr;
348  }
349  if (actualCheckSum == dataCheckSum)
350  { // valid ASCII Data
351  // Update data pointer and info
352  instance->dataPtr = instance->pktPtr = head;
353  instance->dataHdr.id = 0;
354  instance->dataHdr.size = (uint32_t)(instance->buf.scan - head);
355  instance->dataHdr.offset = 0;
356  return _PTYPE_ASCII_NMEA;
357  }
358  }
359 
360  // Invalid data or checksum failure.
361  instance->rxErrorCount++;
362  return _PTYPE_PARSE_ERROR;
363 }

365 static protocol_type_t processUbloxByte(is_comm_instance_t* instance)
366 {
367  switch (instance->parseState)
368  {
369  case 1: // preamble 2
370  if (*(instance->buf.scan - 1) != UBLOX_START_BYTE2)
371  {
372  // corrupt data
373  instance->rxErrorCount++;
374  reset_parser(instance);
375  return _PTYPE_PARSE_ERROR;
376  }
377  // fall through
378  case 2: // class id
379  // fall through
380  case 3: // message id
381  // fall through
382  case 4: // length byte 1
383  // fall through
384  case 0:
385  instance->parseState++;
386  break;
387 
388  case 5: // length byte 2
389  {
390  uint32_t len = BE_SWAP16(*((uint16_t*)(void*)(instance->buf.scan - 2)));
391 
392  // if length is greater than available buffer, we cannot parse this ublox packet - ublox header is 6 bytes
393  if (len > instance->buf.size - 6)
394  {
395  instance->rxErrorCount++;
396  reset_parser(instance);
397  return _PTYPE_PARSE_ERROR;
398  }
399  instance->parseState = -((int32_t)len + 2);
400  }
401  break;
402 
403  default:
404  if (++instance->parseState == 0)
405  {
406  // end of ublox packet, if checksum passes, send the external id
407  instance->hasStartByte = 0;
408  uint8_t actualChecksum1 = *(instance->buf.scan - 2);
409  uint8_t actualChecksum2 = *(instance->buf.scan - 1);
410  uint8_t calcChecksum1 = 0;
411  uint8_t calcChecksum2 = 0;
412 
413  // calculate checksum, skipping the first two preamble bytes and the last two bytes which are the checksum
414  for (uint8_t* ptr = instance->buf.head + 2, *ptrEnd = instance->buf.scan - 2; ptr < ptrEnd; ptr++)
415  {
416  calcChecksum1 += *ptr;
417  calcChecksum2 += calcChecksum1;
418  }
419  if (actualChecksum1 == calcChecksum1 && actualChecksum2 == calcChecksum2)
420  { // Checksum passed - Valid ublox packet
421  // Update data pointer and info
422  instance->dataPtr = instance->buf.head;
423  instance->dataHdr.id = 0;
424  instance->dataHdr.size = (uint32_t)(instance->buf.scan - instance->buf.head);
425  instance->dataHdr.offset = 0;
426  instance->pktPtr = instance->buf.head;
427  reset_parser(instance);
428  return _PTYPE_UBLOX;
429  }
430  else
431  { // Checksum failure
432  instance->rxErrorCount++;
433  reset_parser(instance);
434  return _PTYPE_PARSE_ERROR;
435  }
436  }
437  }
438 
439  return _PTYPE_NONE;
440 }

443 {
444  switch (instance->parseState)
445  {
446  case 0:
447  case 1:
448  instance->parseState++;
449  break;

451  case 2:
452  {
453  uint32_t msgLength = getBitsAsUInt32(instance->buf.head, 14, 10);

499 int is_comm_free(is_comm_instance_t* instance)
500 {
501 // if (instance == 0 || instance->buf.start == 0)
502 // {
503 // return -1;
504 // }
505 
506  is_comm_buffer_t *buf = &(instance->buf);
507 
508  int bytesFree = (int)(buf->end - buf->tail);
509 
510  // if we are out of free space, we need to either move bytes over or start over
511  if (bytesFree == 0)
512  {
513  if ((int)(buf->head - buf->start) < (int)(buf->size / 3)) // if ring buffer start index is less than this and no space is left, clear the entire ring buffer
514  { // we will be hung unless we flush the ring buffer, we have to drop bytes in this case and the caller
515  // will need to resend the data
516  buf->head = buf->start;
517  buf->tail = buf->start;
518  buf->scan = buf->start;
519  }
520  else
521  { // shift over the remaining data in the hopes that we will get a valid packet by appending the next read call
522  memmove(buf->start, buf->head, buf->tail - buf->head);
523  int shift = (int)(buf->head - buf->start);
524  buf->head -= shift;
525  buf->tail -= shift;
526  buf->scan -= shift;
527  }
528 
529  // re-calculate free byte count
530  bytesFree = (int)(buf->end - buf->tail);
531  }
532 
533  return bytesFree;
534 }

744 void is_decode_binary_packet_footer(packet_ftr_t* ftr, uint8_t* ptrSrc, uint8_t** ptrSrcEnd, uint32_t* checksum)
745 {
746  int state = 0;
747  uint8_t* currentPtr = (*ptrSrcEnd) - 1;
748  memset(ftr, 0, sizeof(uint32_t));
749 
750  // we need a state machine to ensure we don't overrun ptrSrcEnd
751  while (state != 7 && currentPtr > ptrSrc)
752  {
753  switch (state)
754  {
755  case 0: // packet end byte
756  ftr->stopByte = *currentPtr--;
757  state = 1;
758  break;
759 
760  case 1: // packet checksum 1
761  ftr->cksum1 = *currentPtr--;
762  state = (3 - (*currentPtr == PSC_RESERVED_KEY));
763  break;
764 
765  case 2: // packet checksum 1 is encoded
766  ftr->cksum1 = ~ftr->cksum1;
767  currentPtr--;
768  state = 3;
769  break;
770 
771  case 3: // packet checksum 2
772  ftr->cksum2 = *currentPtr--;
773  state = (5 - (*currentPtr == PSC_RESERVED_KEY));
774  break;
775 
776  case 4: // packet checksum 2 is encoded
777  ftr->cksum2 = ~ftr->cksum2;
778  currentPtr--;
779  state = 5;
780  break;
781 
782  case 5: // packet checksum 3
783  ftr->cksum3 = *currentPtr;
784  state = (7 - (*(currentPtr - 1) == PSC_RESERVED_KEY));
785  break;
786 
787  case 6: // packet checksum 3 is encoded
788  ftr->cksum3 = ~ftr->cksum3;
789  currentPtr--;
790  state = 7;
791  break;
792 
793  default:
794  break;
795  }
796  }
797  *ptrSrcEnd = currentPtr;
798  *checksum = ((uint32_t)ftr->cksum1) | (0x0000FF00 & ((uint32_t)ftr->cksum2 << 8)) | (0x00FF0000 & ((uint32_t)ftr->cksum3 << 16));
799 }

801 int is_decode_binary_packet_byte(uint8_t** _ptrSrc, uint8_t** _ptrDest, uint32_t* checksum, uint32_t shift)
802 {
803  uint8_t* ptrSrc = *_ptrSrc;
804 
805  // packet id byte
806  uint32_t val = *ptrSrc++;
807  switch (val)
808  {
809  case PSC_ASCII_START_BYTE:
810  case PSC_ASCII_END_BYTE:
811  case PSC_START_BYTE:
812  case PSC_END_BYTE:
813  case RTCM3_START_BYTE:
814  case UBLOX_START_BYTE1:
815  // corrupt data
816  return -1;
817 
818  case PSC_RESERVED_KEY:
819  // skip special byte
820  val = (~(*ptrSrc++) & 0x000000FF);
821  // fall through
822  default:
823  *checksum ^= (val << shift);
824  *((*_ptrDest)++) = (uint8_t)val;
825  }
826  *_ptrSrc = ptrSrc;
827 
828  return 0;
829 }

831 int is_encode_binary_packet(void* srcBuffer, unsigned int srcBufferLength, packet_hdr_t* hdr, uint8_t additionalPktFlags, void* encodedPacket, int encodedPacketLength)
832 {
833  // Ensure data size is small enough, assuming packet size could double after encoding.
834  if (srcBufferLength > MAX_PKT_BODY_SIZE)
835  {
836  return -1;
837  }
838 
839  // Update Packet Counter
840  uint8_t* ptrSrc;
841  uint8_t* ptrSrcEnd;
842  uint8_t* ptrDest = (uint8_t*)encodedPacket;
843  uint8_t* ptrDestEnd = ptrDest + encodedPacketLength;
844  uint32_t shifter = 0;
845  uint32_t checkSumValue = CHECKSUM_SEED;
846  uint32_t val;
847 
848  if (ptrDest >= ptrDestEnd)
849  {
850  return -1;
851  }
852  // Packet header -------------------------------------------------------------------------------------------
853  *ptrDest++ = PSC_START_BYTE;
854 
855  // PID
856  if (ptrDest >= ptrDestEnd)
857  {
858  return -1;
859  }
860  val = hdr->pid;
861  ptrDest = encodeByteAddToBuffer(val, ptrDest);
862  checkSumValue ^= val;
863 
864  // Counter
865  if (ptrDest >= ptrDestEnd)
866  {
867  return -1;
868  }
869  val = hdr->counter;
870  ptrDest = encodeByteAddToBuffer(val, ptrDest);
871  checkSumValue ^= (val << 8);
872 
873  // Flags
874  if (ptrDest >= ptrDestEnd)
875  {
876  return -1;
877  }
878  val = hdr->flags | additionalPktFlags | CPU_IS_LITTLE_ENDIAN | CM_PKT_FLAGS_CHECKSUM_24_BIT;
879  ptrDest = encodeByteAddToBuffer(val, ptrDest);
880  checkSumValue ^= (val << 16);
881 
882  // Packet body ----------------------------------------------------------------------------------------------
883  if (srcBufferLength > 0)
884  {
885  ptrSrc = (uint8_t*)srcBuffer;
886  ptrSrcEnd = ptrSrc + srcBufferLength;
887 
888  // copy body bytes, doing encoding and checksum
889  while (ptrSrc != ptrSrcEnd && ptrDest < ptrDestEnd)
890  {
891  val = *ptrSrc++;
892  checkSumValue ^= (val << shifter);
893 
894  // increment shifter
895  shifter += 8;
896 
897  // reset if shifter equals 24
898  shifter *= (shifter != 24);
899 
900  ptrDest = encodeByteAddToBuffer(val, ptrDest);
901  }
902  }
903 
904  // footer ----------------------------------------------------------------------------------------------------
905 
906  // checksum byte 3
907  if (ptrDest >= ptrDestEnd)
908  {
909  return -1;
910  }
911  val = (uint8_t)((checkSumValue >> 16) & 0xFF);
912  ptrDest = encodeByteAddToBuffer(val, ptrDest);
913 
914  // checksum byte 2
915  if (ptrDest >= ptrDestEnd)
916  {
917  return -1;
918  }
919  val = (uint8_t)(checkSumValue >> 8) & 0xFF;
920  ptrDest = encodeByteAddToBuffer(val, ptrDest);
921 
922  // checksum byte 1
923  if (ptrDest >= ptrDestEnd)
924  {
925  return -1;
926  }
927  val = (uint8_t)(checkSumValue & 0xFF);
928  ptrDest = encodeByteAddToBuffer(val, ptrDest);
929 
930  // packet end byte
931  if (ptrDest >= ptrDestEnd)
932  {
933  return -1;
934  }
935  *ptrDest++ = PSC_END_BYTE;
936  return (int)(ptrDest - (uint8_t*)encodedPacket);
937 }

940 int is_decode_binary_packet(packet_t* pkt, unsigned char* pbuf, int pbufSize)
941 {

943  // all other data can potentially be garbage
944  if (pbufSize < 8)
945  {
946  // corrupt data
947  return -1;
948  }
949 
950  // decode the body and calculate checksum
951  uint8_t* ptrSrc = pbuf;
952  uint8_t* ptrSrcEnd = pbuf + pbufSize;
953  packet_ftr_t ftr;
954  uint32_t actualCheckSumValue;
955 
956  is_decode_binary_packet_footer(&ftr, ptrSrc, &ptrSrcEnd, &actualCheckSumValue);
957  uint32_t shifter = 0;
958  uint32_t checkSumValue = CHECKSUM_SEED;
959 
960  // start packet byte
961  uint8_t* ptrDest = (uint8_t*)&pkt->hdr;
962  *ptrDest++ = *ptrSrc++;
963 
964  if
965  (
966  // packet id
967  is_decode_binary_packet_byte(&ptrSrc, &ptrDest, &checkSumValue, 0) ||
968 
969  // packet counter
970  is_decode_binary_packet_byte(&ptrSrc, &ptrDest, &checkSumValue, 8) ||
971

973  is_decode_binary_packet_byte(&ptrSrc, &ptrDest, &checkSumValue, 16)
974  )
975  {
976  return -1;
977  }
978 
979  // decode the body - start shift 0
980  ptrDest = pkt->body.ptr;
981  while (ptrSrc < ptrSrcEnd)
982  {
983  if (is_decode_binary_packet_byte(&ptrSrc, &ptrDest, &checkSumValue, shifter))
984  {
985  return -1;
986  }
987 
988  shifter += 8;
989 
990  // reset if shifter equals 24
991  shifter *= (shifter != 24);
992  }
993 
994  if (actualCheckSumValue != checkSumValue)
995  {
996  // corrupt data
997  return -1;
998  }
999

1003  return -1;
