194 void InertialSense::LoggerThread(void* info)
195 {
196  bool running = true;
197  InertialSense* inertialSense = (InertialSense*)info;
198 
199  // gather up packets in memory
200  map<int, vector<p_data_t>> packets;
201 
202  while (running)
203  {
204  SLEEP_MS(2);
205  {
206  // lock so we can read and clear m_logPackets
207  cMutexLocker logMutexLocker(&inertialSense->m_logMutex);
208  for (map<int, vector<p_data_t>>::iterator i = inertialSense->m_logPackets.begin(); i != inertialSense->m_logPackets.end(); i++)
209  {
210  packets[i->first] = i->second;
211  }
212 
213  // clear shared memory
214  inertialSense->m_logPackets.clear();
215 
216  // update running state
217  running = inertialSense->m_logger.Enabled();
218  }
219 
220  if (running)
221  {
222  // log the packets
223  for (map<int, vector<p_data_t>>::iterator i = packets.begin(); i != packets.end(); i++)
224  {
225  for (size_t j = 0; j < i->second.size(); j++)
226  {
227  if (!inertialSense->m_logger.LogData(i->first, &i->second[j].hdr, i->second[j].buf))
228  {
229  // Failed to write to log
230  SLEEP_MS(20);
231  }
232  }
233 
234  // clear all log data for this pHandle
235  i->second.clear();
236  }
237  }
238  }
239 }

289 bool InertialSense::OpenServerConnection(const string& connectionString)
290 {
291  bool opened = false;
292 
293  // if no serial connection, fail
294  if (!IsOpen())
295  {
296  return opened;
297  }
298 
299  CloseServerConnection();
300  vector<string> pieces;
301  SplitString(connectionString, ":", pieces);
302  if (pieces.size() < 3)
303  {
304  return opened;
305  }
306  else if (pieces[0] == "RTCM3")
307  {
308  if (pieces[1] == "SERIAL")
309  {
310  if (pieces.size() < 4)
311  {
312  return opened;
313  }
314  else if ((opened = (m_serialServer.Open(pieces[2].c_str(), atoi(pieces[3].c_str())))))
315  {
316  m_clientStream = &m_serialServer;
317  }
318  }
319  else
320  {
321  opened = (m_tcpClient.Open(pieces[1], atoi(pieces[2].c_str())) == 0);
322  string url = (pieces.size() > 3 ? pieces[3] : "");
323  string user = (pieces.size() > 4 ? pieces[4] : "");
324  string pwd = (pieces.size() > 5 ? pieces[5] : "");
325  m_tcpClient.HttpGet(url, "InertialSense", user, pwd);
326  }
327  m_parser = cGpsParser::CreateParser(GpsParserTypeRtcm3, this);
328  }
329  else if (pieces[0] == "IS")
330  {
331  opened = (m_tcpClient.Open(pieces[1], atoi(pieces[2].c_str())) == 0);
332  m_parser = cGpsParser::CreateParser(GpsParserTypeInertialSense, this);
333  }
334  else if (pieces[0] == "UBLOX")
335  {
336  opened = (m_tcpClient.Open(pieces[1], atoi(pieces[2].c_str())) == 0);
337  m_parser = cGpsParser::CreateParser(GpsParserTypeUblox, this);
338  }
339 
340  if (opened)
341  {
342  // configure as RTK rover
343  uint32_t cfgBits = SYS_CFG_BITS_RTK_ROVER;
344  for (size_t i = 0; i < m_comManagerState.serialPorts.size(); i++)
345  {
346  sendDataComManager((int)i, DID_FLASH_CONFIG, &cfgBits, sizeof(cfgBits), OFFSETOF(nvm_flash_cfg_t, sysCfgBits));
347  }
348  if (m_clientStream == NULLPTR)
349  {
350  m_clientStream = &m_tcpClient;
351  }
352  }
353 
354  return opened;
355 }

370 bool InertialSense::CreateHost(const string& ipAndPort)
371 {
372  // if no serial connection, fail
373  if (!IsOpen())
374  {
375  return false;
376  }
377 
378  CloseServerConnection();
379  size_t colon = ipAndPort.find(':', 0);
380  if (colon == string::npos)
381  {
382  return false;
383  }
384  StopBroadcasts();
385  string host = ipAndPort.substr(0, colon);
386  string portString = ipAndPort.substr(colon + 1);
387  int port = (int)strtol(portString.c_str(), NULLPTR, 10);
388  bool opened = (m_tcpServer.Open(host, port) == 0);
389  if (opened)
390  {
391  // configure as RTK base station
392  uint32_t cfgBits = SYS_CFG_BITS_RTK_BASE_STATION;
393  sendDataComManager(0, DID_FLASH_CONFIG, &cfgBits, sizeof(cfgBits), OFFSETOF(nvm_flash_cfg_t, sysCfgBits));
394  cfgBits = 1; // raw messages
395  sendDataComManager(0, DID_CONFIG, &cfgBits, sizeof(cfgBits), OFFSETOF(config_t, msgCfgBits));
396  }
397  return opened;
398 }

410 bool InertialSense::Update()
411 {
412  uint8_t buffer[4096];
413  if (m_clientStream != NULLPTR)
414  {
415  int count = m_clientStream->Read(buffer, sizeof(buffer));
416  if (count > 0)
417  {
418  m_clientServerByteCount += count;
419  m_parser->Write(buffer, count);
420  }
421  }
422  else if (m_tcpServer.IsOpen() && m_comManagerState.serialPorts.size() != 0)
423  {
424  // as a tcp server, only the first serial port is read from
425  int count = serialPortRead(&m_comManagerState.serialPorts[0], buffer, sizeof(buffer));
426  if (count > 0)
427  {
428  // forward data on to connected clients
429  m_tcpServer.Write(buffer, count);
430  m_clientServerByteCount += count;
431  }
432  m_tcpServer.Update();
433  }
434  stepComManager();
435 
436  // if any serial ports have closed, shut everything down
437  for (size_t i = 0; i < m_comManagerState.serialPorts.size(); i++)
438  {
439  if (!serialPortIsOpen(&m_comManagerState.serialPorts[i]))
440  {
441  Close();
442  return false;
443  }
444  }
445  return true;
446 }

547 bool InertialSense::BootloadFile(const string& comPort, const string& fileName, pfnBootloadProgress uploadProgress, pfnBootloadProgress verifyProgress, char* errorBuffer, int errorBufferLength)
548 {
549  // test file exists
550  {
551  ifstream tmpStream(fileName);
552  if (!tmpStream.good())
553  {
554  if (errorBuffer != NULLPTR)
555  {
556  SNPRINTF(errorBuffer, errorBufferLength, "Bootloader file does not exist");
557  }
558  return false;
559  }
560  }
561 
562  // for debug
563  // SERIAL_PORT_DEFAULT_TIMEOUT = 9999999;
564 
565  vector<string> portStrings;
566  vector<bootloader_state_t> state;
567  if (comPort == "*")
568  {
569  cISSerialPort::GetComPorts(portStrings);
570  }
571  else
572  {
573  SplitString(comPort, ",", portStrings);
574  }
575  state.resize(portStrings.size());
576 
577  // for each port requested, setup a thread to do the bootloader for that port
578  for (size_t i = 0; i < portStrings.size(); i++)
579  {
580  memset(&state[i].serial, 0, sizeof(state[i].serial));
581  serialPortSetPort(&state[i].serial, portStrings[i].c_str());
582  serialPortPlatformInit(&state[i].serial);
583  state[i].param.uploadProgress = uploadProgress;
584  state[i].param.verifyProgress = verifyProgress;
585  state[i].param.fileName = fileName.c_str();
586  state[i].param.error = errorBuffer;
587  state[i].param.errorLength = errorBufferLength;
588  state[i].param.port = &state[i].serial;
589  state[i].param.verifyFileName = NULLPTR;
590  state[i].param.flags.bitFields.enableVerify = (verifyProgress != NULLPTR);
591  state[i].thread = threadCreateAndStart(bootloaderThread, &state[i]);
592  }
593 
594  // wait for all threads to finish
595  for (size_t i = 0; i < state.size(); i++)
596  {
597  threadJoinAndFree(state[i].thread);
598  }
599 
600  // if any thread failed, we return failure
601  for (size_t i = 0; i < state.size(); i++)
602  {
603  if (!state[i].result)
604  {
605  return false;
606  }
607  }
608 
609  // all threads succeeded
610  return true;
611 }

613 size_t InertialSense::SplitString(const string& s, const string& delimiter, vector<string>& result)
614 {
615  result.clear();

617  size_t pos2;

647 bool InertialSense::OpenSerialPorts(const char* port, int baudRate)
648 {
649  Close();
650 
651  if (port == NULLPTR || validateBaudRate(baudRate) != 0)
652  {
653  return false;
654  }
655 
656  // split port on comma in case we need to open multiple serial ports
657  vector<string> ports;
658  size_t maxCount = UINT32_MAX;
659 
660  // handle wildcard, auto-detect serial ports
661  if (port[0] == '*')

664  if (port[1] != '\0')
665  {
666  maxCount = atoi(port + 1);
667  maxCount = (maxCount == 0 ? UINT32_MAX : maxCount);
668  }
669  }
670  else
671  {
672  // comma separated list of serial ports
673  SplitString(port, ",", ports);
674  }
675 
676  // open serial ports
677  for (size_t i = 0; i < ports.size(); i++)
678  {
679  serial_port_t serial;
680  memset(&serial, 0, sizeof(serial));
681  serialPortPlatformInit(&serial);
682  if (serialPortOpen(&serial, ports[i].c_str(), baudRate, 0) == 0)
683  {
684  // failed to open
685  serialPortClose(&serial);
686  }
687  else
688  {
689  m_comManagerState.serialPorts.push_back(serial);
690  }
691  }
692 
693  // setup com manager
694  initComManager((int)m_comManagerState.serialPorts.size(), 10, 10, 10, staticReadPacket, staticSendPacket, 0, staticProcessRxData, 0, 0);
695 
696  // re-initialize data sets
697  config_t configTemplate;
698  memset(&configTemplate, 0, sizeof(configTemplate));
699  m_comManagerState.config.resize(m_comManagerState.serialPorts.size(), configTemplate);
700  dev_info_t devInfoTemplate;
701  memset(&devInfoTemplate, 0, sizeof(devInfoTemplate));
702  m_comManagerState.devInfo.resize(m_comManagerState.serialPorts.size(), devInfoTemplate);
703  nvm_flash_cfg_t flashTemplate;
704  memset(&flashTemplate, 0, sizeof(flashTemplate));
705  m_comManagerState.flashConfig.resize(m_comManagerState.serialPorts.size(), flashTemplate);
706 
707  // negotiate baud rate by querying device info - don't return out until it negotiates or times out
708  // if the baud rate is already correct, the request for the message should succeed very quickly
709  time_t startTime = time(0);
710 
711  // try to auto-baud for up to 3 seconds, then abort if we didn't get a valid packet
712  // we wait until we get a valid serial number and manufacturer
713  while (!HasReceivedResponseFromAllDevices() && time(0) - startTime < 3)
714  {
715  for (size_t i = 0; i < m_comManagerState.serialPorts.size(); i++)
716  {
717  getDataComManager((int)i, DID_CONFIG, 0, 0, 0);
718  getDataComManager((int)i, DID_DEV_INFO, 0, 0, 0);
719  getDataComManager((int)i, DID_FLASH_CONFIG, 0, 0, 0);
720  }
721 
722  SLEEP_MS(12);
723  stepComManager();
724  SLEEP_MS(12);
725  }
726 
727  bool removedSerials = false;
728 
729  // remove each failed device where communications were not received
730  for (int i = (int)(m_comManagerState.devInfo.size() - 1); i >= 0; i--)
731  {
732  if (!HasReceivedResponseFromDevice(i))
733  {
734  RemoveDevice(i);
735  removedSerials = true;
736  }
737  }
738 
739  // if no devices left, all failed, we return failure
740  if (m_comManagerState.serialPorts.size() == 0)
741  {
742  Close();
743  return false;
744  }
745 
746  // remove ports if we are over max count
747  while (m_comManagerState.serialPorts.size() > maxCount)
748  {
749  RemoveDevice(m_comManagerState.serialPorts.size() - 1);
750  removedSerials = true;
751  }
752 
753  // setup com manager again if serial ports dropped out with new count of serial ports
754  if (removedSerials)
755  {
756  initComManager((int)m_comManagerState.serialPorts.size(), 10, 10, 10, staticReadPacket, staticSendPacket, 0, staticProcessRxData, 0, 0);
757  }
758 
759  return true;
760 }
