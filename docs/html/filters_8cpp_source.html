36 void init_iir_filter(iif_filter_t *f)
37 {
38  int gama;
39  int alpha;
40  float TsFc;
41 
42  if(f->opt.n_channels > MAX_NUMBER_IIR_CHANNELS)
43  {
44 // dg_printf("IIR channels exceeded max number: %d. Consider increasing max number", MAX_NUMBER_IIR_CHANNELS);
45 // exit(1);
46  }
47 
48  f->opt.bit_shift = (ACCUM_WORD_NBITS-1-f->opt.sig_word_nbits)/2;
49 
50  // Gama = adc_to_iir = alpha + beta
51  gama = 1<<f->opt.bit_shift;
52  TsFc = f->opt.Fc/f->opt.Fs;
53  alpha = (int)( TsFc/(1.0 + TsFc) * (float)gama );
54  f->opt.beta = gama - alpha;
55  // alpha_x allows us to combine the adc_to_iir and alpha multiplies into one step
56  f->opt.alpha_x = alpha<<f->opt.bit_shift;
57  f->opt.iir_to_adc = 1.0f/(float)gama;
58 }

61 void iir_filter_u16(iif_filter_t *f, unsigned short input[], float output[])
62 {
63  unsigned int i,j;
64 
65  for (j=0; j<f->opt.input_size; j+=f->opt.n_channels)
66  {
67  for (i=0; i<f->opt.n_channels; i++)
68  {
69  f->accum[i] = f->opt.beta*f->accum[i] + f->opt.alpha_x*((int)input[j+i]);
70  f->accum[i] >>= f->opt.bit_shift;

72  }
73

75  output[i] = f->opt.iir_to_adc*(float)f->accum[i];

306 void integrateDeltaThetaVelBortz(ixVector3 theta, ixVector3 vel, imus_t *imu, imus_t *imuLast, float Nsteps, float dti)
307 {
308  ixVector3 wb, ab, deltaW, deltaA, tmp1, tmp2, tmp3, tmp4;
309  float Kw, mag_theta2, mag_theta4, Kw0;
310 
311  // for jj = 1: Nint
312  // wb = W0 + (jj - 1) / Nint * (W1 - W0);
313  // ab = A0 + (jj - 1) / Nint * (A1 - A0);
314  // % Bortz's formula
315  // phi_dot = wb + 0.5 * cross(phi, wb) + 1 / 12 * cross(phi, cross(phi, wb));
316  // % Savage's formula (faster): approximates phi with gyro integral
317  // % phi_dot = Wi + 0.5 * cross(WintB, wb)
318  // phi = phi + phi_dot * dt / Nint;
319  // dv = dv + (ab + cross(phi, ab)) * dt / Nint;
320  // end
321 
322  sub_Vec3_Vec3(deltaW, imu->pqr, imuLast->pqr);
323  sub_Vec3_Vec3(deltaA, imu->acc, imuLast->acc);
324  Kw0 = 1.0f / 12.0f;
325  for (int jj = 0; jj < Nsteps; jj++)
326  {
327  float jjDivNsteps = ((float)jj) / Nsteps;
328 
329  // Current angular rate and acceleration vectors (linear interpolation)
330  for (int i = 0; i < 3; i++)
331  {
332  wb[i] = imuLast->pqr[i] + deltaW[i] * jjDivNsteps;
333  ab[i] = imuLast->acc[i] + deltaA[i] * jjDivNsteps;
334  }
335  // Coning and sculling integrals
336  cross_Vec3(tmp1, theta, wb);
337  cross_Vec3(tmp2, theta, tmp1);
338  cross_Vec3(tmp3, theta, ab);
339  cross_Vec3(tmp4, theta, tmp3);
340  mag_theta2 = dot_Vec3(theta);
341  mag_theta4 = mag_theta2 * mag_theta2;
342  Kw = Kw0 + mag_theta2 / 720.0f + mag_theta4 / 30240.0f + mag_theta4 * mag_theta2 / 1209600.0f;
343  for (int i = 0; i < 3; i++) {
344  theta[i] += (wb[i] + 0.5f * tmp1[i] + Kw * tmp2[i]) * dti;
345  vel[i] += (ab[i] + tmp3[i] + 0.5f * tmp4[i]) * dti;
346  }
347  }
348 }
