36void init_iir_filter(iif_filter_t *f)
37{
38 int gama;
39 int alpha;
40 float TsFc;
41
42 if(f->opt.n_channels > MAX_NUMBER_IIR_CHANNELS)
43 {
44// dg_printf("IIR channels exceeded max number: %d. Consider increasing max number", MAX_NUMBER_IIR_CHANNELS);
45// exit(1);
46 }
47
48 f->opt.bit_shift = (ACCUM_WORD_NBITS-1-f->opt.sig_word_nbits)/2;
49
50 // Gama = adc_to_iir = alpha + beta
51 gama = 1<<f->opt.bit_shift;
52 TsFc = f->opt.Fc/f->opt.Fs;
53 alpha = (int)( TsFc/(1.0 + TsFc) * (float)gama );
54 f->opt.beta = gama - alpha;
55 // alpha_x allows us to combine the adc_to_iir and alpha multiplies into one step
56 f->opt.alpha_x = alpha<<f->opt.bit_shift;
57 f->opt.iir_to_adc = 1.0f/(float)gama;
58}

61void iir_filter_u16(iif_filter_t *f, unsigned short input[], float output[])
62{
63 unsigned int i,j;
64
65 for (j=0; j<f->opt.input_size; j+=f->opt.n_channels)
66 {
67 for (i=0; i<f->opt.n_channels; i++)
68 {
69 f->accum[i] = f->opt.beta*f->accum[i] + f->opt.alpha_x*((int)input[j+i]);
70 f->accum[i] >>= f->opt.bit_shift;
71 }
72 }
73
74 for (i=0; i<f->opt.n_channels; i++)
75 output[i] = f->opt.iir_to_adc*(float)f->accum[i];
76}

111void running_mean_filter( float input[], float mean[], int arraySize, int sampleCount )
112{
113 int i;
114 float alpha;
115
116 if( sampleCount == 0 )
117 alpha = 1.0f;
118 else
119 alpha = 1.0f / (float)sampleCount;
120
121 // Find running average
122 for( i=0; i<arraySize; i++ )
123 mean[i] = (1.0f-alpha)*mean[i] + (alpha)*input[i];

172{
173 // Find best IMU value
174 if (di->imu1ok && di->imu2ok)
175 {
176 // Find average of two IMU samples
177 result->time = di->imu.time;
178
179 for (int i = 0; i < 3; i++)
180 {
181 result->I.pqr[i] = 0.5f * (di->imu.I[0].pqr[i] + di->imu.I[1].pqr[i]);
182 result->I.acc[i] = 0.5f * (di->imu.I[0].acc[i] + di->imu.I[1].acc[i]);
183 }
184 }
185 else if (di->imu1ok)
186 {
187 result->time = di->imu.time;
188 result->I = di->imu.I[0];
189 }
190 else if (di->imu2ok)
191 {
192 result->time = di->imu.time;
193 result->I = di->imu.I[1];

272float deltaThetaDeltaVelRiemannSum( preintegrated_imu_t *output, dual_imu_t *imu, dual_imu_t *imuLast )
273{
274 ixVector3 tmp3;
275 float dt = (float)(imu->time - imuLast->time);
276
277 // Use Riemann Sum integral
278 for( int i=0; i<2; i++)
279 {
280 // IMU 1 - Delta Theta
281 mul_Vec3_X( tmp3, imu->I[i].pqr, dt );
282 add_Vec3_Vec3( output->theta1, output->theta1, tmp3 );
283
284 // IMU 1 - Delta Velocity
285 mul_Vec3_X( tmp3, imu->I[i].acc, dt );
286 add_Vec3_Vec3( output->vel1, output->vel1, tmp3 );
287 }
288
289 // Update history
290 *imuLast = *imu;
291
292 return dt;
293}

322void integrateDeltaThetaVelBortz(ixVector3 theta, ixVector3 vel, imus_t *imu, imus_t *imuLast, float Nsteps, float dti)
323{
324 ixVector3 wb, ab, deltaW, deltaA, tmp1, tmp2, tmp3, tmp4;
325 float Kw, mag_theta2, mag_theta4, Kw0;
326
327 // for jj = 1: Nint
328 // wb = W0 + (jj - 1) / Nint * (W1 - W0);
329 // ab = A0 + (jj - 1) / Nint * (A1 - A0);
330 // % Bortz's formula
331 // phi_dot = wb + 0.5 * cross(phi, wb) + 1 / 12 * cross(phi, cross(phi, wb));
332 // % Savage's formula (faster): approximates phi with gyro integral
333 // % phi_dot = Wi + 0.5 * cross(WintB, wb)
334 // phi = phi + phi_dot * dt / Nint;
335 // dv = dv + (ab + cross(phi, ab)) * dt / Nint;
336 // end
337
338 sub_Vec3_Vec3(deltaW, imu->pqr, imuLast->pqr);
339 sub_Vec3_Vec3(deltaA, imu->acc, imuLast->acc);
340 Kw0 = 1.0f / 12.0f;
341 for (int jj = 0; jj < Nsteps; jj++)
342 {
343 float jjDivNsteps = ((float)jj) / Nsteps;
344
345 // Current angular rate and acceleration vectors (linear interpolation)
346 for (int i = 0; i < 3; i++)
347 {
348 wb[i] = imuLast->pqr[i] + deltaW[i] * jjDivNsteps;
349 ab[i] = imuLast->acc[i] + deltaA[i] * jjDivNsteps;
350 }
351 // Coning and sculling integrals
352 cross_Vec3(tmp1, theta, wb);
353 cross_Vec3(tmp2, theta, tmp1);
354 cross_Vec3(tmp3, theta, ab);
355 cross_Vec3(tmp4, theta, tmp3);
356 mag_theta2 = dot_Vec3(theta);
357 mag_theta4 = mag_theta2 * mag_theta2;
358 Kw = Kw0 + mag_theta2 / 720.0f + mag_theta4 / 30240.0f + mag_theta4 * mag_theta2 / 1209600.0f;
359 for (int i = 0; i < 3; i++) {
360 theta[i] += (wb[i] + 0.5f * tmp1[i] + Kw * tmp2[i]) * dti;
361 vel[i] += (ab[i] + tmp3[i] + 0.5f * tmp4[i]) * dti;
362 }
363 }
364}

367/* Direct numerical integration of coning and sculling integrals using Bortz's formula */
368float deltaThetaDeltaVelBortz(preintegrated_imu_t *output, dual_imu_t *imu, dual_imu_t *imuLast, int Nsteps, bool enableIMU1, bool enableIMU2)
369{
370 float dt = (float)(imu->time - imuLast->time);
371 float dti = dt / (float)Nsteps;
372
373 if (enableIMU1)
374 { // IMU 1
375 integrateDeltaThetaVelBortz(output->theta1, output->vel1, &(imu->I[0]), &(imuLast->I[0]), (float)Nsteps, (float)dti);
376 }
377
378 if (enableIMU2)
379 { // IMU 2
380 integrateDeltaThetaVelBortz(output->theta2, output->vel2, &(imu->I[1]), &(imuLast->I[1]), (float)Nsteps, (float)dti);
381 }
382
383 // Update history
384 *imuLast = *imu;
385
386 return dt;
387}
