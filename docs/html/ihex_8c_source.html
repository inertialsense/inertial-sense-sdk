30static int parse_hex_line(char* theline, uint8_t bytes[], int* addr, int* num, int* code)
31{
32 int sum, len, cksum;
33 char *ptr;
34
35 *num = 0;
36 if (theline[0] != ':') return 0;
37 if (strlen(theline) < 11) return 0;
38 ptr = theline+1;
39 if (!sscanf(ptr, "%02x", &len)) return 0;
40 ptr += 2;
41 if ( strlen(theline) < (11 + ((size_t)len * 2)) ) return 0;
42 if (!sscanf(ptr, "%04x", addr)) return 0;
43 ptr += 4;
44 /* printf("Line: length=%d Addr=%d\n", len, *addr); */
45 if (!sscanf(ptr, "%02x", code)) return 0;
46 ptr += 2;
47 sum = (len & 255) + ((*addr >> 8) & 255) + (*addr & 255) + (*code & 255);
48 while(*num != len)
49 {
50 if (!sscanf(ptr, "%02hhx", &bytes[*num])) return 0;
51 ptr += 2;
52 sum += bytes[*num] & 0xFF;
53 (*num)++;
54 if (*num >= 256) return 0;
55 }
56 if (!sscanf(ptr, "%02x", &cksum)) return 0;
57 if ( ((sum & 255) + (cksum & 255)) & 255 ) return 0; /* checksum error */
58 return 1;
59}

61static int ihex_load_section(FILE** ihex_file, ihex_image_section_t* section)
62{
63 char line[512]; // Max line length is 256
64 int addr, n, status;
65 uint8_t bytes[256];
66 int i, lineno=1;
67 int minaddr=65536, maxaddr=0;
68 bool alreadysetaddr = false;
69 bool eof = false;
70 long last_line;
71 uint32_t address = 0;
72
73 uint8_t* image_local = malloc(MAX_IHEX_SECTION_LEN);
74 memset(image_local, 0xFF, MAX_IHEX_SECTION_LEN); // Any unspecified bytes default to 0xFF
75
76 while (!feof(*ihex_file) && !ferror(*ihex_file)) {
77
78 // Open file and get a line
79 line[0] = '\0';
80
81 last_line = ftell(*ihex_file);
82 fgets(line, 512, *ihex_file);
83
84 // Turn end of line characters into cstring terminators
85 if (line[strlen(line) - 1] == '\n') line[strlen(line) - 1] = '\0';
86 if (line[strlen(line) - 1] == '\r') line[strlen(line) - 1] = '\0';
87
88 // Parse the line
89 if (parse_hex_line(line, bytes, &addr, &n, &status))
90 {
91 if (status == 0) // Copy data into memory
92 {
93 for (i = 0; i < n; i++)
94 {
95 image_local[addr] = bytes[i];
96 if (addr < minaddr) minaddr = addr;
97 if (addr > maxaddr) maxaddr = addr;
98 addr++;
99
100 if (addr > MAX_IHEX_SECTION_LEN) return -1; // Buffer exceeded
101 }
102 }
103 else if (status == 1) // End of file
104 {
105 eof = true;
106 break;
107 }
108 else if (status == 4) // Programming location. *should* also work if there are no code == 4 in the hex file
109 {
110 if (alreadysetaddr)
111 {
112 // Go back one line, so the next call to this function gets the address to write to
113 fseek(*ihex_file, last_line, SEEK_SET);
114
115 break; // Not finished with file, but reached end of sector
116 }
117
118 address = bytes[1] << 16 | bytes[0] << 24;
119 alreadysetaddr = true;
120 }
121 }
122 else
123 {
124 printf(" Error line: %d\n", lineno);
125 }
126
127 lineno++;
128 }
129
130 // Base the section length on what address the end is at, since hex file may skip over unused bytes in some lines
131 section->len = maxaddr - minaddr + 1;
132
133 // Allocate a buffer for the image to reside in
134 section->image = malloc(section->len);
135
136 if (section->image != NULL && section->len)
137 {
138 section->address = address;
139
140 // printf(" Loaded %d bytes between:", section->len);
141 // printf(" 0x%04X to 0x%04X at address:", minaddr, maxaddr);
142 // printf(" 0x%08X\n", section->address);
143
144 memcpy(section->image, &image_local[minaddr], section->len);
145
146 if(image_local) free(image_local);
147
148 return eof ? 1 : 0; // Return 1 if end of file reached
149 }
150
151 if(image_local) free(image_local);
152
153 return -1; // Malloc failed or nothing in sector
154}

156static void ihex_unload_section(ihex_image_section_t* section)
157{
158 if (section->image != NULL)
159 {
160 free(section->image);
161 section->image = NULL;
162 section->address = 0;
163 section->len = 0;

167size_t ihex_load_sections(const char* ihex_filename, ihex_image_section_t* image, size_t num_slots)
168{
169 FILE* ihex_file;
170 ihex_file = fopen(ihex_filename, "r");
171 if(ihex_file==NULL) return 0;
172
173 size_t iter = 0;
174 size_t numSections = 0;
175
176 // printf("Loading sections from %s\n", ihex_filename);
177
178 // Find sections
179 do {
180 int ret = ihex_load_section(&ihex_file, &image[iter]);
181
182 if (ret < 0) break; // Error
183 numSections++;
184 if (ret == 1) break; // Last sector in file found (EOF)
185 } while (++iter < num_slots);
186
187 // ACT ON SECTIONS HERE
188 fclose(ihex_file);
189
190 // printf("\n");
191
192 return numSections;
193}
