23 cISStream* cISClient::OpenConnectionToServer(const string& connectionString, bool *enableGpggaForwarding)
24 {
25  vector<string> pieces;
26  splitString(connectionString, ':', pieces);
27  if (pieces.size() < 4)
28  {
29  return NULLPTR;
30  }
31 
32  string type = pieces[0]; // TCP, SERIAL
33  string protocol = pieces[1]; // RTCM3, UBLOX, IS
34 
35  if (type == "SERIAL")
36  {
37  cISSerialPort *clientStream = new cISSerialPort();
38 
39  string portName = pieces[2]; // /dev/ttyACM0
40  string baudrate = pieces[3]; // 921600
41 
42  if (clientStream->Open(portName, atoi(baudrate.c_str())))
43  {
44  return clientStream;
45  }
46  }
47  else if(type == "TCP")
48  {
49  cISTcpClient *clientStream = new cISTcpClient();
50 
51  string host = (pieces[2].size()>0 ? pieces[2] : "127.0.0.1"); // ipAddr/URL
52  string port = (pieces[3]);
53  string subUrl = (pieces.size() > 4 ? pieces[4] : "");
54  string username = (pieces.size() > 5 ? pieces[5] : "");
55  string password = (pieces.size() > 6 ? pieces[6] : "");
56 
57  if (clientStream->Open(host, atoi(port.c_str()), 100) == -1)
58  {
59  return NULLPTR;
60  }
61 
62  if (subUrl.size() != 0)
63  { // Connect NTRIP if specified - https://igs.bkg.bund.de/root_ftp/NTRIP/documentation/NtripDocumentation.pdf
64  string userAgent = "NTRIP Inertial Sense"; // NTRIP standard requires "NTRIP" to be at the start of the User-Agent string.
65  clientStream->HttpGet(subUrl, userAgent, username, password);
66  if (enableGpggaForwarding != NULL)
67  {
68  *enableGpggaForwarding = true;
69  }
70  }
71 
72  return clientStream;
73  }
74 
75  return NULLPTR;
76 }
