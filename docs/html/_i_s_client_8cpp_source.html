20// connectionString usage:
21// [type]:[protocol]:[ip/url]:[port]:[mountpoint]:[username]:[password]
22// [TCP]:[RTCM3]:[ip/url]:[port]:[mountpoint]:[username]:[password]
23// [TCP]:[RTCM3]:[ip/url]:[port]
24// [SERIAL]:[RTCM3]:[serial port]:[baudrate]
25cISStream* cISClient::OpenConnectionToServer(const string& connectionString, bool *enableGpggaForwarding)
26{
27 vector<string> pieces;
28 splitString(connectionString, ':', pieces);
29 if (pieces.size() < 4)
30 {
31 return NULLPTR;
32 }
33
34 string type = pieces[0]; // TCP, SERIAL
35 string protocol = pieces[1]; // RTCM3, UBLOX, IS
36
37 if (type == "SERIAL")
38 {
39 cISSerialPort *clientStream = new cISSerialPort();
40
41 string portName = pieces[2]; // /dev/ttyACM0
42 string baudrate = pieces[3]; // 921600
43
44 if (clientStream->Open(portName, atoi(baudrate.c_str())))
45 {
46 return clientStream;
47 }
48 }
49 else if(type == "TCP")
50 {
51 cISTcpClient *clientStream = new cISTcpClient();
52
53 string host = (pieces[2].size()>0 ? pieces[2] : "127.0.0.1"); // ipAddr/URL
54 string port = (pieces[3]);
55 string subUrl = (pieces.size() > 4 ? pieces[4] : "");
56 string username = (pieces.size() > 5 ? pieces[5] : "");
57 string password = (pieces.size() > 6 ? pieces[6] : "");
58
59 if (clientStream->Open(host, atoi(port.c_str()), 100) == -1)
60 {
61 return NULLPTR;
62 }
63
64 if (subUrl.size() != 0)
65 { // Connect NTRIP if specified - https://igs.bkg.bund.de/root_ftp/NTRIP/documentation/NtripDocumentation.pdf
66 string userAgent = "NTRIP Inertial Sense"; // NTRIP standard requires "NTRIP" to be at the start of the User-Agent string.
67 clientStream->HttpGet(subUrl, userAgent, username, password);
68 if (enableGpggaForwarding != NULL)
69 {
70 *enableGpggaForwarding = true;
71 }
72 }
73
74 return clientStream;
75 }
76
77 return NULLPTR;
78}
