69bool cDeviceLogSerial::SaveData(p_data_hdr_t* dataHdr, const uint8_t* dataBuf)
70{
71 cDeviceLog::SaveData(dataHdr, dataBuf);
72
73 // Add serial number if available
74 if (dataHdr->id == DID_DEV_INFO && !copyDataPToStructP2(&m_devInfo, dataHdr, dataBuf, sizeof(dev_info_t)))
75 {
76 int start = dataHdr->offset;
77 int end = dataHdr->offset + dataHdr->size;
78 int snOffset = offsetof(dev_info_t, serialNumber);
79
80 // Did we really get the serial number?
81 if (start <= snOffset && (int)(snOffset + sizeof(uint32_t)) <= end)
82 {
83 m_chunk.m_hdr.devSerialNum = m_devInfo.serialNumber;
84 }
85 }
86
87 // Ensure data will fit in chunk. If not, create new chunk
88 int32_t dataBytes = sizeof(p_data_hdr_t) + dataHdr->size;
89 if (dataBytes > m_chunk.GetBuffFree())
90 {
91 // Save chunk to file and clear
92 if (!WriteChunkToFile())
93 {
94 return false;
95 }
96 else if (m_fileSize >= m_maxFileSize)
97 {
98 // Close existing file
99 CloseAllFiles();
100 }
101 }
102 // Add data header and data buffer to chunk
103 if (!m_chunk.PushBack((unsigned char*)dataHdr, sizeof(p_data_hdr_t), (unsigned char*)dataBuf, dataHdr->size))
104 {
105 return false;
106 }
107
108 return true;
109}

112bool cDeviceLogSerial::WriteChunkToFile()
113{
114 // Make sure we have data to write
115 if (m_chunk.GetDataSize() == 0)
116 {
117 return false;
118 }
119
120 // Create first file if it doesn't exist
121 if (m_pFile == NULLPTR)
122 {
123 OpenNewSaveFile();
124 }
125
126 // Validate file pointer
127 if (m_pFile == NULLPTR)
128 {
129 return false;
130 }
131
132 // Write chunk to file
133 int fileBytes = m_chunk.WriteToFile(m_pFile, 0);
134 if (!m_pFile->good())
135 {
136 return false;
137 }
138
139 // File byte size
140 m_fileSize += fileBytes;
141 m_logSize += fileBytes;
142
143 return true;
144}

167p_data_t* cDeviceLogSerial::ReadDataFromChunk()
168{
169 // Ensure chunk has data
170 if (m_chunk.GetDataSize() <= 0)
171 {
172 return NULL;
173 }
174
175 p_data_t* data = (p_data_t*)m_chunk.GetDataPtr();
176 int size = data->hdr.size + sizeof(p_data_hdr_t);
177 if (m_chunk.PopFront(size))
178 {
179 return data;
180 }
181 else
182 {
183 return NULL;
184 }
185}
