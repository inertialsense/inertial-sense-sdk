17 // Reversed bytes in a float.
18 // compiler will likely inline this as it's a tiny function
19 void flipFloat(uint8_t* ptr)
20 {
21  uint8_t tmp1 = *ptr++;
22  uint8_t tmp2 = *ptr++;
23  uint8_t tmp3 = *ptr++;
24  uint8_t tmp4 = *ptr;
25  *ptr-- = tmp1;
26  *ptr-- = tmp2;
27  *ptr-- = tmp3;
28  *ptr = tmp4;
29 }

31 float flipFloatCopy(float val)
32 {
33  float flippedFloat;
34  uint8_t* ptr = (uint8_t*)&val;
35  uint8_t* ptr2 = (uint8_t*)&flippedFloat;
36  uint8_t tmp1 = *ptr++;
37  uint8_t tmp2 = *ptr++;
38  uint8_t tmp3 = *ptr++;
39  uint8_t tmp4 = *ptr;
40  *ptr2++ = tmp4;
41  *ptr2++ = tmp3;
42  *ptr2++ = tmp2;
43  *ptr2 = tmp1;
44  return flippedFloat;
45 }

47 void flipDouble(void* ptr)
48 {
49  const uint32_t* w = (const uint32_t*)(ptr);
50  union
51  {
52  double v;
53  uint32_t w[2];
54  } u;
55  u.w[0] = w[1];
56  u.w[1] = w[0];
57  *(double*)ptr = u.v;
58 }

60 double flipDoubleCopy(double val)
61 {
62  union
63  {
64  double v;
65  uint32_t w[2];
66  } u1, u2;
67  u1.v = val;
68  u2.w[1] = SWAP32(u1.w[0]);
69  u2.w[0] = SWAP32(u1.w[1]);
70  return u2.v;
71 }

73 void flipEndianess32(uint8_t* data, int dataLength)
74 {
75  // data must be 4 byte aligned to swap endian-ness
76  if (dataLength & 0x00000003)
77  {
78  return;
79  }
80 
81  uint32_t* dataPtr = (void*)data;
82  uint32_t* dataPtrEnd = (void*)(data + dataLength);
83  uint32_t tmp;
84  while (dataPtr < dataPtrEnd)
85  {
86  tmp = *dataPtr;
87  *dataPtr++ = SWAP32(tmp);
88  }
89 }

91 void flipDoubles(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
92 {
93  uint16_t* doubleOffsetsEnd = offsets + offsetsLength;
94  int offsetToDouble;
95  int maxDoubleOffset = dataLength - 8;
96  int isDouble;
97  while (offsets < doubleOffsetsEnd)
98  {
99  offsetToDouble = (*offsets++);
100  isDouble = ((offsetToDouble & 0x8000) == 0);
101  offsetToDouble = (offsetToDouble & 0x7FFF) - offset;
102  if (offsetToDouble >= 0 && offsetToDouble <= maxDoubleOffset)
103  {
104  if (isDouble)
105  {
106  flipDouble(data + offsetToDouble);
107  }
108  else
109  {
110  uint64_t* ptr = (void*)(data + offsetToDouble);
111  *ptr = SWAP64(*ptr);
112  }
113  }
114  }
115 }

117 void flipStrings(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
118 {
119  uint16_t* stringOffsetsEnd = offsets + offsetsLength;
120  int offsetToString;
121  int lengthOfString;
122  int maxStringOffset;
123 
124  while (offsets < stringOffsetsEnd)
125  {
126  offsetToString = (*offsets++) - offset;
127  lengthOfString = (*offsets++);
128  maxStringOffset = dataLength - lengthOfString;
129  if (offsetToString >= 0 && offsetToString <= maxStringOffset)
130  {
131  flipEndianess32(data + offsetToString, lengthOfString);
132  }
133  }
134 }

579 uint32_t checksum32(const void* data, int count)
580 {
581  if (count < 1 || count % 4 != 0)
582  {
583  return 0;
584  }
585 
586  uint32_t checksum = 0;
587  uint32_t* dataPtr = (uint32_t*)data;
588  uint32_t* dataEnd = dataPtr + (count / 4);
589 
590  while (dataPtr < dataEnd)
591  {
592  checksum ^= *dataPtr++;
593  }
594 
595  return checksum;
596 }

654 void julianToDate(double julian, int32_t* year, int32_t* month, int32_t* day, int32_t* hour, int32_t* minute, int32_t* second, int32_t* millisecond)
655 {
656  double j1, j2, j3, j4, j5;
657  double intgr = floor(julian);
658  double frac = julian - intgr;
659  double gregjd = 2299161.0;
660  if (intgr >= gregjd)
661  {
662  //Gregorian calendar correction
663  double tmp = floor(((intgr - 1867216.0) - 0.25) / 36524.25);
664  j1 = intgr + 1.0 + tmp - floor(0.25 * tmp);
665  }
666  else
667  {
668  j1 = intgr;
669  }
670 
671  //correction for half day offset
672  double dayfrac = frac + 0.5;
673  if (dayfrac >= 1.0)
674  {
675  dayfrac -= 1.0;
676  ++j1;
677  }
678 
679  j2 = j1 + 1524.0;
680  j3 = floor(6680.0 + ((j2 - 2439870.0) - 122.1) / 365.25);
681  j4 = floor(j3 * 365.25);
682  j5 = floor((j2 - j4) / 30.6001);
683 
684  double d = floor(j2 - j4 - floor(j5 * 30.6001));
685  double m = floor(j5 - 1);
686  if (m > 12)
687  {
688  m -= 12;
689  }
690  double y = floor(j3 - 4715.0);
691  if (m > 2)
692  {
693  --y;
694  }
695  if (y <= 0)
696  {
697  --y;
698  }
699 
700  //
701  // get time of day from day fraction
702  //
703  double hr = floor(dayfrac * 24.0);
704  double mn = floor((dayfrac * 24.0 - hr) * 60.0);
705  double f = ((dayfrac * 24.0 - hr) * 60.0 - mn) * 60.0;
706  double sc = f;
707  if (f - sc > 0.5)
708  {
709  ++sc;
710  }
711 
712  if (y < 0)
713  {
714  y = -y;
715  }
716  if (year)
717  {
718  *year = (int32_t)y;
719  }
720  if (month)
721  {
722  *month = (int32_t)m;
723  }
724  if (day)
725  {
726  *day = (int32_t)d;
727  }
728  if (hour)
729  {
730  *hour = (int32_t)hr;
731  }
732  if (minute)
733  {
734  *minute = (int32_t)mn;
735  }
736  if (second)
737  {
738  *second = (int32_t)sc;
739  }
740  if (millisecond)
741  {
742  *millisecond = (int32_t)((sc - floor(sc)) * 1000.0);
743  }
744 }

806 /* satellite system+prn/slot number to satellite number ------------------------
807 * convert satellite system+prn/slot number to satellite number
808 * args : int sys I satellite system (SYS_GPS,SYS_GLO,...)
809 * int prn I satellite prn/slot number
810 * return : satellite number (0:error)
811 *-----------------------------------------------------------------------------*/
812 int satNo(int sys, int prn)
813 {
814  if (prn <= 0) return 0;
815  switch (sys) {
816  case SYS_GPS:
817  if (prn < MINPRNGPS || MAXPRNGPS < prn) return 0;
818  return prn - MINPRNGPS + 1;
819  case SYS_GLO:
820  if (prn < MINPRNGLO || MAXPRNGLO < prn) return 0;
821  return NSATGPS + prn - MINPRNGLO + 1;
822  case SYS_GAL:
823  if (prn < MINPRNGAL || MAXPRNGAL < prn) return 0;
824  return NSATGPS + NSATGLO + prn - MINPRNGAL + 1;
825  case SYS_QZS:
826  if (prn < MINPRNQZS || MAXPRNQZS < prn) return 0;
827  return NSATGPS + NSATGLO + NSATGAL + prn - MINPRNQZS + 1;
828  case SYS_CMP:
829  if (prn < MINPRNCMP || MAXPRNCMP < prn) return 0;
830  return NSATGPS + NSATGLO + NSATGAL + NSATQZS + prn - MINPRNCMP + 1;
831  case SYS_IRN:
832  if (prn < MINPRNIRN || MAXPRNIRN < prn) return 0;
833  return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + prn - MINPRNIRN + 1;
834  case SYS_LEO:
835  if (prn < MINPRNLEO || MAXPRNLEO < prn) return 0;
836  return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN +
837  prn - MINPRNLEO + 1;
838  case SYS_SBS:
839  if (prn < MINPRNSBS || MAXPRNSBS < prn) return 0;
840  return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN + NSATLEO +
841  prn - MINPRNSBS + 1;
842  }
843  return 0;
844 }
