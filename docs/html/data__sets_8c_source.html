17// Reversed bytes in a float.
18// compiler will likely inline this as it's a tiny function
19void flipFloat(uint8_t* ptr)
20{
21 uint8_t tmp1 = *ptr++;
22 uint8_t tmp2 = *ptr++;
23 uint8_t tmp3 = *ptr++;
24 uint8_t tmp4 = *ptr;
25 *ptr-- = tmp1;
26 *ptr-- = tmp2;
27 *ptr-- = tmp3;
28 *ptr = tmp4;
29}
30
31float flipFloatCopy(float val)
32{
33 float flippedFloat;
34 uint8_t* ptr = (uint8_t*)&val;
35 uint8_t* ptr2 = (uint8_t*)&flippedFloat;
36 uint8_t tmp1 = *ptr++;
37 uint8_t tmp2 = *ptr++;
38 uint8_t tmp3 = *ptr++;
39 uint8_t tmp4 = *ptr;
40 *ptr2++ = tmp4;
41 *ptr2++ = tmp3;
42 *ptr2++ = tmp2;
43 *ptr2 = tmp1;
44 return flippedFloat;
45}

47void flipDouble(void* ptr)
48{
49 const uint32_t* w = (const uint32_t*)(ptr);
50 union
51 {
52 double v;
53 uint32_t w[2];
54 } u;
55 u.w[0] = w[1];
56 u.w[1] = w[0];
57 *(double*)ptr = u.v;
58}
59
60double flipDoubleCopy(double val)
61{
62 union
63 {
64 double v;
65 uint32_t w[2];
66 } u1, u2;
67 u1.v = val;
68 u2.w[1] = SWAP32(u1.w[0]);
69 u2.w[0] = SWAP32(u1.w[1]);
70 return u2.v;
71}

73void flipEndianess32(uint8_t* data, int dataLength)
74{
75 // data must be 4 byte aligned to swap endian-ness
76 if (dataLength & 0x00000003)
77 {
78 return;
79 }
80
81 uint32_t* dataPtr = (void*)data;
82 uint32_t* dataPtrEnd = (void*)(data + dataLength);
83 uint32_t tmp;
84 while (dataPtr < dataPtrEnd)
85 {
86 tmp = *dataPtr;
87 *dataPtr++ = SWAP32(tmp);
88 }
89}

91void flipDoubles(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
92{
93 uint16_t* doubleOffsetsEnd = offsets + offsetsLength;
94 int offsetToDouble;
95 int maxDoubleOffset = dataLength - 8;
96 int isDouble;
97 while (offsets < doubleOffsetsEnd)
98 {
99 offsetToDouble = (*offsets++);
100 isDouble = ((offsetToDouble & 0x8000) == 0);
101 offsetToDouble = (offsetToDouble & 0x7FFF) - offset;
102 if (offsetToDouble >= 0 && offsetToDouble <= maxDoubleOffset)
103 {
104 if (isDouble)
105 {
106 flipDouble(data + offsetToDouble);
107 }
108 else
109 {
110 uint64_t* ptr = (void*)(data + offsetToDouble);
111 *ptr = SWAP64(*ptr);
112 }
113 }
114 }
115}

117void flipStrings(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
118{
119 uint16_t* stringOffsetsEnd = offsets + offsetsLength;
120 int offsetToString;
121 int lengthOfString;
122 int maxStringOffset;
123
124 while (offsets < stringOffsetsEnd)
125 {
126 offsetToString = (*offsets++) - offset;
127 lengthOfString = (*offsets++);
128 maxStringOffset = dataLength - lengthOfString;
129 if (offsetToString >= 0 && offsetToString <= maxStringOffset)
130 {
131 flipEndianess32(data + offsetToString, lengthOfString);
132 }
133 }
134}

585uint32_t checksum32(const void* data, int count)
586{
587 if (count < 1 || count % 4 != 0)
588 {
589 return 0;
590 }
591
592 uint32_t checksum = 0;
593 uint32_t* dataPtr = (uint32_t*)data;
594 uint32_t* dataEnd = dataPtr + (count / 4);
595
596 while (dataPtr < dataEnd)
597 {
598 checksum ^= *dataPtr++;
599 }
600
601 return checksum;
602}

604// This function skips the first 4 bytes (one 4 byte word), which are assumed to be the checksum in the serial number flash memory data structure.
605uint32_t serialNumChecksum32(const void* data, int size)
606{
607 return checksum32((const uint8_t*)data + 4, size - 4);
608}
609
610// This function skips the first 8 bytes (two 4 byte words), which are assumed to be the size and checksum in flash memory data structures.
611uint32_t flashChecksum32(const void* data, int size)
612{
613 return checksum32((const uint8_t*)data + 8, size - 8);
614}

665void julianToDate(double julian, int32_t* year, int32_t* month, int32_t* day, int32_t* hour, int32_t* minute, int32_t* second, int32_t* millisecond)
666{
667 double j1, j2, j3, j4, j5;
668 double intgr = floor(julian);
669 double frac = julian - intgr;
670 double gregjd = 2299161.0;
671 if (intgr >= gregjd)
672 {
673 //Gregorian calendar correction
674 double tmp = floor(((intgr - 1867216.0) - 0.25) / 36524.25);
675 j1 = intgr + 1.0 + tmp - floor(0.25 * tmp);
676 }
677 else
678 {
679 j1 = intgr;
680 }
681
682 //correction for half day offset
683 double dayfrac = frac + 0.5;
684 if (dayfrac >= 1.0)
685 {
686 dayfrac -= 1.0;
687 ++j1;
688 }
689
690 j2 = j1 + 1524.0;
691 j3 = floor(6680.0 + ((j2 - 2439870.0) - 122.1) / 365.25);
692 j4 = floor(j3 * 365.25);
693 j5 = floor((j2 - j4) / 30.6001);
694
695 double d = floor(j2 - j4 - floor(j5 * 30.6001));
696 double m = floor(j5 - 1);
697 if (m > 12)
698 {
699 m -= 12;
700 }
701 double y = floor(j3 - 4715.0);
702 if (m > 2)
703 {
704 --y;
705 }
706 if (y <= 0)
707 {
708 --y;
709 }
710
711 //
712 // get time of day from day fraction
713 //
714 double hr = floor(dayfrac * 24.0);
715 double mn = floor((dayfrac * 24.0 - hr) * 60.0);
716 double f = ((dayfrac * 24.0 - hr) * 60.0 - mn) * 60.0;
717 double sc = f;
718 if (f - sc > 0.5)
719 {
720 ++sc;
721 }
722
723 if (y < 0)
724 {
725 y = -y;
726 }
727 if (year)
728 {
729 *year = (int32_t)y;
730 }
731 if (month)
732 {
733 *month = (int32_t)m;
734 }
735 if (day)
736 {
737 *day = (int32_t)d;
738 }
739 if (hour)
740 {
741 *hour = (int32_t)hr;
742 }
743 if (minute)
744 {
745 *minute = (int32_t)mn;
746 }
747 if (second)
748 {
749 *second = (int32_t)sc;
750 }
751 if (millisecond)
752 {
753 *millisecond = (int32_t)((sc - floor(sc)) * 1000.0);
754 }
755}

817/* satellite system+prn/slot number to satellite number ------------------------
818* convert satellite system+prn/slot number to satellite number
819* args : int sys I satellite system (SYS_GPS,SYS_GLO,...)
820* int prn I satellite prn/slot number
821* return : satellite number (0:error)
822*-----------------------------------------------------------------------------*/
823int satNo(int sys, int prn)
824{
825 if (prn <= 0) return 0;
826 switch (sys) {
827 case SYS_GPS:
828 if (prn < MINPRNGPS || MAXPRNGPS < prn) return 0;
829 return prn - MINPRNGPS + 1;
830 case SYS_GLO:
831 if (prn < MINPRNGLO || MAXPRNGLO < prn) return 0;
832 return NSATGPS + prn - MINPRNGLO + 1;
833 case SYS_GAL:
834 if (prn < MINPRNGAL || MAXPRNGAL < prn) return 0;
835 return NSATGPS + NSATGLO + prn - MINPRNGAL + 1;
836 case SYS_QZS:
837 if (prn < MINPRNQZS || MAXPRNQZS < prn) return 0;
838 return NSATGPS + NSATGLO + NSATGAL + prn - MINPRNQZS + 1;
839 case SYS_CMP:
840 if (prn < MINPRNCMP || MAXPRNCMP < prn) return 0;
841 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + prn - MINPRNCMP + 1;
842 case SYS_IRN:
843 if (prn < MINPRNIRN || MAXPRNIRN < prn) return 0;
844 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + prn - MINPRNIRN + 1;
845 case SYS_LEO:
846 if (prn < MINPRNLEO || MAXPRNLEO < prn) return 0;
847 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN +
848 prn - MINPRNLEO + 1;
849 case SYS_SBS:
850 if (prn < MINPRNSBS || MAXPRNSBS < prn) return 0;
851 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN + NSATLEO +
852 prn - MINPRNSBS + 1;
853 }
854 return 0;
855}
