17// Reversed bytes in a float.
18// compiler will likely inline this as it's a tiny function
19void flipFloat(uint8_t* ptr)
20{
21 uint8_t tmp1 = *ptr++;
22 uint8_t tmp2 = *ptr++;
23 uint8_t tmp3 = *ptr++;
24 uint8_t tmp4 = *ptr;
25 *ptr-- = tmp1;
26 *ptr-- = tmp2;
27 *ptr-- = tmp3;
28 *ptr = tmp4;
29}
30
31float flipFloatCopy(float val)
32{
33 float flippedFloat;
34 uint8_t* ptr = (uint8_t*)&val;
35 uint8_t* ptr2 = (uint8_t*)&flippedFloat;
36 uint8_t tmp1 = *ptr++;
37 uint8_t tmp2 = *ptr++;
38 uint8_t tmp3 = *ptr++;
39 uint8_t tmp4 = *ptr;
40 *ptr2++ = tmp4;
41 *ptr2++ = tmp3;
42 *ptr2++ = tmp2;
43 *ptr2 = tmp1;
44 return flippedFloat;
45}

47void flipDouble(void* ptr)
48{
49 const uint32_t* w = (const uint32_t*)(ptr);
50 union
51 {
52 double v;
53 uint32_t w[2];
54 } u;
55 u.w[0] = w[1];
56 u.w[1] = w[0];
57 *(double*)ptr = u.v;
58}
59
60double flipDoubleCopy(double val)
61{
62 union
63 {
64 double v;
65 uint32_t w[2];
66 } u1, u2;
67 u1.v = val;
68 u2.w[1] = SWAP32(u1.w[0]);
69 u2.w[0] = SWAP32(u1.w[1]);
70 return u2.v;
71}

73void flipEndianess32(uint8_t* data, int dataLength)
74{
75 // data must be 4 byte aligned to swap endian-ness
76 if (dataLength & 0x00000003)
77 {
78 return;
79 }
80
81 uint32_t* dataPtr = (void*)data;
82 uint32_t* dataPtrEnd = (void*)(data + dataLength);
83 uint32_t tmp;
84 while (dataPtr < dataPtrEnd)
85 {
86 tmp = *dataPtr;
87 *dataPtr++ = SWAP32(tmp);
88 }
89}

91void flipDoubles(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
92{
93 uint16_t* doubleOffsetsEnd = offsets + offsetsLength;
94 int offsetToDouble;
95 int maxDoubleOffset = dataLength - 8;
96 int isDouble;
97 while (offsets < doubleOffsetsEnd)
98 {
99 offsetToDouble = (*offsets++);
100 isDouble = ((offsetToDouble & 0x8000) == 0);
101 offsetToDouble = (offsetToDouble & 0x7FFF) - offset;
102 if (offsetToDouble >= 0 && offsetToDouble <= maxDoubleOffset)
103 {
104 if (isDouble)
105 {
106 flipDouble(data + offsetToDouble);
107 }
108 else
109 {
110 uint64_t* ptr = (void*)(data + offsetToDouble);
111 *ptr = SWAP64(*ptr);
112 }
113 }
114 }
115}

117void flipStrings(uint8_t* data, int dataLength, int offset, uint16_t* offsets, uint16_t offsetsLength)
118{
119 uint16_t* stringOffsetsEnd = offsets + offsetsLength;
120 int offsetToString;
121 int lengthOfString;
122 int maxStringOffset;
123
124 while (offsets < stringOffsetsEnd)
125 {
126 offsetToString = (*offsets++) - offset;
127 lengthOfString = (*offsets++);
128 maxStringOffset = dataLength - lengthOfString;
129 if (offsetToString >= 0 && offsetToString <= maxStringOffset)
130 {
131 flipEndianess32(data + offsetToString, lengthOfString);
132 }
133 }
134}

579uint32_t checksum32(const void* data, int count)
580{
581 if (count < 1 || count % 4 != 0)
582 {
583 return 0;
584 }
585
586 uint32_t checksum = 0;
587 uint32_t* dataPtr = (uint32_t*)data;
588 uint32_t* dataEnd = dataPtr + (count / 4);
589
590 while (dataPtr < dataEnd)
591 {
592 checksum ^= *dataPtr++;
593 }
594
595 return checksum;
596}

598// This function skips the first 4 bytes (one 4 byte word), which are assumed to be the checksum in the serial number flash memory data structure.
599uint32_t serialNumChecksum32(const void* data, int size)
600{
601 return checksum32((const uint8_t*)data + 4, size - 4);
602}
603
604// This function skips the first 8 bytes (two 4 byte words), which are assumed to be the size and checksum in flash memory data structures.
605uint32_t flashChecksum32(const void* data, int size)
606{
607 return checksum32((const uint8_t*)data + 8, size - 8);
608}

656void julianToDate(double julian, int32_t* year, int32_t* month, int32_t* day, int32_t* hour, int32_t* minute, int32_t* second, int32_t* millisecond)
657{
658 double j1, j2, j3, j4, j5;
659 double intgr = floor(julian);
660 double frac = julian - intgr;
661 double gregjd = 2299161.0;
662 if (intgr >= gregjd)
663 {
664 //Gregorian calendar correction
665 double tmp = floor(((intgr - 1867216.0) - 0.25) / 36524.25);
666 j1 = intgr + 1.0 + tmp - floor(0.25 * tmp);
667 }
668 else
669 {
670 j1 = intgr;
671 }
672
673 //correction for half day offset
674 double dayfrac = frac + 0.5;
675 if (dayfrac >= 1.0)
676 {
677 dayfrac -= 1.0;
678 ++j1;
679 }
680
681 j2 = j1 + 1524.0;
682 j3 = floor(6680.0 + ((j2 - 2439870.0) - 122.1) / 365.25);
683 j4 = floor(j3 * 365.25);
684 j5 = floor((j2 - j4) / 30.6001);
685
686 double d = floor(j2 - j4 - floor(j5 * 30.6001));
687 double m = floor(j5 - 1);
688 if (m > 12)
689 {
690 m -= 12;
691 }
692 double y = floor(j3 - 4715.0);
693 if (m > 2)
694 {
695 --y;
696 }
697 if (y <= 0)
698 {
699 --y;
700 }
701
702 //
703 // get time of day from day fraction
704 //
705 double hr = floor(dayfrac * 24.0);
706 double mn = floor((dayfrac * 24.0 - hr) * 60.0);
707 double f = ((dayfrac * 24.0 - hr) * 60.0 - mn) * 60.0;
708 double sc = f;
709 if (f - sc > 0.5)
710 {
711 ++sc;
712 }
713
714 if (y < 0)
715 {
716 y = -y;
717 }
718 if (year)
719 {
720 *year = (int32_t)y;
721 }
722 if (month)
723 {
724 *month = (int32_t)m;
725 }
726 if (day)
727 {
728 *day = (int32_t)d;
729 }
730 if (hour)
731 {
732 *hour = (int32_t)hr;
733 }
734 if (minute)
735 {
736 *minute = (int32_t)mn;
737 }
738 if (second)
739 {
740 *second = (int32_t)sc;
741 }
742 if (millisecond)
743 {
744 *millisecond = (int32_t)((sc - floor(sc)) * 1000.0);
745 }
746}

808/* satellite system+prn/slot number to satellite number ------------------------
809* convert satellite system+prn/slot number to satellite number
810* args : int sys I satellite system (SYS_GPS,SYS_GLO,...)
811* int prn I satellite prn/slot number
812* return : satellite number (0:error)
813*-----------------------------------------------------------------------------*/
814int satNo(int sys, int prn)
815{
816 if (prn <= 0) return 0;
817 switch (sys) {
818 case SYS_GPS:
819 if (prn < MINPRNGPS || MAXPRNGPS < prn) return 0;
820 return prn - MINPRNGPS + 1;
821 case SYS_GLO:
822 if (prn < MINPRNGLO || MAXPRNGLO < prn) return 0;
823 return NSATGPS + prn - MINPRNGLO + 1;
824 case SYS_GAL:
825 if (prn < MINPRNGAL || MAXPRNGAL < prn) return 0;
826 return NSATGPS + NSATGLO + prn - MINPRNGAL + 1;
827 case SYS_QZS:
828 if (prn < MINPRNQZS || MAXPRNQZS < prn) return 0;
829 return NSATGPS + NSATGLO + NSATGAL + prn - MINPRNQZS + 1;
830 case SYS_CMP:
831 if (prn < MINPRNCMP || MAXPRNCMP < prn) return 0;
832 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + prn - MINPRNCMP + 1;
833 case SYS_IRN:
834 if (prn < MINPRNIRN || MAXPRNIRN < prn) return 0;
835 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + prn - MINPRNIRN + 1;
836 case SYS_LEO:
837 if (prn < MINPRNLEO || MAXPRNLEO < prn) return 0;
838 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN +
839 prn - MINPRNLEO + 1;
840 case SYS_SBS:
841 if (prn < MINPRNSBS || MAXPRNSBS < prn) return 0;
842 return NSATGPS + NSATGLO + NSATGAL + NSATQZS + NSATCMP + NSATIRN + NSATLEO +
843 prn - MINPRNSBS + 1;
844 }
845 return 0;
846}
