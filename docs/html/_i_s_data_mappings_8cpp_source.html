193 {
194  static const string timestampFields[] = { "time", "timeOfWeek", "timeOfWeekMs", "seconds" };
195  const map_name_to_info_t& offsetMap = mappings[id];
196 
197  if (offsetMap.size() != 0)
198  {
199  for (size_t i = 0; i < _ARRAY_ELEMENT_COUNT(timestampFields); i++)
200  {
201  map_name_to_info_t::const_iterator timestampField = offsetMap.find(timestampFields[i]);
202  if (timestampField != offsetMap.end())
203  {
204  timestamps[id] = (const data_info_t*)&timestampField->second;
205  return;
206  }
207  }
208  }
209 
210  timestamps[id] = NULLPTR; // ensure value is not garbage
211 }
