56bool cISLogger::LogHeaderIsCorrupt(const p_data_hdr_t* hdr)
57{
58 bool corrupt = (hdr != NULL &&
59 (
60 hdr->size == 0 ||
61 hdr->offset + hdr->size > MAX_DATASET_SIZE ||
62 hdr->id == 0 ||
63// hdr->id >= DID_COUNT ||
64 hdr->offset % 4 != 0 ||
65 hdr->size % 4 != 0
66// #if !defined(DONT_CHECK_LOG_DATA_SET_SIZE)
67// || (cISDataMappings::GetSize(hdr->id) > 0 && hdr->offset + hdr->size > cISDataMappings::GetSize(hdr->id))
68// #endif
69 ));
70 return corrupt;
71}

104void cISLogger::Update()
105{
106 // if we have a timeout param and the time has elapsed, shutdown
107 if (m_lastCommTime == 0)
108 {
109 m_lastCommTime = GetTime();
110 }
111 else if (m_timeoutFlushSeconds > 0 && GetTime() - m_lastCommTime >= m_timeoutFlushSeconds)
112 {
113 for (unsigned int i = 0; i < m_devices.size(); i++)
114 {
115 m_devices[i]->Flush();
116 }
117 }
118 ISFileManager::TouchFile(m_directory + "/stats.txt");
119}

122bool cISLogger::InitSaveCommon(eLogType logType, const string& directory, const string& subDirectory, int numDevices, float maxDiskSpacePercent, uint32_t maxFileSize, bool useSubFolderTimestamp)
123{
124 static const int minFileCount = 50;
125 static const int maxFileCount = 10000;
126
127 // Close any open files
128 CloseAllFiles();
129
130 m_logType = logType;
131 m_directory = (directory.empty() ? DEFAULT_LOGS_DIRECTORY : directory);
132 maxDiskSpacePercent = _CLAMP(maxDiskSpacePercent, 0.01f, 0.99f);
133 uint64_t availableSpace = ISFileManager::GetDirectorySpaceAvailable(m_directory);
134 m_maxDiskSpace = (maxDiskSpacePercent <= 0.0f ? availableSpace : (uint64_t)(availableSpace * maxDiskSpacePercent));
135
136 // ensure there are between min and max file count
137 if (maxFileSize > m_maxDiskSpace / minFileCount)
138 {
139 m_maxFileSize = (uint32_t)(m_maxDiskSpace / minFileCount);
140 }
141 else if (maxFileSize < m_maxDiskSpace / maxFileCount)
142 {
143 m_maxFileSize = (uint32_t)(m_maxDiskSpace / maxFileCount);
144 }
145 else
146 {
147 m_maxFileSize = maxFileSize;
148 }
149
150 m_maxFileSize = _MIN(m_maxFileSize, maxFileSize);
151
152 // create root dir
153 _MKDIR(m_directory.c_str());
154
155 if (useSubFolderTimestamp)
156 {
157 // create time stamp dir
158 m_directory = directory + "/" + m_timeStamp;
159 _MKDIR(m_directory.c_str());
160
161 if (!subDirectory.empty())
162 {
163 // create sub dir
164 m_directory += "/" + subDirectory;
165 _MKDIR(m_directory.c_str());
166 }
167 }
168
169 // create empty stats file to track timestamps
170 string str = m_directory + "/" + subDirectory + "/stats.txt";
171 cISLogFileBase* statsFile = CreateISLogFile(str, "w");
172 CloseISLogFile(statsFile);
173
174 // Initialize devices
175 return InitDevicesForWriting(numDevices);
176}

179string cISLogger::CreateCurrentTimestamp()
180{
181 char buf[80];
182
183#if PLATFORM_IS_EVB_2
184 date_time_t rtc;
185#if USE_RTC_DATE_TIME // use RTC
186 rtc_get_date(RTC, &rtc.year, &rtc.month, &rtc.day, &rtc.week);
187 rtc_get_time(RTC, &rtc.hour, &rtc.minute, &rtc.second);
188#else // use uINS GPS time
189 rtc = g_gps_date_time;
190#endif
191 SNPRINTF(buf, _ARRAY_BYTE_COUNT(buf), "%.4d%.2d%.2d_%.2d%.2d%.2d",
192 (int)rtc.year, (int)rtc.month, (int)rtc.day,
193 (int)rtc.hour, (int)rtc.minute, (int)rtc.second);
194#else
195 // Create timestamp
196 time_t rawtime = GetTime();
197 struct tm * timeinfo = localtime(&rawtime);
198 strftime(buf, 80, "%Y%m%d_%H%M%S", timeinfo);
199#endif
200
201 return string(buf);
202}

263bool cISLogger::LoadFromDirectory(const string& directory, eLogType logType, vector<string> serials)
264{
265 // Delete and clear prior devices
266 Cleanup();
267 m_logType = logType;
268 string fileExtensionRegex;
269 set<string> serialNumbers;
270
271 switch (logType)
272 {
273 default:
274 case cISLogger::LOGTYPE_DAT: fileExtensionRegex = "\\.dat$"; break;
275 case cISLogger::LOGTYPE_SDAT: fileExtensionRegex = "\\.sdat$"; break;
276 case cISLogger::LOGTYPE_CSV: fileExtensionRegex = "\\.csv$"; break;
277 case cISLogger::LOGTYPE_JSON: fileExtensionRegex = "\\.json$"; break;
278 case cISLogger::LOGTYPE_KML: return false; // fileExtensionRegex = "\\.kml$"; break; // kml read not supported
279 }
280
281 // get all files, sorted by name
282 vector<ISFileManager::file_info_t> files;
283 ISFileManager::GetDirectorySpaceUsed(directory, fileExtensionRegex, files, false, false);
284 if (files.size() == 0)
285 {
286 return false;
287 }
288
289 for (size_t i = 0; i < files.size(); i++)
290 {
291 string name = ISFileManager::GetFileName(files[i].name);
292
293 // check for log file prefix
294 size_t endOfLogPrefixIndex = name.find(IS_LOG_FILE_PREFIX);
295 if (endOfLogPrefixIndex != string::npos)
296 {
297 endOfLogPrefixIndex += IS_LOG_FILE_PREFIX_LENGTH;
298 size_t serialNumberEndIndex = name.find('_', endOfLogPrefixIndex);
299 if (serialNumberEndIndex != string::npos)
300 {
301 string serialNumber = name.substr(endOfLogPrefixIndex, serialNumberEndIndex - endOfLogPrefixIndex);
302
303 // if we don't have a timestamp yet, see if we can parse it from the filename, i.e. IS_LOG_FILE_PREFIX 30013_20170103_151023_001
304 if (m_timeStamp.length() == 0)
305 {
306 // find _ at end of timestamp
307 size_t timestampIndex = name.find_last_of('_');
308 if (timestampIndex == string::npos)
309 {
310 timestampIndex = name.find_last_of('.');
311 }
312 if (timestampIndex != string::npos && timestampIndex - ++serialNumberEndIndex == IS_LOG_TIMESTAMP_LENGTH)
313 {
314 m_timeStamp = name.substr(timestampIndex - IS_LOG_TIMESTAMP_LENGTH, IS_LOG_TIMESTAMP_LENGTH);
315 }
316 }
317
318 // check for unique serial numbers
319 if (serialNumbers.find(serialNumber) == serialNumbers.end())
320 {
321 if (serials.size() == 0 || ((find(serials.begin(), serials.end(), "SN" + serialNumber) != serials.end() ||
322 find(serials.begin(), serials.end(), "ALL") != serials.end()))) // and that it is a serial number we want to use
323 {
324 serialNumbers.insert(serialNumber);
325
326 // Add devices
327 {
328#if !PLATFORM_IS_EMBEDDED
329 const std::lock_guard<std::mutex> lock(g_devices_mutex);
330#endif
331 switch (logType)
332 {
333 default:
334 case cISLogger::LOGTYPE_DAT: m_devices.push_back(make_shared<cDeviceLogSerial>()); break;
335#if !defined(PLATFORM_IS_EVB_2) || !PLATFORM_IS_EVB_2
336 case cISLogger::LOGTYPE_SDAT: m_devices.push_back(make_shared<cDeviceLogSorted>()); break;
337 case cISLogger::LOGTYPE_CSV: m_devices.push_back(make_shared<cDeviceLogCSV>()); break;
338 case cISLogger::LOGTYPE_JSON: m_devices.push_back(make_shared<cDeviceLogJSON>()); break;
339#endif
340 }
341 }
342 m_devices.back()->SetupReadInfo(directory, serialNumber, m_timeStamp);
343
344#if LOG_DEBUG_GEN
345 printf("cISLogger::LoadFromDirectory SN%s %s (file %d of %d)\n", serialNumber.c_str(), m_timeStamp.c_str(), (int)i + 1, (int)files.size());
346#endif
347 }
348 }
349 }
350 }
351 }
352
353 for (uint32_t i = 0; i < m_devices.size(); i++)
354 {
355 m_devices[i]->InitDeviceForReading();
356 }
357 return (m_devices.size() != 0);
358}

361bool cISLogger::LogData(unsigned int device, p_data_hdr_t* dataHdr, const uint8_t* dataBuf)
362{
363 m_lastCommTime = GetTime();
364
365 if (!m_enabled)
366 {
367 m_errorFile.lprintf("Logger is not enabled\r\n");
368 return false;
369 }
370 else if (device >= m_devices.size() || dataHdr == NULL || dataBuf == NULL)
371 {
372 m_errorFile.lprintf("Invalid device handle or NULL data\r\n");
373 return false;
374 }
375 else if (LogHeaderIsCorrupt(dataHdr))
376 {
377 m_errorFile.lprintf("Corrupt log header, id: %lu, offset: %lu, size: %lu\r\n", (unsigned long)dataHdr->id, (unsigned long)dataHdr->offset, (unsigned long)dataHdr->size);
378 m_logStats.LogError(dataHdr);
379 }
380 else if (!m_devices[device]->SaveData(dataHdr, dataBuf))
381 {
382 m_errorFile.lprintf("Underlying log implementation failed to save\r\n");
383 m_logStats.LogError(dataHdr);
384 }
385#if 1
386 else
387 {
388 double timestamp = cISDataMappings::GetTimestamp(dataHdr, dataBuf);
389 m_logStats.LogDataAndTimestamp(dataHdr->id, timestamp);
390
391 if (dataHdr->id == DID_DIAGNOSTIC_MESSAGE)
392 {
393 // write to diagnostic text file
394#if 0
395 std::ostringstream outFilePath;
396 outFilePath << m_directory << "/diagnostic_" << m_devices[device]->GetDeviceInfo()->serialNumber << ".txt";
397 cISLogFileBase *outfile = CreateISLogFile(outFilePath.str(), "a");
398#else
399 cISLogFileBase *outfile = CreateISLogFile(m_directory + "/diagnostic_" + std::to_string(m_devices[device]->GetDeviceInfo()->serialNumber) + ".txt", "a");
400#endif
401 std::string msg = (((diag_msg_t*)dataBuf)->message);
402 outfile->write(msg.c_str(), msg.length());
403 if (msg.length() > 0 && *msg.end() != '\n')
404 {
405 outfile->putch('\n');
406 }
407 CloseISLogFile(outfile);
408 }
409 }
410#endif
411 return true;
412}

415p_data_t* cISLogger::ReadData(unsigned int device)
416{
417 if (device >= m_devices.size())
418 {
419 return NULL;
420 }
421
422 p_data_t* data = NULL;
423 while (LogDataIsCorrupt(data = m_devices[device]->ReadData()))
424 {
425 m_errorFile.lprintf("Corrupt log header, id: %lu, offset: %lu, size: %lu\r\n", (unsigned long)data->hdr.id, (unsigned long)data->hdr.offset, (unsigned long)data->hdr.size);
426 m_logStats.LogError(&data->hdr);
427 data = NULL;
428 }
429 if (data != NULL)
430 {
431 double timestamp = cISDataMappings::GetTimestamp(&data->hdr, data->buf);
432 m_logStats.LogDataAndTimestamp(data->hdr.id, timestamp);
433 }
434 return data;

570bool cISLogger::CopyLog(cISLogger& log, const string& timestamp, const string &outputDir, eLogType logType, float maxLogSpacePercent, uint32_t maxFileSize, bool useSubFolderTimestamp, bool enableCsvIns2ToIns1Conversion)
571{
572 m_logStats.Clear();
573 if (!InitSaveTimestamp(timestamp, outputDir, g_emptyString, log.GetDeviceCount(), logType, maxLogSpacePercent, maxFileSize, useSubFolderTimestamp))
574 {
575 return false;
576 }
577 EnableLogging(true);
578 p_data_t* data = NULL;
579 for (unsigned int dev = 0; dev < log.GetDeviceCount(); dev++)
580 {
581 // Copy device info
582 const dev_info_t* devInfo = log.GetDeviceInfo(dev);
583 SetDeviceInfo(devInfo, dev);
584
585#if LOG_DEBUG_GEN || DEBUG_PRINT
586 printf("cISLogger::CopyLog SN%d type %d, (%d of %d)\n", devInfo->serialNumber, logType, dev+1, log.GetDeviceCount());
587#endif
588
589 // Set KML configuration
590 m_devices[dev]->SetKmlConfig(m_showPath, m_showSample, m_showTimeStamp, m_iconUpdatePeriodSec, m_altClampToGround);
591
592 // Copy data
593 for (g_copyReadCount = 0; (data = log.ReadData(dev)); g_copyReadCount++)
594 {
595
596#if LOG_DEBUG_PRINT_READ
597 double timestamp = cISDataMappings::GetTimestamp(&(data->hdr), data->buf);
598 printf("read: %d DID: %3d time: %.4lf\n", g_copyReadCount, data->hdr.id, timestamp);
599 g_copyReadDid = data->hdr.id;
600#endif
601
602 // CSV special cases
603 if (logType == eLogType::LOGTYPE_CSV && enableCsvIns2ToIns1Conversion)
604 {
605 if (data->hdr.id == DID_INS_2)
606 { // Convert INS2 to INS1 when creating .csv logs
607 ins_1_t ins1;
608 ins_2_t ins2;
609
610 copyDataPToStructP(&ins2, data, sizeof(ins_2_t));
611 convertIns2ToIns1(&ins2, &ins1);
612
613 p_data_hdr_t hdr;
614 hdr.id = DID_INS_1;
615 hdr.size = sizeof(ins_1_t);
616 hdr.offset = 0;
617 LogData(dev, &hdr, (uint8_t*)&ins1);
618 }
619 }
620
621 // Save data
622 LogData(dev, &data->hdr, data->buf);
623 }
624 }
625 CloseAllFiles();
626 return true;
627}

629bool cISLogger::ReadAllLogDataIntoMemory(const string& directory, map<uint32_t, vector<vector<uint8_t>>>& data)
630{
631 cISLogger logger;
632 if (!logger.LoadFromDirectory(directory))
633 {
634 return false;
635 }
636 unsigned int deviceId = 0;
637 unsigned int lastDeviceId = 0xFFFFFEFE;
638 p_data_t* p;
639 vector<vector<uint8_t>>* currentDeviceData = NULL;
640 const dev_info_t* info;
641 uint8_t* ptr, *ptrEnd;
642 data.clear();
643
644 // read every piece of data out of every device log
645 while ((p = logger.ReadNextData(deviceId)) != NULL)
646 {
647 // if this is a new device, we need to add the device to the map using the serial number
648 if (deviceId != lastDeviceId)
649 {
650 lastDeviceId = deviceId;
651 info = logger.GetDeviceInfo(deviceId);
652 data[info->serialNumber] = vector<vector<uint8_t>>();
653 currentDeviceData = &data[info->serialNumber];
654 }
655
656 assert(currentDeviceData != NULL);
657
658 // add slots until we have slots at least equal to the data id
659 while (currentDeviceData->size() < p->hdr.id)
660 {
661 currentDeviceData->push_back(vector<uint8_t>());
662 }
663
664 // append each byte of the packet to the slot
665 ptrEnd = p->buf + p->hdr.size;
666 vector<uint8_t>& stream = (*currentDeviceData)[p->hdr.id];
667 for (ptr = p->buf; ptr != ptrEnd; ptr++)
668 {
669 stream.push_back(*ptr);
670 }
671 }
672 return true;
673}
