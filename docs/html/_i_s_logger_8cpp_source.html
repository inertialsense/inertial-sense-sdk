123 bool cISLogger::InitSaveCommon(eLogType logType, const string& directory, const string& subDirectory, int numDevices, float maxDiskSpacePercent, uint32_t maxFileSize, bool useSubFolderTimestamp)
124 {
125  static const int minFileCount = 50;
126  static const int maxFileCount = 10000;
127 
128  // Close any open files
129  CloseAllFiles();
130 
131  m_logType = logType;
132  m_directory = (directory.empty() ? DEFAULT_LOGS_DIRECTORY : directory);
133  maxDiskSpacePercent = _CLAMP(maxDiskSpacePercent, 0.01f, 0.99f);
134  uint64_t availableSpace = ISFileManager::GetDirectorySpaceAvailable(m_directory);
135  m_maxDiskSpace = (maxDiskSpacePercent <= 0.0f ? availableSpace : (uint64_t)(availableSpace * maxDiskSpacePercent));
136 
137  // ensure there are between min and max file count
138  if (maxFileSize > m_maxDiskSpace / minFileCount)
139  {
140  m_maxFileSize = (uint32_t)(m_maxDiskSpace / minFileCount);
141  }
142  else if (maxFileSize < m_maxDiskSpace / maxFileCount)
143  {
144  m_maxFileSize = (uint32_t)(m_maxDiskSpace / maxFileCount);
145  }
146  else
147  {
148  m_maxFileSize = maxFileSize;
149  }
150 
151  m_maxFileSize = _MIN(m_maxFileSize, maxFileSize);
152 
153  // create root dir
154  _MKDIR(m_directory.c_str());
155 
156  if (useSubFolderTimestamp)
157  {
158  // create time stamp dir
159  m_directory = directory + "/" + m_timeStamp;
160  _MKDIR(m_directory.c_str());
161 
162  if (!subDirectory.empty())
163  {
164  // create sub dir
165  m_directory += "/" + subDirectory;
166  _MKDIR(m_directory.c_str());
167  }
168  }
169 
170  // create empty stats file to track timestamps
171  string str = m_directory + "/" + subDirectory + "/stats.txt";
172  cISLogFileBase* statsFile = CreateISLogFile(str, "w");
173  CloseISLogFile(statsFile);
174 
175  // Initialize devices
176  return InitDevicesForWriting(numDevices);
177 }

263 bool cISLogger::LoadFromDirectory(const string& directory, eLogType logType, vector<string> serials)
264 {
265  // Delete and clear prior devices
266  Cleanup();
267  m_logType = logType;
268  string fileExtensionRegex;
269  set<string> serialNumbers;
270 
271  switch (logType)
272  {
273  default:
274  case cISLogger::LOGTYPE_DAT: fileExtensionRegex = "\\.dat$"; break;
275  case cISLogger::LOGTYPE_SDAT: fileExtensionRegex = "\\.sdat$"; break;
276  case cISLogger::LOGTYPE_CSV: fileExtensionRegex = "\\.csv$"; break;
277  case cISLogger::LOGTYPE_JSON: fileExtensionRegex = "\\.json$"; break;
278  case cISLogger::LOGTYPE_KML: return false; // fileExtensionRegex = "\\.kml$"; break; // kml read not supported
279  }
280 
281  // get all files, sorted by name
282  vector<ISFileManager::file_info_t> files;
283  ISFileManager::GetDirectorySpaceUsed(directory, fileExtensionRegex, files, false, false);
284  if (files.size() == 0)
285  {
286  return false;
287  }
288 
289  for (size_t i = 0; i < files.size(); i++)
290  {
291  string name = ISFileManager::GetFileName(files[i].name);
292 
293  // check for log file prefix
294  size_t endOfLogPrefixIndex = name.find(IS_LOG_FILE_PREFIX);
295  if (endOfLogPrefixIndex != string::npos)
296  {
297  endOfLogPrefixIndex += IS_LOG_FILE_PREFIX_LENGTH;
298  size_t serialNumberEndIndex = name.find('_', endOfLogPrefixIndex);
299  if (serialNumberEndIndex != string::npos)
300  {
301  string serialNumber = name.substr(endOfLogPrefixIndex, serialNumberEndIndex - endOfLogPrefixIndex);
302 
303  // if we don't have a timestamp yet, see if we can parse it from the filename, i.e. IS_LOG_FILE_PREFIX 30013_20170103_151023_001
304  if (m_timeStamp.length() == 0)
305  {
306  // find _ at end of timestamp
307  size_t timestampIndex = name.find_last_of('_');
308  if (timestampIndex == string::npos)
309  {
310  timestampIndex = name.find_last_of('.');
311  }
312  if (timestampIndex != string::npos && timestampIndex - ++serialNumberEndIndex == IS_LOG_TIMESTAMP_LENGTH)
313  {
314  m_timeStamp = name.substr(timestampIndex - IS_LOG_TIMESTAMP_LENGTH, IS_LOG_TIMESTAMP_LENGTH);
315  }
316  }
317 
318  // check for unique serial numbers
319  if (serialNumbers.find(serialNumber) == serialNumbers.end())
320  {
321  if (serials.size() == 0 || ((find(serials.begin(), serials.end(), "SN" + serialNumber) != serials.end() ||
322  find(serials.begin(), serials.end(), "ALL") != serials.end()))) // and that it is a serial number we want to use
323  {
324  serialNumbers.insert(serialNumber);
325 
326  // Add devices
327  {
328 #if !PLATFORM_IS_EMBEDDED
329  const std::lock_guard<std::mutex> lock(g_devices_mutex);
330 #endif
331  switch (logType)
332  {
333  default:
334  case cISLogger::LOGTYPE_DAT: m_devices.push_back(make_shared<cDeviceLogSerial>()); break;
335 #if !defined(PLATFORM_IS_EVB_2) || !PLATFORM_IS_EVB_2
336  case cISLogger::LOGTYPE_SDAT: m_devices.push_back(make_shared<cDeviceLogSorted>()); break;
337  case cISLogger::LOGTYPE_CSV: m_devices.push_back(make_shared<cDeviceLogCSV>()); break;
338  case cISLogger::LOGTYPE_JSON: m_devices.push_back(make_shared<cDeviceLogJSON>()); break;
339 #endif
340  }
341  }
342  m_devices.back()->SetupReadInfo(directory, serialNumber, m_timeStamp);
343 
344 #if LOG_DEBUG_GEN
345  printf("cISLogger::LoadFromDirectory SN%s %s (file %d of %d)\n", serialNumber.c_str(), m_timeStamp.c_str(), (int)i + 1, (int)files.size());
346 #endif
347  }
348  }
349  }
350  }
351  }
352 
353  for (uint32_t i = 0; i < m_devices.size(); i++)
354  {
355  m_devices[i]->InitDeviceForReading();
356  }
357  return (m_devices.size() != 0);
358 }

570 bool cISLogger::CopyLog(cISLogger& log, const string& timestamp, const string &outputDir, eLogType logType, float maxLogSpacePercent, uint32_t maxFileSize, bool useSubFolderTimestamp, bool enableCsvIns2ToIns1Conversion)

577  EnableLogging(true);
578  p_data_t* data = NULL;
579  for (unsigned int dev = 0; dev < log.GetDeviceCount(); dev++)
580  {
581  // Copy device info
582  const dev_info_t* devInfo = log.GetDeviceInfo(dev);
583  SetDeviceInfo(devInfo, dev);
584 
585 #if LOG_DEBUG_GEN || DEBUG_PRINT
586  printf("cISLogger::CopyLog SN%d type %d, (%d of %d)\n", devInfo->serialNumber, logType, dev+1, log.GetDeviceCount());
587 #endif
588 
589  // Set KML configuration
590  m_devices[dev]->SetKmlConfig(m_showPath, m_showSample, m_showTimeStamp, m_iconUpdatePeriodSec, m_altClampToGround);
591 
592  // Copy data
593  for (g_copyReadCount = 0; (data = log.ReadData(dev)); g_copyReadCount++)
594  {
595 
596 #if LOG_DEBUG_PRINT_READ
597  double timestamp = cISDataMappings::GetTimestamp(&(data->hdr), data->buf);
598  printf("read: %d DID: %3d time: %.4lf\n", g_copyReadCount, data->hdr.id, timestamp);
599  g_copyReadDid = data->hdr.id;
600 #endif
601 
602  // CSV special cases
603  if (logType == eLogType::LOGTYPE_CSV && enableCsvIns2ToIns1Conversion)

612 
613  p_data_hdr_t hdr;
614  hdr.id = DID_INS_1;
615  hdr.size = sizeof(ins_1_t);
616  hdr.offset = 0;
617  LogData(dev, &hdr, (uint8_t*)&ins1);
618  }
619  }

621  // Save data
622  LogData(dev, &data->hdr, data->buf);
623  }
624  }

629 bool cISLogger::ReadAllLogDataIntoMemory(const string& directory, map<uint32_t, vector<vector<uint8_t>>>& data)
630 {
631  cISLogger logger;
632  if (!logger.LoadFromDirectory(directory))
633  {
634  return false;
635  }
636  unsigned int deviceId = 0;
637  unsigned int lastDeviceId = 0xFFFFFEFE;
638  p_data_t* p;
639  vector<vector<uint8_t>>* currentDeviceData = NULL;
640  const dev_info_t* info;
641  uint8_t* ptr, *ptrEnd;
642  data.clear();
643 
644  // read every piece of data out of every device log
645  while ((p = logger.ReadNextData(deviceId)) != NULL)
646  {
647  // if this is a new device, we need to add the device to the map using the serial number
648  if (deviceId != lastDeviceId)
649  {
650  lastDeviceId = deviceId;
651  info = logger.GetDeviceInfo(deviceId);
652  data[info->serialNumber] = vector<vector<uint8_t>>();
653  currentDeviceData = &data[info->serialNumber];
654  }
655 
656  assert(currentDeviceData != NULL);
657 
658  // add slots until we have slots at least equal to the data id
659  while (currentDeviceData->size() < p->hdr.id)
660  {
661  currentDeviceData->push_back(vector<uint8_t>());
662  }
663 
664  // append each byte of the packet to the slot
665  ptrEnd = p->buf + p->hdr.size;
666  vector<uint8_t>& stream = (*currentDeviceData)[p->hdr.id];
667  for (ptr = p->buf; ptr != ptrEnd; ptr++)
668  {
669  stream.push_back(*ptr);
670  }
671  }
672  return true;
673 }
