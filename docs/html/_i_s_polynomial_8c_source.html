19char ixPolyFit(const int n, const float x[], const float y[], float coef[], const int num_coefs)
20{
21#define MAX_NUM_COEF 3
22#define MAX_NUMBER_SAMPLES 10
23
24 if (num_coefs < 2 || num_coefs > MAX_NUM_COEF ||
25 n < num_coefs || n > MAX_NUMBER_SAMPLES)
26 return -1;
27
28 int p = num_coefs;
29
30#define P_MAX (MAX_NUM_COEF)
31
32 float A[P_MAX * MAX_NUMBER_SAMPLES];
33 float AtA[P_MAX * P_MAX]; // At A
34 float iAtA[P_MAX * P_MAX]; // inv(A)
35 float Aty[P_MAX];
36
37 // Populate A matrix
38 switch (num_coefs)
39 {
40 case 2: // 1st order
41 for (int i = 0; i < n; i++)
42 {
43 A[i*p] = x[i];
44 A[i*p + 1] = 1.0f;
45 }
46 break;
47 case 3: // 2nd order
48 for (int i = 0; i < n; i++)
49 {
50 A[i*p] = x[i] * x[i];
51 A[i*p + 1] = x[i];
52 A[i*p + 2] = 1.0f;
53 }
54 break;
55 }
56
57 // AtA[pxp] = At[pxn] * A[nxp]
58 mul_MatMxN(AtA, A, A, p, n, p, 1, 0, 0);
59
60 // iAtA[pxp] = (AtA[pxp])^-1
61 switch (num_coefs)
62 {
63 case 2: inv_Mat2(iAtA, AtA); break; // 1st order
64 case 3: inv_Mat3(iAtA, AtA); break; // 2nd order
65 }
66
67 // Aty[px1] = At[pxn] * y[nx1]
68 mul_MatMxN(Aty, A, y, p, n, 1, 1, 0, 0);
69
70 // coef[px1] = (At A)^-1 At y = iAtA[pxp] Aty[px1]
71 mul_MatMxN(coef, iAtA, Aty, p, p, 1, 0, 0, 0);
72
73 return 0;
74}

77float ixPolyHorner(const int coef_size, const float coef[], const float x)
78{
79 float y;
80
81 y = coef[0];
82 for (int i = 1; i < coef_size; i++)
