75 dataBuffer.push_back(0);

125int cComDataBuffer::ReadData(int pHandle, uint32_t dataId, vector<uint8_t>& data)
126{
127 data.clear();
128
129 if (dataId == 0 || dataId >= DID_COUNT || dataId >= m_buffers[pHandle].size())
130 {
131 return -1;
132 }
133
134 cMutexLocker lock(&m_mutex);
135
136 struct stat buf;
137 FILE* file = m_buffers[pHandle][dataId].file;
138 fflush(file);
139#ifdef __linux__
140 fstat(fileno(file), &buf);
141#else
142 fstat(_fileno(file), &buf);
143#endif
144 long size = buf.st_size;
145 long pos = ftell(file);
146 long afterWriteOffset = size - pos;
147 size_t read = 0;
148
149 // read everything after write offset
150 data.resize(size);
151 if (pos < size)
152 {
153 read += fread(&data[0], 1, afterWriteOffset, file);
154 }
155
156 // read in anything before write offset
157 if (pos != 0)
158 {
159 fseek(file, 0, SEEK_SET);
160 read += fread(&data[afterWriteOffset], 1, pos, file);
161 }
162
163 if (read != size)
164 {
165 return ferror(file);
166 }
167
168 // reset file pointer to correct location for next read or write
169 fseek(file, (long)m_buffers[pHandle][dataId].writeOffset, SEEK_SET);
170
171 return 0;
172}
