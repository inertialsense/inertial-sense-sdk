110 void messageStatsAppend(string message, mul_msg_stats_t &msgStats, unsigned int ptype, int id, int timeMs)
111 {
112  switch (ptype)
113  {
114  case _PTYPE_INERTIAL_SENSE_CMD:
115  case _PTYPE_INERTIAL_SENSE_DATA:
116  if (msgStats.isb.find(id) == msgStats.isb.end())
117  { // Create new
118  msgStats.isb[id] = createNewMsgStats(timeMs, cISDataMappings::GetDataSetName(id));
119  }
120 
121  { // Update count and timestamps
122  updateTimeMs(msgStats.isb[id], timeMs);
123  }
124  break;
125 
126  case _PTYPE_ASCII_NMEA:
127  if (msgStats.ascii.find(id) == msgStats.ascii.end())
128  { // Create new
129  msgStats.ascii[id] = createNewMsgStats(timeMs);
130  }
131 
132  { // Update count and timestamps
133  updateTimeMs(msgStats.ascii[id], timeMs);
134  }
135  break;
136 
137  case _PTYPE_UBLOX:
138  if (msgStats.ublox.find(id) == msgStats.ublox.end())
139  { // Create new
140  uint8_t msgClass = (uint8_t)id;
141  uint8_t msgID = (uint8_t)(id >> 8);
142  msgStats.ublox[id] = createNewMsgStats(timeMs, messageDescriptionUblox(msgClass, msgID));
143  }
144 
145  { // Update count and timestamps
146  updateTimeMs(msgStats.ublox[id], timeMs);
147  }
148  break;
149 
150  case _PTYPE_RTCM3:
151  if (msgStats.rtcm3.find(id) == msgStats.rtcm3.end())
152  { // Create new
153  msgStats.rtcm3[id] = createNewMsgStats(timeMs, messageDescriptionRtcm3(id));
154  }
155 
156  { // Update count and timestamps
157  msg_stats_t &s = msgStats.rtcm3[id];
158  updateTimeMs(s, timeMs);
159 
160  if (id == 1029)
161  {
162  s.description = string("Text String: ") + message;
163  }
164  }
165  break;
166 
167  case _PTYPE_INERTIAL_SENSE_ACK:
168  {
169  updateTimeMs(msgStats.ack, timeMs);
170  }
171  break;
172 
173  default:
174  case _PTYPE_PARSE_ERROR:
175  {
176  updateTimeMs(msgStats.parseError, timeMs);
177  }
178  break;
179  }
180 }

182 string messageStatsSummary(mul_msg_stats_t &msgStats)
183 {
184  string str;
185 #define BUF_SIZE 512
186  char buf[BUF_SIZE];
187 
188  if (!msgStats.isb.empty())
189  {
190  str.append("Inertial Sense Binary: __________________\n");
191  str.append(" DID Count dtMs Description\n");
192  std::map<int, msg_stats_t>::iterator it;
193  for (it = msgStats.isb.begin(); it != msgStats.isb.end(); it++)
194  {
195  int did = it->first;
196  msg_stats_t &s = it->second;
197  int dtMs = (s.prevTimeMs ? (s.timeMs - s.prevTimeMs) : 0);
198 
199  SNPRINTF(buf, BUF_SIZE, "%4d %7d %5d %s\n", did, s.count, dtMs, s.description.c_str());
200  str.append(string(buf));
201  }
202  }
203 
204  if (!msgStats.ascii.empty())
205  {
206  str.append("ASCII: __________________________________\n");
207  str.append(" ID Count dtMs Description\n");
208  std::map<int, msg_stats_t>::iterator it;
209  for (it = msgStats.ascii.begin(); it != msgStats.ascii.end(); it++)
210  {
211  msg_stats_t &s = it->second;
212  int dtMs = (s.prevTimeMs ? (s.timeMs - s.prevTimeMs) : 0);
213  union
214  {
215  int id;
216  char str[8];
217  } val = {};
218  val.id = it->first;
219  SNPRINTF(buf, BUF_SIZE, "%s %7d %5d %s\n", val.str, s.count, dtMs, s.description.c_str());
220  str.append(string(buf));
221  }
222  }
223 
224  if (!msgStats.ublox.empty())
225  {
226  str.append("Ublox: __________________________________\n");
227  str.append("(Class ID) Count dtMs Description\n");
228  std::map<int, msg_stats_t>::iterator it;
229  for (it = msgStats.ublox.begin(); it != msgStats.ublox.end(); it++)
230  {
231  int id = it->first;
232  msg_stats_t &s = it->second;
233  uint8_t msgClass = (uint8_t)id;
234  uint8_t msgID = (uint8_t)(id >> 8);
235  int dtMs = (s.prevTimeMs ? (s.timeMs - s.prevTimeMs) : 0);
236  SNPRINTF(buf, BUF_SIZE, "(0x%02x 0x%02x) %7d %5d %s\n", msgClass, msgID, s.count, dtMs, s.description.c_str());
237  str.append(string(buf));
238  }
239  }
240 
241  if (!msgStats.rtcm3.empty())
242  {
243  str.append("RTCM3: __________________________________\n");
244  str.append(" ID Count dtMs Description\n");
245  std::map<int, msg_stats_t>::iterator it;

247  {
248  int id = it->first;

250  int dtMs = (s.prevTimeMs ? (s.timeMs - s.prevTimeMs) : 0);
251  SNPRINTF(buf, BUF_SIZE, "%3d %7d %5d %s\n", id, s.count, dtMs, s.description.c_str());
252  str.append(string(buf));
253  }
254  }
255 
256  if (msgStats.ack.count>5)
257  {
258  str.append("Acknowledge: ____________________________\n");
259  str.append(" Count dtMs\n");
260  msg_stats_t &s = msgStats.ack;
261  int dtMs = (s.prevTimeMs ? (s.timeMs - s.prevTimeMs) : 0);
262  SNPRINTF(buf, BUF_SIZE, "%8d %5d\n", s.count, dtMs);
263  str.append(string(buf));
264  }
265
