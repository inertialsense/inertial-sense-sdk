59 int32_t cDataChunk::PushBack(uint8_t* d1, int32_t d1Size, uint8_t* d2, int32_t d2Size)
60 {
61  // Ensure data will fit
62  int32_t nBytes = d1Size + d2Size;
63  if (m_dataHead == NULLPTR ||
64  nBytes > GetBuffFree())
65  {
66  return 0;
67  }
68  if (d1Size > 0)
69  {
70  memcpy(m_dataTail, d1, d1Size);
71  m_dataTail += d1Size;
72  m_hdr.dataSize += d1Size;
73  }
74  if (d2Size > 0)
75  {
76  memcpy(m_dataTail, d2, d2Size);
77  m_dataTail += d2Size;
78  m_hdr.dataSize += d2Size;
79  }
80  m_hdr.invDataSize = ~m_hdr.dataSize;
81  return nBytes;
82 }
