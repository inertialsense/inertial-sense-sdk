335 // Return true on refresh
336 void cInertialSenseDisplay::ProcessData(p_data_t* data, bool enableReplay, double replaySpeedX)
337 {
338  if (m_displayMode == DMODE_QUIET)
339  {
340  return;
341  }
342 
343  unsigned int curTimeMs = current_timeMs();
344  m_rxCount++;
345 
346  m_enableReplay = enableReplay;
347  m_replaySpeedX = replaySpeedX;
348 
349  if (enableReplay)
350  {
351  static bool isTowMode = false;
352  static unsigned int gpsTowMsOffset = 0;
353  static unsigned int msgTimeMsOffset = 0;
354  unsigned int msgTimeMs = 0;
355 
356  // Copy only new data
357  uDatasets d = {};
358  copyDataPToStructP(&d, data, sizeof(uDatasets));
359 
360  // Record message time. In either ToW or time since boot.
361  switch (data->hdr.id)
362  {
363  // Time of week - double
364  case DID_INS_1:
365  case DID_INS_2:
366  msgTimeMs = (unsigned int)(1000.0 * d.ins1.timeOfWeek);
367  isTowMode = true;
368  break;
369 
370  // Time of week - uint32 ms
371  case DID_SYS_PARAMS:
372  msgTimeMs = d.gpsPos.timeOfWeekMs;
373  isTowMode = true;
374  break;
375 
376  case DID_GPS1_POS:
377  case DID_GPS1_RTK_POS:
378  msgTimeMs = d.gpsPos.timeOfWeekMs;
379  gpsTowMsOffset = (unsigned int)(1000.0 * d.gpsPos.towOffset);
380  isTowMode = true;
381  break;
382 
383  case DID_GPS1_RTK_POS_REL:
384  msgTimeMs = d.gpsRtkRel.timeOfWeekMs;
385  isTowMode = true;
386  break;
387 
388  case DID_GPS1_RTK_POS_MISC:
389  msgTimeMs = d.gpsPos.timeOfWeekMs;
390  gpsTowMsOffset = (unsigned int)(1000.0 * d.gpsPos.towOffset);
391  isTowMode = false;
392  break;
393 
394  // Time since boot - double
395  case DID_MAGNETOMETER:
396  case DID_BAROMETER:
397  case DID_SYS_SENSORS:
398  case DID_PREINTEGRATED_IMU:
399  case DID_DUAL_IMU:
400  case DID_INL2_STATES:
401  case DID_GPS_BASE_RAW:
402  if (isTowMode)
403  msgTimeMs = (unsigned int)(1000.0 * d.imu.time) + gpsTowMsOffset;
404  else
405  msgTimeMs = (unsigned int)(1000.0 * d.imu.time);
406  break;
407 
408  // Unidentified data type
409 // default: printf("Unknown DID %d\t", data->hdr.id); return;
410  }
411 
412 
413  // Control replay rate
414  if (msgTimeMs != 0 && replaySpeedX > 0.0)
415  {
416  for (;;)
417  {
418  curTimeMs = current_timeMs();
419 
420  // Replay speed
421  unsigned int replayTimeMs = (unsigned int)(long)(((double)curTimeMs) * replaySpeedX);
422 
423  // Reinitialize message offset
424  if ((msgTimeMs + msgTimeMsOffset - replayTimeMs) > 1500)
425  msgTimeMsOffset = replayTimeMs - msgTimeMs;
426 
427  // Proceed if we're caught up
428  if (replayTimeMs >= msgTimeMs + msgTimeMsOffset)
429  break;
430 
431  // Add delay
432 // SLEEP_US(1000);
433  SLEEP_MS(10);
434  }
435  }
436  }
437 
438 
439  static unsigned int timeSinceClearMs = 0;
440  static char idHist[DID_COUNT] = { 0 };
441  if (m_displayMode != DMODE_SCROLL)
442  {
443  // Clear display every 2 seconds or if we start seeing new messages.
444  if (curTimeMs - timeSinceClearMs > 2000 || curTimeMs < timeSinceClearMs || idHist[data->hdr.id] == 0)
445  {
446  Clear();
447  idHist[data->hdr.id] = 1;
448  timeSinceClearMs = curTimeMs;
449  }
450  }
451 
452  // Save data to be displayed from PrintData()
453  switch (m_displayMode)
454  {
455  default:
456  m_displayMode = DMODE_PRETTY;
457  // fall through
458  case DMODE_PRETTY:
459  // Data stays at fixed location (no scroll history)
460  DataToVector(data);
461  if (m_outputOnceDid == data->hdr.id)
462  { // Exit as soon as we display DID
463  Home();
464  cout << VectortoString();
465  exit(1);
466  }
467  break;
468 
469  case DMODE_EDIT:
470  if (m_editData.did == data->hdr.id)
471  {
472  m_editData.pData = *data;
473  }
474  break;
475 
476  case DMODE_STATS:
477  DataToStats(data);
478  break;
479 
480  case DMODE_SCROLL: // Scroll display
481  cout << DataToString(data) << endl;
482  break;
483  }
484 }

527  Home();
528  cout << Connected() << endl;
