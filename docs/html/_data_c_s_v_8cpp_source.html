67int cDataCSV::ReadHeaderFromFile(FILE* pFile, uint32_t id, vector<data_info_t>& columnHeaders)
68{
69#if PLATFORM_IS_EMBEDDED
70
71 return 0;
72
73#else
74
75 (void)id;
76
77 char line[8192];
78 line[0] = '\0';
79 if (fgets(line, _ARRAY_BYTE_COUNT(line), pFile) == NULL || ferror(pFile) != 0)
80 {
81 return 0;
82 }
83 const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(id);
84 stringstream stream(line);
85 string columnHeader;
86 columnHeaders.clear();
87 while (stream.good())
88 {
89 getline(stream, columnHeader, ',');
90 while (columnHeader.length() > 0 && (columnHeader[columnHeader.size() - 1] == '\r' || columnHeader[columnHeader.size() - 1] == '\n'))
91 {
92 columnHeader.resize(columnHeader.size() - 1);
93 }
94 if (columnHeader.length() == 0)
95 {
96 columnHeader = "UNKNOWN";
97 }
98 if (columnHeader == "_ID_")
99 {
100 columnHeaders.push_back({ 0xFFFFFFFF, 0xFFFFFFFF, DataTypeInt64, (eDataFlags)0, "_ID_" });
101 }
102 else
103 {
104 map_name_to_info_t::const_iterator offset = offsetMap->find(columnHeader);
105 if (offset == offsetMap->end())
106 {
107 columnHeaders.push_back({ 0xFFFFFFFF, 0xFFFFFFFF, DataTypeBinary, (eDataFlags)0, columnHeader });
108 }
109 else
110 {
111 columnHeaders.push_back(offset->second);
112 }
113 }
114 }
115 return (int)strnlen(line, _ARRAY_BYTE_COUNT(line));
116#endif
117}

119int cDataCSV::WriteDataToFile(uint64_t orderId, FILE* pFile, const p_data_hdr_t& dataHdr, const uint8_t* dataBuf)
120{
121 // Verify file pointer
122 if (pFile == NULL)
123 {
124 return 0;
125 }
126 if (cISDataMappings::GetSize(dataHdr.id) == 0)
127 {
128 return 0;
129 }
130 string s;
131 if (!DataToStringCSV(dataHdr, dataBuf, s))
132 {
133 return 0;
134 }
135 char tmp[64];
136 SNPRINTF(tmp, 64, "%llu", (long long unsigned int)orderId);
137 fputs(tmp, pFile);
138 fputc(',', pFile);
139 fputs(s.c_str(), pFile);
140
141 // return the data string length plus comma plus order id string
142 return (int)s.length() + 1 + (int)strnlen(tmp, sizeof(tmp));
143}

145bool cDataCSV::StringCSVToData(string& s, p_data_hdr_t& hdr, uint8_t* buf, uint32_t bufSize, const vector<data_info_t>& columnHeaders)
146{
147 const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(hdr.id);
148 if (offsetMap == NULLPTR || hdr.offset != 0 || hdr.size == 0 || bufSize < hdr.size)
149 {
150 return false;
151 }
152 while (s.length() > 0 && (s[s.size() - 1] == '\r' || s[s.size() - 1] == '\n'))
153 {
154 s.resize(s.size() - 1);
155 }
156 // for parsing logic, last comma will allow last field to parse properly
157 s += ",";
158 string columnData;
159 string::const_iterator start = s.begin();
160 uint32_t index = 0;
161 hdr.offset = 0;
162 hdr.size = cISDataMappings::GetSize(hdr.id);
163 bool inQuotes = false;
164 uint32_t foundQuotes = 0;
165 memset(buf, 0, hdr.size);
166 for (string::const_iterator i = start; i < s.end() && index < columnHeaders.size(); i++)
167 {
168 if (*i == ',' && !inQuotes)
169 {
170 // end field
171 columnData = string(start + foundQuotes, i - foundQuotes);
172 start = i + 1;
173 const data_info_t& data = columnHeaders[index++];
174 if (data.dataOffset < MAX_DATASET_SIZE && !cISDataMappings::StringToData(columnData.c_str(), (int)columnData.length(), &hdr, buf, data))
175 {
176 return false;
177 }
178 foundQuotes = false;
179 }
180 else if (*i == '"')
181 {
182 inQuotes = !inQuotes;
183 foundQuotes |= (uint32_t)inQuotes;
184 }
185 }
186 return true;
187}

190bool cDataCSV::DataToStringCSV(const p_data_hdr_t& hdr, const uint8_t* buf, string& csv)
191{
192 csv.clear();
193 string columnData;
194 const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(hdr.id);
195 if (offsetMap == NULLPTR)
196 {
197 return false;
198 }
199 char tmp[IS_DATA_MAPPING_MAX_STRING_LENGTH];
200 const uint8_t* bufPtr = buf;
201 uint8_t tmpBuffer[MAX_DATASET_SIZE];
202 uint32_t size = cISDataMappings::GetSize(hdr.id);
203 if (size > hdr.size)
204 {
205 // copy into temp buffer, zeroing out bytes that are not part of this packet
206 memset(tmpBuffer, 0, hdr.offset);
207 memcpy(tmpBuffer + hdr.offset, buf, hdr.size);
208 uint32_t dataEnd = hdr.offset + hdr.size;
209 memset(tmpBuffer + dataEnd, 0, size - dataEnd);
210 bufPtr = tmpBuffer;
211 }
212
213 // copy header as we are now storing an entire struct, even if we got a partial hdr and buf
214 p_data_hdr_t hdrCopy = hdr;
215 hdrCopy.offset = 0;
216 hdrCopy.size = size;
217
218 for (map_name_to_info_t::const_iterator offset = offsetMap->begin(); offset != offsetMap->end(); offset++)
219 {
220 cISDataMappings::DataToString(offset->second, &hdrCopy, bufPtr, tmp);
221 if (csv.length() != 0)
222 {
223 csv += ",";
224 }
225 csv += tmp;
226 }
227 csv += "\n";
228 return true;
229}
