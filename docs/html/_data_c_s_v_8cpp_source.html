65 int cDataCSV::ReadHeaderFromFile(FILE* pFile, uint32_t id, vector<data_info_t>& columnHeaders)
66 {
67 #if PLATFORM_IS_EMBEDDED
68 
69  return 0;
70 
71 #else
72 
73  (void)id;
74 
75  char line[8192];
76  line[0] = '\0';
77  if (fgets(line, _ARRAY_BYTE_COUNT(line), pFile) == NULL || ferror(pFile) != 0)
78  {
79  return 0;
80  }
81  const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(id);
82  stringstream stream(line);
83  string columnHeader;
84  columnHeaders.clear();
85  while (stream.good())
86  {
87  getline(stream, columnHeader, ',');
88  while (columnHeader.length() > 0 && (columnHeader[columnHeader.size() - 1] == '\r' || columnHeader[columnHeader.size() - 1] == '\n'))
89  {
90  columnHeader.resize(columnHeader.size() - 1);
91  }
92  if (columnHeader.length() == 0)
93  {
94  columnHeader = "UNKNOWN";
95  }
96  if (columnHeader == "_ID_")
97  {
98  columnHeaders.push_back({ 0xFFFFFFFF, 0xFFFFFFFF, DataTypeInt64, (eDataFlags)0, "_ID_" });
99  }
100  else
101  {
102  map_name_to_info_t::const_iterator offset = offsetMap->find(columnHeader);
103  if (offset == offsetMap->end())
104  {
105  columnHeaders.push_back({ 0xFFFFFFFF, 0xFFFFFFFF, DataTypeBinary, (eDataFlags)0, columnHeader });
106  }
107  else
108  {
109  columnHeaders.push_back(offset->second);
110  }
111  }
112  }
113  return (int)strnlen(line, _ARRAY_BYTE_COUNT(line));
114 #endif
115 }

143 bool cDataCSV::StringCSVToData(string& s, p_data_hdr_t& hdr, uint8_t* buf, uint32_t bufSize, const vector<data_info_t>& columnHeaders)
144 {
145  const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(hdr.id);
146  if (offsetMap == NULLPTR || hdr.offset != 0 || hdr.size == 0 || bufSize < hdr.size)
147  {
148  return false;
149  }
150  while (s.length() > 0 && (s[s.size() - 1] == '\r' || s[s.size() - 1] == '\n'))
151  {
152  s.resize(s.size() - 1);
153  }
154  // for parsing logic, last comma will allow last field to parse properly
155  s += ",";
156  string columnData;
157  string::const_iterator start = s.begin();
158  uint32_t index = 0;
159  hdr.offset = 0;
160  hdr.size = cISDataMappings::GetSize(hdr.id);
161  bool inQuotes = false;
162  uint32_t foundQuotes = 0;
163  memset(buf, 0, hdr.size);
164  for (string::const_iterator i = start; i < s.end() && index < columnHeaders.size(); i++)
165  {
166  if (*i == ',' && !inQuotes)
167  {
168  // end field
169  columnData = string(start + foundQuotes, i - foundQuotes);
170  start = i + 1;
171  const data_info_t& data = columnHeaders[index++];
172  if (data.dataOffset < MAX_DATASET_SIZE && !cISDataMappings::StringToData(columnData.c_str(), (int)columnData.length(), &hdr, buf, data))
173  {
174  return false;
175  }
176  foundQuotes = false;
177  }
178  else if (*i == '"')
179  {
180  inQuotes = !inQuotes;
181  foundQuotes |= (uint32_t)inQuotes;
182  }
183  }
184  return true;
185 }

188 bool cDataCSV::DataToStringCSV(const p_data_hdr_t& hdr, const uint8_t* buf, string& csv)
189 {
190  csv.clear();
191  string columnData;
192  const map_name_to_info_t* offsetMap = cISDataMappings::GetMapInfo(hdr.id);
193  if (offsetMap == NULLPTR)
194  {
195  return false;
196  }
197  char tmp[IS_DATA_MAPPING_MAX_STRING_LENGTH];
198  const uint8_t* bufPtr = buf;
199  uint8_t tmpBuffer[MAX_DATASET_SIZE];
200  uint32_t size = cISDataMappings::GetSize(hdr.id);
201  if (size > hdr.size)
202  {
203  // copy into temp buffer, zeroing out bytes that are not part of this packet
204  memset(tmpBuffer, 0, hdr.offset);
205  memcpy(tmpBuffer + hdr.offset, buf, hdr.size);
206  uint32_t dataEnd = hdr.offset + hdr.size;
207  memset(tmpBuffer + dataEnd, 0, size - dataEnd);
208  bufPtr = tmpBuffer;
209  }
210 
211  // copy header as we are now storing an entire struct, even if we got a partial hdr and buf
212  p_data_hdr_t hdrCopy = hdr;
213  hdrCopy.offset = 0;
214  hdrCopy.size = size;
215 
216  for (map_name_to_info_t::const_iterator offset = offsetMap->begin(); offset != offsetMap->end(); offset++)
217  {
218  cISDataMappings::DataToString(offset->second, &hdrCopy, bufPtr, tmp);
219  if (csv.length() != 0)
220  {
221  csv += ",";
222  }
223  csv += tmp;
224  }
225  csv += "\n";
226  return true;
227 }
