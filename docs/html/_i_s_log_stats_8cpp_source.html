37 void cLogStatDataId::LogTimestamp(double timestamp)
38 {
39  // check for corrupt data
40  if (_ISNAN(timestamp) || timestamp < 0.0 || timestamp > 999999999999.0)
41  {
42  return;
43  }
44  else if (lastTimestamp > 0.0)
45  {
46  double delta = fabs(timestamp - lastTimestamp);
47  minTimestampDelta = _MIN(delta, minTimestampDelta);
48  maxTimestampDelta = _MAX(delta, maxTimestampDelta);
49  totalTimeDelta += delta;
50  averageTimeDelta = (totalTimeDelta / (double)++timestampDeltaCount);
51  if (lastTimestampDelta != 0.0 && (fabs(delta - lastTimestampDelta) > (lastTimestampDelta * 0.5)))
52  {
53  timestampDropCount++;
54  }
55  lastTimestampDelta = delta;
56  }
57  lastTimestamp = timestamp;
58 }
