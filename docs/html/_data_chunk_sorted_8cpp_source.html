32 // This function reads the next sorted chunk matching the specified DID from all of the files. The file pointers are advanced
33 // if the current chunk dataSerNum is older than the current m_dataSerNum. Files are closed if the last chuch dataSerNum
34 // is older than the current m_dataSerNum.
35 // Returns number of bytes read, or -1 for error
36 int32_t cSortedDataChunk::ReadFromFiles(vector<cISLogFileBase*>& pFiles, uint32_t id, uint32_t dataSerNum)
37 {
38  if (pFiles.size() == 0 || pFiles.front() == NULLPTR)
39  {
40  return -1;
41  }
42 
43  // reset state, prepare to read from file
44  Clear();
45 
46  int32_t nBytes = -1;
47  long int restorePos;
48 
49  // Search through all files for chunk with matching DID
50  for (unsigned int i = 0; i < pFiles.size(); )
51  {
52  cISLogFileBase* pFile = pFiles[i];
53 
54  // Set backup file pointer
55  restorePos = pFile->tell();
56 
57  // Search through single file for chunk with matching DID
58  bool allowFileTrim = true; // This flag indicates there is/are new chunk(s) that should not be trimmed from the restorePos file pointer position.
59  while(1)
60  {
61  nBytes = ReadFromFile(pFile);
62 
63  if (nBytes <= 0)
64  { // Match not found in file
65  break;
66  }
67 
68  p_cnk_data_t* cnkData = (p_cnk_data_t*)GetDataPtr();
69  if (cnkData == NULLPTR)
70  { // No more data. Match not found
71  break;
72  }
73 
74  if (m_subHdr.dHdr.id == id && GetDataSerNum() >= dataSerNum)
75  { // Found matching chunk
76  if (allowFileTrim)
77  {
78  pFile = TrimFile(i, pFiles, restorePos);
79  }
80  break;
81  }
82  else
83  { // Chunk not found
84  nBytes = -1;
85  }
86 
87  if (allowFileTrim)
88  {
89  uint32_t readDataSerNum = GetDataSerNum();
90  if (readDataSerNum < dataSerNum || readDataSerNum == UINT_MAX)
91  { // Move file pointer past old/invalid chunks
92  pFile = TrimFile(i, pFiles, restorePos);
93  }
94  else
95  {
96  allowFileTrim = false;
97  }
98  }
99  }
100 
101  if (pFile)
102  { // Restore file pointer if file is still opened
103  pFile->seek(restorePos);
104  }
105 
106  if (nBytes > 0)
107  { // Found chunk
108  return nBytes;
109  }
110 
111  // No matching chunks in file
112  i++;
113  }
114 
115  return -1;
116 }
