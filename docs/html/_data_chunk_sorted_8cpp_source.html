34// This function reads the next sorted chunk matching the specified DID from all of the files. The file pointers are advanced
35// if the current chunk dataSerNum is older than the current m_dataSerNum. Files are closed if the last chuch dataSerNum
36// is older than the current m_dataSerNum.
37// Returns number of bytes read, or -1 for error
38int32_t cSortedDataChunk::ReadFromFiles(vector<cISLogFileBase*>& pFiles, uint32_t id, uint32_t dataSerNum)
39{
40 if (pFiles.size() == 0 || pFiles.front() == NULLPTR)
41 {
42 return -1;
43 }
44
45 // reset state, prepare to read from file
46 Clear();
47
48 int32_t nBytes = -1;
49 long int restorePos;
50
51 // Search through all files for chunk with matching DID
52 for (unsigned int i = 0; i < pFiles.size(); )
53 {
54 cISLogFileBase* pFile = pFiles[i];
55
56 // Set backup file pointer
57 restorePos = pFile->tell();
58
59 // Search through single file for chunk with matching DID
60 bool allowFileTrim = true; // This flag indicates there is/are new chunk(s) that should not be trimmed from the restorePos file pointer position.
61 while(1)
62 {
63 nBytes = ReadFromFile(pFile);
64
65 if (nBytes <= 0)
66 { // Match not found in file
67 break;
68 }
69
70 p_cnk_data_t* cnkData = (p_cnk_data_t*)GetDataPtr();
71 if (cnkData == NULLPTR)
72 { // No more data. Match not found
73 break;
74 }
75
76 if (m_subHdr.dHdr.id == id && GetDataSerNum() >= dataSerNum)
77 { // Found matching chunk
78 if (allowFileTrim)
79 {
80 pFile = TrimFile(i, pFiles, restorePos);
81 }
82 break;
83 }
84 else
85 { // Chunk not found
86 nBytes = -1;
87 }
88
89 if (allowFileTrim)
90 {
91 uint32_t readDataSerNum = GetDataSerNum();
92 if (readDataSerNum < dataSerNum || readDataSerNum == UINT_MAX)
93 { // Move file pointer past old/invalid chunks
94 pFile = TrimFile(i, pFiles, restorePos);
95 }
96 else
97 {
98 allowFileTrim = false;
99 }
100 }
101 }
102
103 if (pFile)
104 { // Restore file pointer if file is still opened
105 pFile->seek(restorePos);
106 }
107
108 if (nBytes > 0)
109 { // Found chunk
110 return nBytes;
111 }
112
113 // No matching chunks in file
114 i++;
115 }
116
117 return -1;
118}

121cISLogFileBase* cSortedDataChunk::TrimFile(unsigned int i, vector<cISLogFileBase*>& pFiles, long int &restorePos)
122{
123 cISLogFileBase* pFile = pFiles[i];
124
125 // Move file pointer past old chunks
126 //pFile->getpos(&restorePos);
127 restorePos = pFile->tell();
128
129 if (pFile->eof())
130 { // No more data in file
131 pFile->close();
132
133 // Remove file at index from vector
134 pFiles.erase(pFiles.begin() + i);
135 }
136
137 if (i < pFiles.size())
138 { // More files available
139 return pFiles[i];
140 }
141 else
142 { // No more files available
143 return NULLPTR;
144 }
145}
