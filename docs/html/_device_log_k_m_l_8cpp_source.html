332 bool cDeviceLogKML::SaveData(p_data_hdr_t *dataHdr, const uint8_t *dataBuf)
333 {
334  cDeviceLog::SaveData(dataHdr, dataBuf);
335 
336  // Save data to file
337  if (!WriteDateToFile(dataHdr, dataBuf))
338  {
339  return false;
340  }
341 
342  // Convert formats
343  uDatasets& d = (uDatasets&)(*dataBuf);
344  switch (dataHdr->id)
345  {
346  case DID_INS_2:
347  ins_1_t ins1;
348  ins1.week = d.ins2.week;
349  ins1.timeOfWeek = d.ins2.timeOfWeek;
350  ins1.hdwStatus = d.ins2.hdwStatus;
351  ins1.insStatus = d.ins2.insStatus;
352  quat2euler(d.ins2.qn2b, ins1.theta);
353  memcpy(ins1.uvw, d.ins2.uvw, 12);
354  memcpy(ins1.lla, d.ins2.lla, 24);
355 
356  p_data_hdr_t dHdr = { DID_INS_1 , sizeof(ins_1_t), 0 };
357  // Save data to file
358  WriteDateToFile(&dHdr, (uint8_t*)&ins1);
359  break;
360  }
361 
362  return true;
363 }

366 bool cDeviceLogKML::WriteDateToFile(const p_data_hdr_t *dataHdr, const uint8_t* dataBuf)
367 {
368  int kid = cDataKML::DID_TO_KID(dataHdr->id);
369 
370  // Only use selected data
371  if (kid < 0)
372  {
373  return true;
374  }
375 
376  // Reference current log
377  sKmlLog &log = m_Log[kid];
378 
379  // Write date to file
380  int nBytes = m_kml.WriteDataToFile(log.data, dataHdr, dataBuf);
381 
382  // File byte size
383  nBytes = (int)(log.data.size() * cDataKML::BYTES_PER_KID(kid));
384  log.fileSize = nBytes;
385  m_fileSize = _MAX(m_fileSize, log.fileSize);
386  m_logSize = nBytes;
387 
388  if (log.fileSize >= m_maxFileSize)
389  {
390  // Close existing file
391  CloseWriteFile(kid, log);
392  }
393
