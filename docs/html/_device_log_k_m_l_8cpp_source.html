80#define BUF_SIZE 100

82 int styleCnt = 1;
83 string altitudeMode;
84 string iconScale = "0.5";
85 string labelScale = "0.5";

103 // iconUrl = "http://earth.google.com/images/kml-icons/track-directional/track-none.png";

105 // colors are ABGR
106 switch (kid)

108 case cDataKML::KID_INS:

284 }

292

295 bool deadReckoning = log.data[i].deadReckoning;
296
297 // Track style
298 styleStr = string("stylesel_") + to_string(styleCnt++);
299
300 Style = new TiXmlElement("Style");
301 Style->SetAttribute("id", styleStr);
302 Document->LinkEndChild(Style);

304 LineStyle = new TiXmlElement("LineStyle");

311 elem = new TiXmlElement("colorMode");

321 Document->LinkEndChild(Placemark);

323 elem = new TiXmlElement("name");
324 elem->LinkEndChild(new TiXmlText("Tracks"+to_string(tracksNum)));

335 LineString = new TiXmlElement("LineString");

339 int j = 0;

342 if (deadReckoning != log.data[i].deadReckoning)

345 }

347 sKmlLogData& item = (log.data[i]);
348 double lat = _CLAMP(item.lla[0] * DEG2RADMULT, -C_PIDIV2, C_PIDIV2) * RAD2DEGMULT;
349 double lon = _CLAMP(item.lla[1] * DEG2RADMULT, -C_PI, C_PI) * RAD2DEGMULT;

353 {
354 j++;
355 }
356
357 snprintf(buf, BUF_SIZE, "%.8lf,%.8lf,%.3lf ", lon, lat, alt);
358 // if (strcmp("-111.65863637,40.05570543,1418.282 ", buf) == 0)
359 // {
360 // j++;
361 // }
362 // qDebug() << string(buf);

392 log.data.clear();

394 return true;

423bool cDeviceLogKML::SaveData(p_data_hdr_t *dataHdr, const uint8_t *dataBuf)
424{
425 cDeviceLog::SaveData(dataHdr, dataBuf);
426
427 // Save data to file
428 if (!WriteDateToFile(dataHdr, dataBuf))
429 {
430 return false;
431 }
432
433 // Convert formats
434 uDatasets& d = (uDatasets&)(*dataBuf);
435 switch (dataHdr->id)
436 {
437 case DID_INS_2:
438 ins_1_t ins1;
439 ins1.week = d.ins2.week;
440 ins1.timeOfWeek = d.ins2.timeOfWeek;
441 ins1.hdwStatus = d.ins2.hdwStatus;
442 ins1.insStatus = d.ins2.insStatus;
443 quat2euler(d.ins2.qn2b, ins1.theta);
444 memcpy(ins1.uvw, d.ins2.uvw, 12);
445 memcpy(ins1.lla, d.ins2.lla, 24);
446 p_data_hdr_t dHdr = { DID_INS_1 , sizeof(ins_1_t), 0 };
447 // Save data to file
448 if (!WriteDateToFile(&dHdr, (uint8_t*)&ins1))
449 {
450 return false;
451 }
452 break;
453 }
454
455 return true;
456}

459bool cDeviceLogKML::WriteDateToFile(const p_data_hdr_t *dataHdr, const uint8_t* dataBuf)
460{
461 int kid = cDataKML::DID_TO_KID(dataHdr->id);
462
463 // Only use selected data
464 if (kid < 0)
465 {
466 return true;
467 }
468
469 // Reference current log
470 sKmlLog &log = m_Log[kid];
471
472 // Write date to file
473 m_kml.WriteDataToFile(log.data, dataHdr, dataBuf);
474
475 // File byte size
476 int nBytes = (int)(log.data.size() * cDataKML::BYTES_PER_KID(kid));
477 log.fileSize = nBytes;
478 m_fileSize = _MAX(m_fileSize, log.fileSize);
479 m_logSize = nBytes;
480
481 if (log.fileSize >= m_maxFileSize)
482 {
483 // Close existing file
484 if (!CloseWriteFile(kid, log))
485 {
486 return false;
487 }
488 }
489
490 return true;
491}
