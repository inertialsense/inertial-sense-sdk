72 // less than, not equal
73 bool operator() (const std::string& s1, const std::string& s2) const
74 {
75 // return std::lexicographical_compare(s1.begin(), s1.end(), s2.begin(), s2.end(), nocase_compare());
76
77 // we don't need unicode or fancy language handling here, and we do not want branching
78 // so we have hand-coded a highly performant ASCII case insensitive compare here.
79 // this custom code is 3x speed of lexicographical_compare
80 char c1, c2;
81 const char* ptr1 = s1.c_str();
82 const char* ptr2 = s2.c_str();
83 size_t size1 = s1.size();
84 size_t size2 = s2.size();
85 int sizeDiff = (int)size1 - (int)size2;
86
87 // branchless min
88 // y + ((x - y) & ((x - y) >> (sizeof(int) * CHAR_BIT - 1)));
89 size_t minSize = size2 + (sizeDiff & (sizeDiff >> (sizeof(int) * CHAR_BIT - 1)));
90
91 for (size_t i = 0; i < minSize; i++)
92 {
93 c1 = g_asciiToLowerMap[(int)ptr1[i]];
94 c2 = g_asciiToLowerMap[(int)ptr2[i]];
95 if (c1 != c2)
96 {
97 return (c1 < c2);
98 }
99 }
100 return (s1.size() < s2.size());
101 }
102};
