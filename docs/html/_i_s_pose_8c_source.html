46/* Quaternion Conjugate: q* = [ w, -x, -y, -z ] of quaterion q = [ w, x, y, z ]
47 * Rotation in opposite direction.
48 */
49void quatConj(ixQuat result, const ixQuat q)
50{
51 result[0] = q[0];
52 result[1] = -q[1];
53 result[2] = -q[2];
54 result[3] = -q[3];
55}

58/*
59 * Product of two Quaternions. Order of q1 and q2 matters (same as applying two successive DCMs)!!!
60 * Combines two quaternion rotations into one rotation.
61 * result = q1 * q2
62 * Reference: http://www.mathworks.com/help/aeroblks/quaternionmultiplication.html
63 */
64void mul_Quat_Quat(ixQuat result, const ixQuat q1, const ixQuat q2)
65{
66 result[0] = q1[0]*q2[0] - q1[1]*q2[1] - q1[2]*q2[2] - q1[3]*q2[3];
67 result[1] = q1[0]*q2[1] + q1[1]*q2[0] - q1[2]*q2[3] + q1[3]*q2[2];
68 result[2] = q1[0]*q2[2] + q1[1]*q2[3] + q1[2]*q2[0] - q1[3]*q2[1];
69 result[3] = q1[0]*q2[3] - q1[1]*q2[2] + q1[2]*q2[1] + q1[3]*q2[0];
70}

72/*
73* Product of two Quaternions. Order of q1 and q2 matters (same as applying two successive DCMs)!!!
74* Combines two quaternion rotations into one rotation.
75* result = quatConj(q1) * q2
76* Reference: http://www.mathworks.com/help/aeroblks/quaternionmultiplication.html
77*/
78void mul_ConjQuat_Quat(ixQuat result, const ixQuat qc, const ixQuat q2)
79{
80 result[0] = qc[0]*q2[0] + qc[1]*q2[1] + qc[2]*q2[2] + qc[3]*q2[3];
81 result[1] = qc[0]*q2[1] - qc[1]*q2[0] + qc[2]*q2[3] - qc[3]*q2[2];
82 result[2] = qc[0]*q2[2] - qc[1]*q2[3] - qc[2]*q2[0] + qc[3]*q2[1];
83 result[3] = qc[0]*q2[3] + qc[1]*q2[2] - qc[2]*q2[1] - qc[3]*q2[0];
84}

86/*
87* Product of two Quaternions. Order of q1 and q2 matters (same as applying two successive DCMs)!!!
88* Combines two quaternion rotations into one rotation.
89* result = q1 * quatConj(q2)
90* Reference: http://www.mathworks.com/help/aeroblks/quaternionmultiplication.html
91*/
92void mul_Quat_ConjQuat(ixQuat result, const ixQuat q1, const ixQuat qc)
93{
94 result[0] = q1[0]*qc[0] + q1[1]*qc[1] + q1[2]*qc[2] + q1[3]*qc[3];
95 result[1] = -q1[0]*qc[1] + q1[1]*qc[0] + q1[2]*qc[3] - q1[3]*qc[2];
96 result[2] = -q1[0]*qc[2] - q1[1]*qc[3] + q1[2]*qc[0] + q1[3]*qc[1];
97 result[3] = -q1[0]*qc[3] + q1[1]*qc[2] - q1[2]*qc[1] + q1[3]*qc[0];
98}

100/*
101 * Division of two Quaternions. Order matters!!!
102 * result = q1 / q2.
103 * Reference: http://www.mathworks.com/help/aeroblks/quaterniondivision.html
104 */
105void div_Quat_Quat(ixQuat result, const ixQuat q1, const ixQuat q2)
106{
107 f_t d = (f_t)1.0 / (q1[0] * q1[0] + q1[1] * q1[1] + q1[2] * q1[2] + q1[3] * q1[3]);
108
109 result[0] = (q1[0]*q2[0] + q1[1]*q2[1] + q1[2]*q2[2] + q1[3]*q2[3]) * d;
110 result[1] = (q1[0]*q2[1] - q1[1]*q2[0] - q1[2]*q2[3] + q1[3]*q2[2]) * d;
111 result[2] = (q1[0]*q2[2] + q1[1]*q2[3] - q1[2]*q2[0] - q1[3]*q2[1]) * d;
112 result[3] = (q1[0]*q2[3] - q1[1]*q2[2] + q1[2]*q2[1] - q1[3]*q2[0]) * d;
113}

116/* Quaternion rotation from vector v1 to vector v2.
117 */
118void quat_Vec3_Vec3(ixQuat result, const ixVector3 v1, const ixVector3 v2)
119{
120 ixVector3 w1, w2;
121
122 // Normalize input vectors
123 mul_Vec3_X( w1, v1, recipNorm_Vec3(v1) );
124 mul_Vec3_X( w2, v2, recipNorm_Vec3(v2) );
125
126 // q[1:3]
127 cross_Vec3( &result[1], w1, w2 );
128
129 // q[0]
130 result[0] = (f_t)(_SQRT( dot_Vec3(w1) * dot_Vec3(w1) ) + dot_Vec3_Vec3(w1, w2));
131
132 // Normalize quaternion
133 div_Vec4_X( result, result, mag_Vec4(result) );
134}

137/* Computationally simple means to apply quaternion rotation to a vector.
138 * Requires quaternion be normalized first.
139 * If quaternion describes current attitude, then rotation is body -> inertial frame.
140 * Equivalent to a DCM.T * vector multiply.
141 */
142void quatRot(ixVector3 result, const ixQuat q, const ixVector3 v)
143{
144 ixVector3 t;
145 cross_Vec3( t, &q[1], v );
146 mul_Vec3_X( t, t, (f_t)2.0 );
147
148 cross_Vec3( result, &q[1], t );
149 mul_Vec3_X( t, t, q[0] );
150 add_Vec3_Vec3( result, result, t );
151 add_Vec3_Vec3( result, result, v );
152}

155/* Computationally simple means to apply quaternion conjugate (opposite) rotation to a vector
156 * (18 multiplies, 6 subtracts, 6 adds). Using a DCM uses (27 multiplies, 12 adds, 6 subtracts).
157 * Requires quaternion be normalized first.
158 * If quaternion describes current attitude, then rotation is inertial -> body frame.
159 * Equivalent to a DCM * vector multiply.
160 */
161void quatConjRot(ixVector3 result, const ixQuat q, const ixVector3 v)
162{
163 ixQuat qC;
164 ixVector3 t;
165
166 // Rotation in opposite direction
167 quatConj( qC, q );
168
169 cross_Vec3( t, &qC[1], v );
170 mul_Vec3_X( t, t, (f_t)2.0 );
171
172 cross_Vec3( result, &qC[1], t );
173 mul_Vec3_X( t, t, qC[0] );
174 add_Vec3_Vec3( result, result, t );
175 add_Vec3_Vec3( result, result, v );
176}

179/*
180 * This will convert from quaternions to euler angles
181 * q(W,X,Y,Z) -> euler(phi,theta,psi) (rad)
182 *
183 * Reference: http://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
184 */
185void quat2euler(const ixQuat q, ixEuler theta)
186{
187 float sinang = 2 * (q[0] * q[2] - q[3] * q[1]);
188 if (sinang > 1.0f) { sinang = 1.0f; }
189 if (sinang < -1.0f) { sinang = -1.0f; }
190
191 theta[0] = _ATAN2(2 * (q[0]*q[1] + q[2]*q[3]), 1 - 2 * (q[1]*q[1] + q[2]*q[2]));
192 theta[1] = _ASIN (sinang);
193 theta[2] = _ATAN2(2 * (q[0]*q[3] + q[1]*q[2]), 1 - 2 * (q[2]*q[2] + q[3]*q[3]));
194}
195void quat2phiTheta(const ixQuat q, f_t *phi, f_t *theta)
196{
197 float sinang = 2 * (q[0] * q[2] - q[3] * q[1]);
198 if (sinang > 1.0f) { sinang = 1.0f; }
199 if (sinang < -1.0f) { sinang = -1.0f; }
200
201 *phi = _ATAN2( 2 * (q[0] * q[1] + q[2] * q[3]), 1 - 2 * (q[1] * q[1] + q[2] * q[2]) );
202 *theta = _ASIN (sinang);
203}
204void quat2psi(const ixQuat q, f_t *psi)
205{
206 float sinang = 2 * (q[0] * q[2] - q[3] * q[1]);
207 if (sinang > 1.0f) { sinang = 1.0f; }
208 if (sinang < -1.0f) { sinang = -1.0f; }
209
210 *psi = _ATAN2(2 * (q[0]*q[3] + q[1]*q[2]), 1 - 2 * (q[2]*q[2] + q[3]*q[3]));
211}

214/*
215 * This will convert from euler angles to quaternion vector
216 * euler(phi,theta,psi) (rad) -> q(W,X,Y,Z)
217 */
218void euler2quat(const ixEuler euler, ixQuat q)
219{
220 f_t hphi = euler[0] * (f_t)0.5;
221 f_t hthe = euler[1] * (f_t)0.5;
222 f_t hpsi = euler[2] * (f_t)0.5;
223
224 f_t shphi = _SIN(hphi);
225 f_t chphi = _COS(hphi);
226
227 f_t shthe = _SIN(hthe);
228 f_t chthe = _COS(hthe);
229
230 f_t shpsi = _SIN(hpsi);
231 f_t chpsi = _COS(hpsi);
232
233 q[0] = chphi * chthe * chpsi + shphi * shthe * shpsi;
234 q[1] = shphi * chthe * chpsi - chphi * shthe * shpsi;
235 q[2] = chphi * shthe * chpsi + shphi * chthe * shpsi;
236 q[3] = chphi * chthe * shpsi - shphi * shthe * chpsi;
237}

240/*
241* Convert ECEF quaternion to NED euler at specified LLA (rad)
242*/
243void qe2b2EulerNedLLA(ixVector3 eul, const ixVector4 qe2b, const ixVector3d lla)
244{
245 ixVector3 eulned;
246 ixVector4 qe2n;
247 ixVector4 qn2b;
248
249 eulned[0] = 0.0f;
250 eulned[1] = ((float)-lla[0]) - 0.5f * C_PI_F;
251 eulned[2] = (float)lla[1];
252 euler2quat(eulned, qe2n);
253 mul_Quat_ConjQuat(qn2b, qe2b, qe2n);
254 quat2euler(qn2b, eul);
255}

259 * This will construct a direction cosine matrix from
260 * the psi angle - rotates from NE to body frame
261 *
262 * body = tBL(2,2)*NE
263 *
264 */
265void psiDCM(const f_t psi, ixMatrix2 m)
266{
267 f_t cpsi = _COS(psi); // cos(psi)
268 f_t spsi = _SIN(psi); // sin(psi)
269
270 // Row 1
271 m[0] = cpsi;
272 m[1] = spsi;
273 // Row 2
274 m[2] = -spsi;
275 m[3] = cpsi;
276}

295/*
296 * This will construct a direction cosine matrix from
297 * euler angles in the standard rotation sequence
298 * [phi][theta][psi] from NED to body frame
299 *
300 * body = tBL(3,3)*NED
301 *
302 * reference: http://en.wikipedia.org/wiki/Rotation_representation_%28mathematics%29
303 */
304void eulerDCM(const ixEuler euler, ixMatrix3 m)
305{
306 f_t cphi = _COS(euler[0]); // cos(phi)
307 f_t cthe = _COS(euler[1]); // cos(theta)
308 f_t cpsi = _COS(euler[2]); // cos(psi)
309
310 f_t sphi = _SIN(euler[0]); // sin(phi)
311 f_t sthe = _SIN(euler[1]); // sin(theta)
312 f_t spsi = _SIN(euler[2]); // sin(psi)
313
314 // Row 1
315 m[0] = cpsi*cthe;
316 m[1] = spsi*cthe;
317 m[2] = -sthe;
318 // Row 2
319 m[3] = -spsi*cphi + cpsi*sthe*sphi;
320 m[4] = cpsi*cphi + spsi*sthe*sphi;
321 m[5] = cthe*sphi;
322 // Row 3
323 m[6] = spsi*sphi + cpsi*sthe*cphi;
324 m[7] = -cpsi*sphi + spsi*sthe*cphi;
325 m[8] = cthe*cphi;
326}

353* This will construct the transpose matrix of
354* the direction cosine matrix from
355* euler angles in the standard rotation sequence
356* [phi][theta][psi] from NED to body frame
357*
358* body = tBL(3,3)*NED
359*
360* reference: http://en.wikipedia.org/wiki/Rotation_representation_%28mathematics%29
361*/
362void eulerDCM_Trans(const ixEuler euler, ixMatrix3 m )
363{
364 f_t cphi = _COS( euler[0] ); // cos(phi)
365 f_t cthe = _COS( euler[1] ); // cos(theta)
366 f_t cpsi = _COS( euler[2] ); // cos(psi)
367
368 f_t sphi = _SIN( euler[0] ); // sin(phi)
369 f_t sthe = _SIN( euler[1] ); // sin(theta)
370 f_t spsi = _SIN( euler[2] ); // sin(psi)
371
372 // Col 1
373 m[0] = cpsi*cthe;
374 m[3] = spsi*cthe;
375 m[6] = -sthe;
376 // Col 2
377 m[1] = -spsi*cphi + cpsi*sthe*sphi;
378 m[4] = cpsi*cphi + spsi*sthe*sphi;
379 m[7] = cthe*sphi;
380 // Col 3
381 m[2] = spsi*sphi + cpsi*sthe*cphi;
382 m[5] = -cpsi*sphi + spsi*sthe*cphi;
383 m[8] = cthe*cphi;
384}

404/*
405 * This will construct a direction cosine matrix from
406 * quaternions in the standard rotation sequence
407 * [phi][theta][psi] from NED to body frame
408 * (18 multiplies, 6 adds, 6 subtracts)
409 *
410 * body = tBL(3,3)*NED
411 * q(4,1)
412 *
413 * Reference: http://en.wikipedia.org/wiki/Rotation_representation_%28mathematics%29
414 */
415void quatDCM(const ixQuat q, ixMatrix3 mat)
416{
417 f_t q0q1 = q[0]*q[1];
418 f_t q0q2 = q[0]*q[2];
419 f_t q0q3 = q[0]*q[3];
420 f_t q1q1 = q[1]*q[1];
421 f_t q1q2 = q[1]*q[2];
422 f_t q1q3 = q[1]*q[3];
423 f_t q2q2 = q[2]*q[2];
424 f_t q2q3 = q[2]*q[3];
425 f_t q3q3 = q[3]*q[3];
426
427 // Row 1
428 mat[0] = (f_t)1.0 - (f_t)2 * (q2q2 + q3q3);
429 mat[1] = (f_t)2 * (q1q2 + q0q3);
430 mat[2] = (f_t)2 * (q1q3 - q0q2);
431 // Row 2
432 mat[3] = (f_t)2 * (q1q2 - q0q3);
433 mat[4] = (f_t)1.0 - (f_t)2 * (q1q1 + q3q3);
434 mat[5] = (f_t)2 * (q2q3 + q0q1);
435 // Row 3
436 mat[6] = (f_t)2 * (q1q3 + q0q2);
437 mat[7] = (f_t)2 * (q2q3 - q0q1);
438 mat[8] = (f_t)1.0 - (f_t)2 * (q1q1 + q2q2);
439}
440void quatdDCM(const ixVector4d q, ixMatrix3 mat )
441{
442 f_t q0q1 = (f_t)(q[0] * q[1]);
443 f_t q0q2 = (f_t)(q[0] * q[2]);
444 f_t q0q3 = (f_t)(q[0] * q[3]);
445 f_t q1q1 = (f_t)(q[1] * q[1]);
446 f_t q1q2 = (f_t)(q[1] * q[2]);
447 f_t q1q3 = (f_t)(q[1] * q[3]);
448 f_t q2q2 = (f_t)(q[2] * q[2]);
449 f_t q2q3 = (f_t)(q[2] * q[3]);
450 f_t q3q3 = (f_t)(q[3] * q[3]);
451
452 // Row 1
453 mat[0] = (f_t)1.0 - (f_t)2.0 * (q2q2 + q3q3);
454 mat[1] = (f_t)2.0 * (q1q2 + q0q3);
455 mat[2] = (f_t)2.0 * (q1q3 - q0q2);
456 // Row 2
457 mat[3] = (f_t)2.0 * (q1q2 - q0q3);
458 mat[4] = (f_t)1.0 - (f_t)2.0 * (q1q1 + q3q3);
459 mat[5] = (f_t)2.0 * (q2q3 + q0q1);
460 // Row 3
461 mat[6] = (f_t)2.0 * (q1q3 + q0q2);
462 mat[7] = (f_t)2.0 * (q2q3 - q0q1);
463 mat[8] = (f_t)1.0 - (f_t)2.0 * (q1q1 + q2q2);
464}

466/*
467 * This will construct quaternions from a direction cosine
468 * matrix in the standard rotation sequence.
469 * [phi][theta][psi] from NED to body frame
470 *
471 * body = tBL(3,3)*NED
472 * q(4,1)
473 *
474 * Reference: http://en.wikipedia.org/wiki/Rotation_representation_%28mathematics%29
475 */
476void DCMquat(const ixMatrix3 mat, ixQuat q)
477{
478 f_t d;
479
480 q[0] = (f_t)0.5 * _SQRT((f_t)1.0 + mat[0] + mat[4] + mat[8]);
481
482 d = (f_t)1.0 / ((f_t)4.0 * q[0]);
483
484 q[1] = d * (mat[5] - mat[7]);
485 q[2] = d * (mat[6] - mat[2]);
486 q[3] = d * (mat[1] - mat[3]);
487}

489/*
490 * This will construct the euler omega-cross matrix
491 * wx(3,3)
492 * p, q, r (rad/sec)
493 */
494void eulerWx(const ixEuler euler, ixMatrix3 mat)
495{
496 f_t p = euler[0];
497 f_t q = euler[1];
498 f_t r = euler[2];
499
500 // Row 1
501 mat[0] = 0;
502 mat[1] = -r;
503 mat[2] = q;
504 // Row 2
505 mat[3] = r;
506 mat[4] = 0;
507 mat[5] = -p;
508 // Row 3
509 mat[6] = -q;
510 mat[7] = p;
511 mat[8] = 0;
512}

515 * This will construct the quaternion omega matrix
516 * W(4,4)
517 * p, q, r (rad/sec)
518 */
519void quatW(const ixEuler euler, ixMatrix4 mat)
520{

523 f_t r = euler[2] * (f_t)0.5;
524

526 mat[0] = 0;
527 mat[1] = -p;
528 mat[2] = -q;
529 mat[3] = -r;
530 // Row 2
531 mat[4] = p;
532 mat[5] = 0;
533 mat[6] = r;
534 mat[7] = -q;
535 // Row 3
536 mat[8] = q;
537 mat[9] = -r;
538 mat[10] = 0;
539 mat[11] = p;
540 // Row 4
541 mat[12] = r;
542 mat[13] = q;
543 mat[14] = -p;
544 mat[15] = 0;
545}

548/*
549* Convert quaternion to rotation axis (and angle). Quaternion must be normalized.
550*/
551void quatRotAxis(const ixQuat q, ixVector3 pqr)
552{
553 // Normalize quaternion
554// mul_Vec4_X( q, q, 1/mag_Vec4(q) );
555
556// f_t theta = _ACOS( q[0] ) * (f_t)2.0;
557 f_t sin_a, d;
558
559 sin_a = _SQRT( (f_t)1.0 - q[0] * q[0] );

572/*
573 * Compute the derivative of the ixEuler angle psi with respect
574 * to the quaternion Q. The result is a row vector
575 *
576 * d(psi)/d(q0)
577 * d(psi)/d(q1)
578 * d(psi)/d(q2)
579 * d(psi)/d(q3)
580 */
581void dpsi_dq(const ixQuat q, ixQuat dq)
582{
583 f_t t1 = 1 - 2 * (q[2]*q[2] + q[3]*q[2]);
584 f_t t2 = 2 * (q[1]*q[2] + q[0]*q[3]);
585 f_t err = 2 / ( t1*t1 + t2*t2 );
586
587 dq[0] = err * (q[3]*t1);
588 dq[1] = err * (q[2]*t1);
589 dq[2] = err * (q[1]*t1 + 2 * q[2]*t2);
590 dq[3] = err * (q[0]*t1 + 2 * q[3]*t2);
591}
